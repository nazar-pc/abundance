14/// Getter for single disk plotted sectors
15#[derive(Debug)]
16pub struct SingleDiskPlottedSectors {
17    pub(super) public_key_hash: Blake3Hash,
18    pub(super) shard_commitments_roots_cache: ShardCommitmentsRootsCache,
19    pub(super) pieces_in_sector: u16,
20    pub(super) farmer_protocol_info: FarmerProtocolInfo,
21    pub(super) sectors_metadata: Arc<AsyncRwLock<Vec<SectorMetadataChecksummed>>>,
22}
23
24#[async_trait]
25impl PlottedSectors for SingleDiskPlottedSectors {
26    async fn get(
27        &self,
28    ) -> Result<
29        Box<dyn Stream<Item = Result<PlottedSector, FarmError>> + Unpin + Send + '_>,
30        FarmError,
31    > {
32        let sectors_metadata = self.sectors_metadata.read().await.clone();
33        Ok(Box::new(stream::iter(
34            (SectorIndex::ZERO..).zip(sectors_metadata).map(
35                move |(sector_index, sector_metadata)| {
36                    let sector_id = SectorId::new(
37                        &self.public_key_hash,
38                        &self
39                            .shard_commitments_roots_cache
40                            .get(sector_metadata.history_size),
41                        sector_index,
42                        sector_metadata.history_size,
43                    );
44
45                    let mut piece_indexes = Vec::with_capacity(usize::from(self.pieces_in_sector));
46                    (PieceOffset::ZERO..)
47                        .take(usize::from(self.pieces_in_sector))
48                        .map(|piece_offset| {
49                            sector_id.derive_piece_index(
50                                piece_offset,
51                                sector_metadata.history_size,
52                                self.farmer_protocol_info.max_pieces_in_sector,
53                                self.farmer_protocol_info.recent_segments,
54                                self.farmer_protocol_info.recent_history_fraction,
55                            )
56                        })
57                        .collect_into(&mut piece_indexes);
58
59                    Ok(PlottedSector {
60                        sector_id,
61                        sector_index,
62                        sector_metadata,
63                        piece_indexes,
64                    })
65                },
66            ),
67        )))
68    }
69}
