32trait ArrayIndexingPolyfill<T> {
33    /// The same as [`<[T]>::get_unchecked()`]
34    unsafe fn get_unchecked(&self, index: usize) -> &T;
35    /// The same as [`<[T]>::get_unchecked_mut()`]
36    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T;
37}
38
39#[cfg(target_arch = "spirv")]
40impl<const N: usize, T> ArrayIndexingPolyfill<T> for [T; N] {
41    #[inline(always)]
42    unsafe fn get_unchecked(&self, index: usize) -> &T {
43        &self[index]
44    }
45
46    #[inline(always)]
47    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
48        &mut self[index]
49    }
50}

52fn calculate_left_target_on_demand(parity: u32, r: u32, m: u32) -> u32 {
53    let param_b = u32::from(PARAM_B);
54    let param_c = u32::from(PARAM_C);
55
56    ((r / param_c + m) % param_b) * param_c + (((2 * m + parity) * (2 * m + parity) + r) % param_c)
57}

76// TODO: Reuse code from `ab-proof-of-space` after https://github.com/Rust-GPU/rust-gpu/pull/249 and
77//  https://github.com/Rust-GPU/rust-gpu/discussions/301
78/// Returns the number of matches found.
79///
80/// # Safety
81/// Must be called from [`WORKGROUP_SIZE`] threads with `local_invocation_id` corresponding to the
82/// thread index. `num_subgroups` must be at most [`MAX_SUBGROUPS`] and `subgroup_id` must be within
83/// `0..num_subgroups`. All buckets must come from the `sort_buckets_with_rmap_details` shader.
84// TODO: Try to reduce the `matches` size further by processing `left_bucket` in chunks (like halves
85//  for example)
86#[expect(clippy::too_many_arguments, reason = "Function is inlined anyway")]
87#[inline(always)]
88pub(super) unsafe fn find_matches_in_buckets_impl(
89    subgroup_local_invocation_id: u32,
90    subgroup_id: u32,
91    num_subgroups: u32,
92    local_invocation_id: u32,
93    left_bucket_index: u32,
94    // TODO: These should use `REDUCED_BUCKET_SIZE`, but it currently doesn't compile:
95    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
96    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
97    right_bucket: &[PositionR; MAX_BUCKET_SIZE],
98    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
99    scratch_space: &mut SharedScratchSpace,
100    rmap: &mut MaybeUninit<Rmap>,
101) -> u32 {
102    let SharedScratchSpace {
103        bucket_size_a,
104        bucket_size_b,
105        num_subgroups_size_a,
106    } = scratch_space;
107
108    let left_base = left_bucket_index * u32::from(PARAM_BC);
109
110    // Zero-initialize `rmap`
111    let rmap = {
112        const {
113            assert!(size_of::<Rmap>().is_multiple_of(size_of::<u32>()));
114            assert!(align_of::<Rmap>() == align_of::<u32>());
115
116            const fn assert_copy<T: Copy>() {}
117            assert_copy::<Rmap>();
118        }
119        // TODO: Proper parallel version currently doesn't compile, remove zeroing hack once it
120        //  does: https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
121        // // SAFETY: `Rmap` is a simple `Copy` struct for which zero-initialization is valid
122        // let rmap_words = unsafe {
123        //     rmap.as_mut_ptr()
124        //         .cast::<[MaybeUninit<u32>; size_of::<Rmap>() / size_of::<u32>()]>()
125        //         .as_mut_unchecked()
126        // };
127        // for word_index in (0..size_of::<Rmap>())
128        //     .skip(local_invocation_id as usize)
129        //     .step_by(WORKGROUP_SIZE as usize)
130        // {
131        //     // SAFETY: `word_index` is within bounds of `Rmap`
132        //     unsafe {
133        //         rmap_words.get_unchecked_mut(word_index).write(0);
134        //     }
135        // }
136        Rmap::zeroing_hack(rmap, local_invocation_id, WORKGROUP_SIZE);
137
138        workgroup_memory_barrier_with_group_sync();
139
140        // SAFETY: Initialized with zeroes
141        let rmap = unsafe { rmap.assume_init_mut() };
142
143        for index in
144            (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
145        {
146            let PositionR { position, r } = right_bucket[index];
147
148            // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
149            if position == Position::SENTINEL {
150                break;
151            }
152
153            // SAFETY: Guaranteed by function contract
154            unsafe {
155                rmap.add_with_data_parallel(r, position);
156            }
157        }
158
159        workgroup_memory_barrier_with_group_sync();
160
161        rmap
162    };
163
164    // Load both into shared memory and precompute `rmap_bit_positions`. `rmap_bit_position`s for
165    // non-sentinel positions are guaranteed to be initialized
166    let (left_bucket_positions, left_rs) = {
167        let left_bucket_positions =
168            <Position as PositionExt>::uninit_array_from_repr_mut(bucket_size_a);
169        let left_rs = bucket_size_b;
170        // TODO: More idiomatic version currently doesn't compile:
171        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
172        // for ((&left_position, position), r) in left_bucket
173        //     .iter()
174        //     .zip(left_bucket_positions.iter_mut())
175        //     .zip(left_rs.iter_mut())
176        //     .skip(local_invocation_id as usize)
177        //     .step_by(WORKGROUP_SIZE as usize)
178        // {
179        for index in
180            (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
181        {
182            let PositionR { position, r } = left_bucket[index];
183            let (r, _data) = r.split();
184            let left_bucket_position = &mut left_bucket_positions[index];
185            let r_entry = &mut left_rs[index];
186
187            left_bucket_position.write(position);
188
189            // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
190            if position == Position::SENTINEL {
191                break;
192            }
193
194            r_entry.write(r);
195        }
196
197        workgroup_memory_barrier_with_group_sync();
198
199        // TODO: Correct version currently doesn't compile:
200        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
201        // // SAFETY: Just initialized
202        // let left_bucket_positions = unsafe {
203        //     mem::transmute::<
204        //         &mut [MaybeUninit<Position>; REDUCED_BUCKETS_SIZE],
205        //         &mut [Position; REDUCED_BUCKETS_SIZE],
206        //     >(left_bucket_positions)
207        // };
208        //
209        // (&*left_bucket_positions, &*left_rs)
210        (left_bucket_positions, &*left_rs)
211    };
212
213    let parity = left_base % 2;
214
215    const CHUNK_SIZE: usize = WORKGROUP_SIZE as usize / PARAM_M as usize;
216    const {
217        // `CHUNK_SIZE` with `PARAM_M` must cover workgroup exactly
218        assert!(CHUNK_SIZE as u32 * PARAM_M as u32 == WORKGROUP_SIZE);
219        // The bucket size should be possible to iterate in exact chunks
220        assert!(REDUCED_BUCKET_SIZE.is_multiple_of(CHUNK_SIZE));
221    }
222    let shared_subgroup_totals = num_subgroups_size_a;
223    let mut global_match_batch_offset = 0_u32;
224    // TODO: More idiomatic version currently doesn't compile:
225    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
226    // for (&left_positions, left_rs) in left_bucket_positions
227    //     .as_chunks::<CHUNK_SIZE>()
228    //     .0
229    //     .iter()
230    //     .zip(left_rs.as_chunks::<CHUNK_SIZE>().0)
231    // {
232    //     let left_positions = &left_bucket_positions.as_chunks::<CHUNK_SIZE>().0[index];
233    //     let left_rs = &left_rs.as_chunks::<CHUNK_SIZE>().0[index];
234    //
235    //     let index_within_chunk = local_invocation_id as usize / CHUNK_SIZE;
236    //     let left_position = left_positions[index_within_chunk];
237    for chunk_index in 0..left_bucket_positions.len() / CHUNK_SIZE {
238        // First `PARAM_M` invocations in a workgroup process the first chunk index, next
239        // `PARAM_M` process the second chunk index and so on, with each chunk index corresponding
240        // to `PARAM_M` `r_target` values
241        let index_within_chunk = local_invocation_id as usize / PARAM_M as usize;
242        let left_position = unsafe {
243            left_bucket_positions[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init()
244        };
245
246        // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
247        // Check if reached the end of the bucket
248        let ([right_position_a, right_position_b], left_r) = if left_position == Position::SENTINEL
249        {
250            // `left_r` value doesn't matter here, it will not be read/used anyway
251            ([Position::ZERO; _], 0)
252        } else {
253            // TODO: More idiomatic version currently doesn't compile:
254            //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
255            // // SAFETY: `left_position` is not sentinel, hence `left_r` must be initialized
256            // let left_r = unsafe { left_rs[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init() };
257            let left_r =
258                unsafe { left_rs[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init() };
259            let m = local_invocation_id % PARAM_M as u32;
260            let r_target = calculate_left_target_on_demand(parity, left_r, m);
261
262            // SAFETY: Targets are always limited to `PARAM_BC`
263            let positions = unsafe { rmap.get(RmapBitPosition::new(r_target)) };
264
265            (positions, left_r)
266        };
267
268        let local_matches_count = (right_position_a != Position::ZERO) as u32
269            + (right_position_b != Position::ZERO) as u32;
270
271        // Add up the numbers of matches in the subgroup up to the current lane (exclusive)
272        let local_matches_prefix = subgroup_exclusive_i_add(local_matches_count);
273        {
274            // Add up the numbers of matches in the subgroup (total)
275            let subgroup_matches_count = subgroup_i_add(local_matches_count);
276
277            // TODO: should have been `subgroup_elect()`, but it is not implemented in `wgpu` yet:
278            //  https://github.com/gfx-rs/wgpu/issues/5555
279            if subgroup_local_invocation_id == 0 {
280                // SAFETY: Guaranteed by function contract
281                unsafe { shared_subgroup_totals.get_unchecked_mut(subgroup_id as usize) }
282                    .write(subgroup_matches_count);
283            }
284        }
285
286        workgroup_memory_barrier_with_group_sync();
287
288        // Calculate offset for matches written by this subgroup and update global match batch
289        // offset
290        let mut subgroup_matches_offset = global_match_batch_offset;
291        for current_subgroup_id in 0..num_subgroups {
292            // SAFETY: Guaranteed by function contract
293            let subgroup_matches_count = unsafe {
294                shared_subgroup_totals
295                    .get_unchecked(current_subgroup_id as usize)
296                    .assume_init()
297            };
298            if current_subgroup_id < subgroup_id {
299                subgroup_matches_offset += subgroup_matches_count;
300            }
301            global_match_batch_offset += subgroup_matches_count;
302        }
303
304        // Calculate offset where to write local matches into
305        let mut local_matches_offset = subgroup_matches_offset + local_matches_prefix;
306
307        if right_position_a != Position::ZERO {
308            let y = Y::from(left_r + left_base);
309
310            // TODO: More idiomatic version currently doesn't compile:
311            //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
312            // let Some(m) = matches.get_mut(local_matches_offset as usize) else {
313            //     continue;
314            // };
315            if (local_matches_offset as usize) < matches.len() {
316                let m = &mut matches[local_matches_offset as usize];
317
318                m.write(Match {
319                    left_position,
320                    left_y: y,
321                    right_position: right_position_a,
322                });
323
324                local_matches_offset += 1;
325
326                if right_position_b != Position::ZERO {
327                    // TODO: More idiomatic version currently doesn't compile:
328                    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
329                    // let Some(m) = matches.get_mut(local_matches_offset as usize) else {
330                    //     continue;
331                    // };
332                    if (local_matches_offset as usize) < matches.len() {
333                        let m = &mut matches[local_matches_offset as usize];
334
335                        m.write(Match {
336                            left_position,
337                            left_y: y,
338                            right_position: right_position_b,
339                        });
340                    }
341                }
342            }
343        }
344
345        // Make sure workgroup progresses predictably in phases for offsets to work properly
346        control_barrier::<
347            { Scope::Workgroup as u32 },
348            { Scope::Workgroup as u32 },
349            { Semantics::NONE.bits() },
350        >();
351    }
352
353    global_match_batch_offset.min(REDUCED_MATCHES_COUNT as u32)
354}

356/// # Safety
357/// Must be called from [`WORKGROUP_SIZE`] threads. `num_subgroups` must be at most
358/// [`MAX_SUBGROUPS`]. All buckets must come from the `sort_buckets_with_rmap_details` shader.
359#[spirv(compute(threads(256), entry_point_name = "find_matches_in_buckets"))]
360#[expect(
361    clippy::too_many_arguments,
362    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
363)]
364pub unsafe fn find_matches_in_buckets(
365    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
366    #[spirv(workgroup_id)] workgroup_id: UVec3,
367    #[spirv(num_workgroups)] num_workgroups: UVec3,
368    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
369    #[spirv(subgroup_id)] subgroup_id: u32,
370    #[spirv(num_subgroups)] num_subgroups: u32,
371    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] buckets: &[[PositionR;
372          MAX_BUCKET_SIZE]],
373    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)]
374    matches: &mut [[MaybeUninit<Match>; REDUCED_MATCHES_COUNT]],
375    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] matches_counts: &mut [MaybeUninit<
376        u32,
377    >],
378    #[spirv(workgroup)] scratch_space: &mut SharedScratchSpace,
379    // Non-modern GPUs do not have enough space in the shared memory
380    #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
381    #[spirv(workgroup)]
382    rmap: &mut MaybeUninit<Rmap>,
383    #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
384    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)]
385    rmap: &mut [MaybeUninit<Rmap>; MAX_SUBGROUPS],
386) {
387    let local_invocation_id = local_invocation_id.x;
388    let workgroup_id = workgroup_id.x;
389    let num_workgroups = num_workgroups.x;
390
391    // TODO: More idiomatic version currently doesn't compile:
392    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
393    // for (left_bucket_index, (([left_bucket, right_bucket], matches), matches_count)) in buckets
394    //     .array_windows::<2>()
395    //     .zip(matches)
396    //     .zip(matches_counts)
397    //     .enumerate()
398    //     .skip(workgroup_id as usize)
399    //     .step_by(num_workgroups as usize)
400    // {
401    for left_bucket_index in
402        (workgroup_id as usize..buckets.len() - 1).step_by(num_workgroups as usize)
403    {
404        let left_bucket = &buckets[left_bucket_index];
405        let right_bucket = &buckets[left_bucket_index + 1];
406        let matches = &mut matches[left_bucket_index];
407        let matches_count = &mut matches_counts[left_bucket_index];
408
409        // TODO: Truncate buckets to reduced size here once it compiles:
410        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
411        // SAFETY: Guaranteed by function contract
412        matches_count.write(unsafe {
413            find_matches_in_buckets_impl(
414                subgroup_local_invocation_id,
415                subgroup_id,
416                num_subgroups,
417                local_invocation_id,
418                left_bucket_index as u32,
419                left_bucket,
420                right_bucket,
421                matches,
422                scratch_space,
423                #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
424                rmap,
425                #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
426                &mut rmap[subgroup_id as usize],
427            )
428        });
429    }
