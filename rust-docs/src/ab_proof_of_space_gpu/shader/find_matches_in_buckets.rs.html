33trait ArrayIndexingPolyfill<T> {
34    /// The same as [`<[T]>::get_unchecked()`]
35    unsafe fn get_unchecked(&self, index: usize) -> &T;
36    /// The same as [`<[T]>::get_unchecked_mut()`]
37    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T;
38}

40#[cfg(target_arch = "spirv")]
41impl<const N: usize, T> ArrayIndexingPolyfill<T> for [T; N] {
42    #[inline(always)]
43    unsafe fn get_unchecked(&self, index: usize) -> &T {
44        &self[index]
45    }
46
47    #[inline(always)]
48    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
49        &mut self[index]
50    }
51}

53fn calculate_left_target_on_demand(parity: u32, r: u32, m: u32) -> u32 {
54    let param_b = u32::from(PARAM_B);
55    let param_c = u32::from(PARAM_C);
56
57    ((r / param_c + m) % param_b) * param_c + (((2 * m + parity) * (2 * m + parity) + r) % param_c)
58}

79/// Returns the number of matches found.
80///
81/// # Safety
82/// Must be called from [`WORKGROUP_SIZE`] threads with `local_invocation_id` corresponding to the
83/// thread index. `num_subgroups` must be at most [`MAX_SUBGROUPS`] and `subgroup_id` must be within
84/// `0..num_subgroups`.
85// TODO: Try to reduce the `matches` size further by processing `left_bucket` in chunks (like halves
86//  for example)
87#[expect(clippy::too_many_arguments, reason = "Function is inlined anyway")]
88#[inline(always)]
89pub(super) unsafe fn find_matches_in_buckets_impl(
90    subgroup_id: u32,
91    num_subgroups: u32,
92    local_invocation_id: u32,
93    left_bucket_index: u32,
94    // TODO: These should use `REDUCED_BUCKET_SIZE`, but it currently doesn't compile:
95    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
96    left_bucket: &[PositionY; MAX_BUCKET_SIZE],
97    right_bucket: &[PositionY; MAX_BUCKET_SIZE],
98    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
99    scratch_space: &mut SharedScratchSpace,
100    rmap: &mut MaybeUninit<Rmap>,
101) -> u32 {
102    let SharedScratchSpace {
103        bucket_size_a,
104        bucket_size_b,
105        num_subgroups_size_a,
106    } = scratch_space;
107
108    let left_base = left_bucket_index * u32::from(PARAM_BC);
109    let right_base = left_base + u32::from(PARAM_BC);
110
111    // Zero-initialize `rmap`
112    let rmap = {
113        const {
114            assert!(size_of::<Rmap>().is_multiple_of(size_of::<u32>()));
115            assert!(align_of::<Rmap>() == align_of::<u32>());
116
117            const fn assert_copy<T: Copy>() {}
118            assert_copy::<Rmap>();
119        }
120        // TODO: Proper parallel version currently doesn't compile, remove zeroing hack once it
121        //  does: https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
122        // // SAFETY: `Rmap` is a simple `Copy` struct for which zero-initialization is valid
123        // let rmap_words = unsafe {
124        //     rmap.as_mut_ptr()
125        //         .cast::<[MaybeUninit<u32>; size_of::<Rmap>() / size_of::<u32>()]>()
126        //         .as_mut_unchecked()
127        // };
128        // for word_index in (0..size_of::<Rmap>())
129        //     .skip(local_invocation_id as usize)
130        //     .step_by(WORKGROUP_SIZE as usize)
131        // {
132        //     // SAFETY: `word_index` is within bounds of `Rmap`
133        //     unsafe {
134        //         rmap_words.get_unchecked_mut(word_index).write(0);
135        //     }
136        // }
137        Rmap::zeroing_hack(rmap, local_invocation_id, WORKGROUP_SIZE);
138
139        // No barrier here, but `rmap` is not used until after the barrier below (as part of reading
140        // right bucket), so it will be synchronized there together
141
142        // SAFETY: Initialized with zeroes
143        unsafe { rmap.assume_init_mut() }
144    };
145
146    // Load both into shared memory and precompute `rmap_bit_positions`. `rmap_bit_position`s for
147    // non-sentinel positions are guaranteed to be initialized
148    let (right_bucket_positions, right_rmap_bit_positions) = {
149        let right_bucket_positions =
150            <Position as PositionExt>::uninit_array_from_repr_mut(bucket_size_a);
151        let rmap_bit_positions =
152            <RmapBitPosition as RmapBitPositionExt>::uninit_array_from_repr_mut(bucket_size_b);
153        // TODO: More idiomatic version currently doesn't compile:
154        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
155        // for ((&right_position, position), rmap_bit_position) in right_bucket
156        //     .iter()
157        //     .zip(right_bucket_positions.iter_mut())
158        //     .zip(rmap_bit_positions.iter_mut())
159        //     .skip(local_invocation_id as usize)
160        //     .step_by(WORKGROUP_SIZE as usize)
161        // {
162        for index in
163            (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
164        {
165            let PositionY { position, y } = right_bucket[index];
166            let right_bucket_position = &mut right_bucket_positions[index];
167            let rmap_bit_position = &mut rmap_bit_positions[index];
168
169            right_bucket_position.write(position);
170
171            // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
172            if position == Position::SENTINEL {
173                break;
174            }
175
176            let r = u32::from(y) - right_base;
177
178            // SAFETY: `r` is within `0..PARAM_BC` range by definition
179            rmap_bit_position.write(unsafe { RmapBitPosition::new(r) });
180        }
181
182        workgroup_memory_barrier_with_group_sync();
183
184        // TODO: Correct version currently doesn't compile:
185        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
186        // // SAFETY: Just initialized
187        // let right_bucket_positions = unsafe {
188        //     mem::transmute::<
189        //         &mut [MaybeUninit<Position>; REDUCED_BUCKETS_SIZE],
190        //         &mut [Position; REDUCED_BUCKETS_SIZE],
191        //     >(right_bucket_positions)
192        // };
193        //
194        // (&*right_bucket_positions, &*rmap_bit_positions)
195        (right_bucket_positions, &*rmap_bit_positions)
196    };
197
198    // Fill `rmap`
199    // TODO: Try to parallelize this part
200    if local_invocation_id == 0 {
201        let mut next_physical_pointer = NextPhysicalPointer::default();
202
203        // TODO: More idiomatic version currently doesn't compile:
204        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
205        // for (&right_position, rmap_bit_position) in
206        //     right_bucket_positions.iter().zip(right_rmap_bit_positions)
207        // {
208        for index in 0..right_bucket_positions.len() {
209            let right_position = unsafe { right_bucket_positions[index].assume_init() };
210            let rmap_bit_position = right_rmap_bit_positions[index];
211
212            // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
213            if right_position == Position::SENTINEL {
214                break;
215            }
216
217            // SAFETY: `rmap_bit_position` for non-sentinel positions were initialized above
218            let rmap_bit_position = unsafe { rmap_bit_position.assume_init() };
219
220            // SAFETY: The right bucket is limited to `REDUCED_BUCKETS_SIZE`, same
221            // `next_physical_pointer` is used here exclusively
222            unsafe {
223                rmap.add(
224                    rmap_bit_position,
225                    right_position,
226                    &mut next_physical_pointer,
227                );
228            }
229        }
230    }
231
232    workgroup_memory_barrier_with_group_sync();
233
234    // Load both into shared memory and precompute `rmap_bit_positions`. `rmap_bit_position`s for
235    // non-sentinel positions are guaranteed to be initialized
236    let (left_bucket_positions, left_rs) = {
237        let left_bucket_positions =
238            <Position as PositionExt>::uninit_array_from_repr_mut(bucket_size_a);
239        let left_rs = bucket_size_b;
240        // TODO: More idiomatic version currently doesn't compile:
241        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
242        // for ((&left_position, position), r) in left_bucket
243        //     .iter()
244        //     .zip(left_bucket_positions.iter_mut())
245        //     .zip(left_rs.iter_mut())
246        //     .skip(local_invocation_id as usize)
247        //     .step_by(WORKGROUP_SIZE as usize)
248        // {
249        for index in
250            (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
251        {
252            let PositionY { position, y } = left_bucket[index];
253            let left_bucket_position = &mut left_bucket_positions[index];
254            let r = &mut left_rs[index];
255
256            left_bucket_position.write(position);
257
258            // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
259            if position == Position::SENTINEL {
260                break;
261            }
262
263            r.write(u32::from(y) - left_base);
264        }
265
266        workgroup_memory_barrier_with_group_sync();
267
268        // TODO: Correct version currently doesn't compile:
269        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
270        // // SAFETY: Just initialized
271        // let left_bucket_positions = unsafe {
272        //     mem::transmute::<
273        //         &mut [MaybeUninit<Position>; REDUCED_BUCKETS_SIZE],
274        //         &mut [Position; REDUCED_BUCKETS_SIZE],
275        //     >(left_bucket_positions)
276        // };
277        //
278        // (&*left_bucket_positions, &*left_rs)
279        (left_bucket_positions, &*left_rs)
280    };
281
282    let parity = left_base % 2;
283
284    const CHUNK_SIZE: usize = WORKGROUP_SIZE as usize / PARAM_M as usize;
285    const {
286        // `CHUNK_SIZE` with `PARAM_M` must cover workgroup exactly
287        assert!(CHUNK_SIZE as u32 * PARAM_M as u32 == WORKGROUP_SIZE);
288        // The bucket size should be possible to iterate in exact chunks
289        assert!(REDUCED_BUCKET_SIZE.is_multiple_of(CHUNK_SIZE));
290    }
291    let shared_subgroup_totals = num_subgroups_size_a;
292    let mut global_match_batch_offset = 0_u32;
293    // TODO: More idiomatic version currently doesn't compile:
294    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
295    // for (&left_positions, left_rs) in left_bucket_positions
296    //     .as_chunks::<CHUNK_SIZE>()
297    //     .0
298    //     .iter()
299    //     .zip(left_rs.as_chunks::<CHUNK_SIZE>().0)
300    // {
301    //     let left_positions = &left_bucket_positions.as_chunks::<CHUNK_SIZE>().0[index];
302    //     let left_rs = &left_rs.as_chunks::<CHUNK_SIZE>().0[index];
303    //
304    //     let index_within_chunk = local_invocation_id as usize / CHUNK_SIZE;
305    //     let left_position = left_positions[index_within_chunk];
306    for chunk_index in 0..left_bucket_positions.len() / CHUNK_SIZE {
307        // First `PARAM_M` invocations in a workgroup process the first chunk index, next
308        // `PARAM_M` process the second chunk index and so on, with each chunk index corresponding
309        // to `PARAM_M` `r_target` values
310        let index_within_chunk = local_invocation_id as usize / PARAM_M as usize;
311        let left_position = unsafe {
312            left_bucket_positions[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init()
313        };
314
315        // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
316        // Check if reached the end of the bucket
317        let ([right_position_a, right_position_b], left_r) = if left_position == Position::SENTINEL
318        {
319            // `left_r` value doesn't matter here, it will not be read/used anyway
320            ([Position::ZERO; _], 0)
321        } else {
322            // TODO: More idiomatic version currently doesn't compile:
323            //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
324            // // SAFETY: `left_position` is not sentinel, hence `left_r` must be initialized
325            // let left_r = unsafe { left_rs[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init() };
326            let left_r =
327                unsafe { left_rs[chunk_index * CHUNK_SIZE + index_within_chunk].assume_init() };
328            let m = local_invocation_id % PARAM_M as u32;
329            let r_target = calculate_left_target_on_demand(parity, left_r, m);
330
331            // SAFETY: Targets are always limited to `PARAM_BC`
332            let positions = unsafe { rmap.get(RmapBitPosition::new(r_target)) };
333
334            (positions, left_r)
335        };
336
337        let local_matches_count = (right_position_a != Position::ZERO) as u32
338            + (right_position_b != Position::ZERO) as u32;
339
340        // Add up the numbers of matches in the subgroup up to the current lane (exclusive)
341        let local_matches_prefix = subgroup_exclusive_i_add(local_matches_count);
342        {
343            // Add up the numbers of matches in the subgroup (total)
344            let subgroup_matches_count = subgroup_i_add(local_matches_count);
345
346            // SAFETY: Guaranteed by function contract
347            unsafe { shared_subgroup_totals.get_unchecked_mut(subgroup_id as usize) }
348                .write(subgroup_matches_count);
349        }
350
351        workgroup_memory_barrier_with_group_sync();
352
353        // Calculate offset for matches written by this subgroup and update global match batch
354        // offset
355        let mut subgroup_matches_offset = global_match_batch_offset;
356        for current_subgroup_id in 0..num_subgroups {
357            // SAFETY: Guaranteed by function contract
358            let subgroup_matches_count = unsafe {
359                shared_subgroup_totals
360                    .get_unchecked(current_subgroup_id as usize)
361                    .assume_init()
362            };
363            if current_subgroup_id < subgroup_id {
364                subgroup_matches_offset += subgroup_matches_count;
365            }
366            global_match_batch_offset += subgroup_matches_count;
367        }
368
369        // Calculate offset where to write local matches into
370        let mut local_matches_offset = subgroup_matches_offset + local_matches_prefix;
371
372        if right_position_a != Position::ZERO {
373            let y = Y::from(left_r + left_base);
374
375            // TODO: More idiomatic version currently doesn't compile:
376            //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
377            // let Some(m) = matches.get_mut(local_matches_offset as usize) else {
378            //     continue;
379            // };
380            if (local_matches_offset as usize) >= matches.len() {
381                continue;
382            }
383            let m = &mut matches[local_matches_offset as usize];
384
385            m.write(Match {
386                left_position,
387                left_y: y,
388                right_position: right_position_a,
389            });
390
391            local_matches_offset += 1;
392
393            if right_position_b != Position::ZERO {
394                // TODO: More idiomatic version currently doesn't compile:
395                //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
396                // let Some(m) = matches.get_mut(local_matches_offset as usize) else {
397                //     continue;
398                // };
399                if (local_matches_offset as usize) >= matches.len() {
400                    continue;
401                }
402                let m = &mut matches[local_matches_offset as usize];
403
404                m.write(Match {
405                    left_position,
406                    left_y: y,
407                    right_position: right_position_b,
408                });
409            }
410        }
411
412        // Make sure workgroup progresses predictably in phases for offsets to work properly
413        control_barrier::<
414            { Scope::Workgroup as u32 },
415            { Scope::Workgroup as u32 },
416            { Semantics::NONE.bits() },
417        >();
418    }
419
420    global_match_batch_offset.min(REDUCED_MATCHES_COUNT as u32)
421}

423/// # Safety
424/// Must be called from [`WORKGROUP_SIZE`] threads. `num_subgroups` must be at most
425/// [`MAX_SUBGROUPS`].
426#[spirv(compute(threads(256), entry_point_name = "find_matches_in_buckets"))]
427#[expect(
428    clippy::too_many_arguments,
429    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
430)]
431pub unsafe fn find_matches_in_buckets(
432    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
433    #[spirv(workgroup_id)] workgroup_id: UVec3,
434    #[spirv(num_workgroups)] num_workgroups: UVec3,
435    #[spirv(subgroup_id)] subgroup_id: u32,
436    #[spirv(num_subgroups)] num_subgroups: u32,
437    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] buckets: &[[PositionY;
438          MAX_BUCKET_SIZE]],
439    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)]
440    matches: &mut [[MaybeUninit<Match>; REDUCED_MATCHES_COUNT]],
441    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] matches_counts: &mut [MaybeUninit<
442        u32,
443    >],
444    #[spirv(workgroup)] scratch_space: &mut SharedScratchSpace,
445    // Non-modern GPUs do not have enough space in the shared memory
446    #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
447    #[spirv(workgroup)]
448    rmap: &mut MaybeUninit<Rmap>,
449    #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
450    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)]
451    rmap: &mut [MaybeUninit<Rmap>; MAX_SUBGROUPS],
452) {
453    let local_invocation_id = local_invocation_id.x;
454    let workgroup_id = workgroup_id.x;
455    let num_workgroups = num_workgroups.x;
456
457    // TODO: More idiomatic version currently doesn't compile:
458    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
459    // for (left_bucket_index, (([left_bucket, right_bucket], matches), matches_count)) in buckets
460    //     .array_windows::<2>()
461    //     .zip(matches)
462    //     .zip(matches_counts)
463    //     .enumerate()
464    //     .skip(workgroup_id as usize)
465    //     .step_by(num_workgroups as usize)
466    // {
467    for left_bucket_index in
468        (workgroup_id as usize..buckets.len() - 1).step_by(num_workgroups as usize)
469    {
470        let left_bucket = &buckets[left_bucket_index];
471        let right_bucket = &buckets[left_bucket_index + 1];
472        let matches = &mut matches[left_bucket_index];
473        let matches_count = &mut matches_counts[left_bucket_index];
474
475        // TODO: Truncate buckets to reduced size here once it compiles:
476        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
477        // SAFETY: Guaranteed by function contract
478        matches_count.write(unsafe {
479            find_matches_in_buckets_impl(
480                subgroup_id,
481                num_subgroups,
482                local_invocation_id,
483                left_bucket_index as u32,
484                left_bucket,
485                right_bucket,
486                matches,
487                scratch_space,
488                #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
489                rmap,
490                #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
491                &mut rmap[subgroup_id as usize],
492            )
493        });
494    }
