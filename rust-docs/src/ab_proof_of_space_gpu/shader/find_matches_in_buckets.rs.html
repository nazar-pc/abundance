20//  https://github.com/Rust-GPU/rust-gpu/discussions/287 is resolved
21pub const WORKGROUP_SIZE: u32 = 256;

23fn calculate_left_target_on_demand(parity: u32, r: u32, m: u32) -> u32 {
24    let param_b = u32::from(PARAM_B);
25    let param_c = u32::from(PARAM_C);
26
27    ((r / param_c + m) % param_b) * param_c + (((2 * m + parity) * (2 * m + parity) + r) % param_c)
28}

36// TODO: Reuse code from `ab-proof-of-space` after https://github.com/Rust-GPU/rust-gpu/pull/249 and
37//  https://github.com/Rust-GPU/rust-gpu/discussions/301
38/// Returns the number of matches found.
39///
40/// # Safety
41/// Must be called from [`WORKGROUP_SIZE`] threads with `local_invocation_id` corresponding to the
42/// thread index. All buckets must contain valid positions.
43// TODO: Try to reduce the `matches` size further by processing `left_bucket` in chunks (like halves
44//  for example)
45#[inline(always)]
46pub(super) unsafe fn find_matches_in_buckets_impl(
47    local_invocation_id: u32,
48    left_bucket_index: u32,
49    // TODO: These should use `REDUCED_BUCKET_SIZE`, but it currently doesn't compile:
50    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
51    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
52    right_bucket: &[PositionR; MAX_BUCKET_SIZE],
53    matches: &mut [MaybeUninit<Match>; MAX_BUCKET_SIZE],
54    shared: &mut FindMatchesShared,
55) -> u32 {
56    let FindMatchesShared {
57        rmap,
58        matches_counter,
59    } = shared;
60
61    for index in
62        (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
63    {
64        let PositionR { position, r } = right_bucket[index];
65
66        // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
67        if position == Position::SENTINEL {
68            break;
69        }
70
71        rmap.add_with_data_parallel(r);
72    }
73
74    workgroup_memory_barrier_with_group_sync();
75
76    let parity = left_bucket_index % 2;
77
78    const CHUNK_SIZE: usize = WORKGROUP_SIZE as usize / PARAM_M as usize;
79    const {
80        // `CHUNK_SIZE` with `PARAM_M` must cover workgroup exactly
81        assert!(CHUNK_SIZE as u32 * PARAM_M as u32 == WORKGROUP_SIZE);
82        // The bucket size should be possible to iterate in exact chunks
83        assert!(REDUCED_BUCKET_SIZE.is_multiple_of(CHUNK_SIZE));
84    }
85    // TODO: More idiomatic version currently doesn't compile:
86    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
87    for chunk_index in 0..REDUCED_BUCKET_SIZE / CHUNK_SIZE {
88        // First `PARAM_M` invocations in a workgroup process the first chunk index, next
89        // `PARAM_M` process the second chunk index and so on, with each chunk index corresponding
90        // to `PARAM_M` `r_target` values
91        let index_within_chunk = local_invocation_id as usize / PARAM_M as usize;
92        let bucket_offset = chunk_index * CHUNK_SIZE + index_within_chunk;
93        let PositionR { position, r } = left_bucket[bucket_offset];
94        let left_r = r.get();
95
96        // TODO: Wouldn't it make more sense to check the size here instead of sentinel?
97        // Check if reached the end of the bucket
98        let (m, local_matches_count) = if position == Position::SENTINEL {
99            (Match::SENTINEL, 0)
100        } else {
101            let m = local_invocation_id % PARAM_M as u32;
102            let r_target = calculate_left_target_on_demand(parity, left_r, m);
103
104            // SAFETY: Right targets are guaranteed to be within `0..PARAM_BC` range
105            let local_matches_count = rmap.num_r_items(unsafe { R::new(r_target) });
106
107            let m = if local_matches_count == 0 {
108                Match::SENTINEL
109            } else {
110                // SAFETY: `bucket_offset` is guaranteed to be within `0..MAX_BUCKET_SIZE` range,
111                // `m` is guaranteed to be within `0..PARAM_M` range, `r_target` is guaranteed to be
112                // within `0..PARAM_BC` range
113
114                unsafe { Match::new(bucket_offset as u32, m, r_target) }
115            };
116
117            (m, local_matches_count)
118        };
119
120        if local_matches_count >= 1 {
121            // SAFETY: TODO: Probably should not be unsafe to begin with:
122            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
123            let local_matches_offset = unsafe {
124                atomic_i_add::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
125                    matches_counter,
126                    local_matches_count,
127                )
128            };
129            matches[local_matches_offset as usize].write(m);
130
131            if local_matches_count == 2 {
132                matches[local_matches_offset as usize + 1].write(m.second_second_position());
133            }
134        }
135    }
136
137    workgroup_memory_barrier_with_group_sync();
138
139    let matches_counter = *matches_counter;
140
141    for index in ((matches_counter + local_invocation_id) as usize..MAX_BUCKET_SIZE)
142        .step_by(WORKGROUP_SIZE as usize)
143    {
144        matches[index].write(Match::SENTINEL);
145    }
146
147    workgroup_memory_barrier_with_group_sync();
148
149    sort_shared_bucket(local_invocation_id, matches, |a, b| {
150        // SAFETY: Initialized above
151        unsafe { a.assume_init() }.cmp_key() <= unsafe { b.assume_init() }.cmp_key()
152    });
153
154    matches_counter.min(REDUCED_MATCHES_COUNT as u32)
155}

157/// # Safety
158/// Must be called from [`WORKGROUP_SIZE`] threads. All buckets must contain valid positions.
159#[spirv(compute(threads(256), entry_point_name = "find_matches_in_buckets"))]
160pub unsafe fn find_matches_in_buckets(
161    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
162    #[spirv(workgroup_id)] workgroup_id: UVec3,
163    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] buckets: &[[PositionR;
164          MAX_BUCKET_SIZE]],
165    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] matches: &mut [[MaybeUninit<Match>;
166              MAX_BUCKET_SIZE]],
167    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] matches_counts: &mut [MaybeUninit<
168        u32,
169    >],
170    #[spirv(workgroup)] shared: &mut FindMatchesShared,
171) {
172    let local_invocation_id = local_invocation_id.x;
173    let workgroup_id = workgroup_id.x;
174
175    let left_bucket_index = workgroup_id as usize;
176    if left_bucket_index >= buckets.len() - 1 {
177        return;
178    }
179
180    let left_bucket = &buckets[left_bucket_index];
181    let right_bucket = &buckets[left_bucket_index + 1];
182    let matches = &mut matches[left_bucket_index];
183    let matches_count = &mut matches_counts[left_bucket_index];
184
185    // TODO: Truncate buckets to reduced size here once it compiles:
186    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
187    // SAFETY: Guaranteed by function contract
188    matches_count.write(unsafe {
189        find_matches_in_buckets_impl(
190            local_invocation_id,
191            left_bucket_index as u32,
192            left_bucket,
193            right_bucket,
194            matches,
195            shared,
196        )
197    });
