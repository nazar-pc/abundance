28/// PRNG extension parameter to avoid collisions

38/// Compute the size of `y` in bits
39pub(super) const fn y_size_bits(k: u8) -> usize {
40    k as usize + PARAM_EXT as usize
41}
42
43/// Number of buckets for a given `k`
44const fn num_buckets(k: u8) -> usize {
45    2_usize
46        .pow(y_size_bits(k) as u32)
47        .div_ceil(PARAM_BC as usize)
48}
49
50pub const NUM_BUCKETS: usize = num_buckets(K);
51// Buckets are matched with a sliding window of `2`, hence one less bucket exists
52pub const NUM_MATCH_BUCKETS: usize = NUM_BUCKETS - 1;
