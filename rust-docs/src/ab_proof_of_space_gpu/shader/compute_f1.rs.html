16const fn gcd(a: u32, b: u32) -> u32 {
17    let mut x = a;
18    let mut y = b;
19    while y != 0 {
20        let t = y;
21        y = x % y;
22        x = t;
23    }
24    x
25}

27const fn lcm(a: u32, b: u32) -> u32 {
28    let g = gcd(a, b);
29    (a / g) * b
30}

34pub const WORKGROUP_SIZE: u32 = 256;
35const CHACHA8_BLOCK_BITS: u32 = size_of::<ChaCha8Block>() as u32 * u8::BITS;
36const CHACHA8_BLOCK_WORDS: usize = size_of::<ChaCha8Block>() / size_of::<u32>();
37// This number is both a multiple of `K` (bits per element) and ChaCha8 block (bits per block)
38const BITS_PER_INVOCATION: u32 = lcm(CHACHA8_BLOCK_BITS, K as u32);
39pub const ELEMENTS_PER_INVOCATION: u32 = BITS_PER_INVOCATION / K as u32;
40const BLOCKS_PER_INVOCATION: u32 = BITS_PER_INVOCATION / CHACHA8_BLOCK_BITS;
41// `+1` is needed due to the way `compute_fn_impl` does slightly outside what it, strictly speaking,
42// needs (for efficiency purposes)
43const INVOCATION_KEYSTREAM_WORDS: usize = BLOCKS_PER_INVOCATION as usize * CHACHA8_BLOCK_WORDS + 1;

120// TODO: Make unsafe and avoid bounds check
121// TODO: Reuse code from `ab-proof-of-space` after https://github.com/Rust-GPU/rust-gpu/pull/249 and
122//  https://github.com/Rust-GPU/rust-gpu/discussions/301
123/// `partial_y_offset` is in bits within `partial_y`
124#[inline(always)]
125fn compute_f1_impl(x: X, chacha8_keystream: &[u32; INVOCATION_KEYSTREAM_WORDS]) -> Y {
126    let skip_bits = (u32::from(K) * u32::from(x)) % BITS_PER_INVOCATION;
127    let skip_u32s = skip_bits / u32::BITS;
128    let partial_y_offset = skip_bits % u32::BITS;
129
130    let high = chacha8_keystream[skip_u32s as usize].to_be();
131    let low = chacha8_keystream[skip_u32s as usize + 1].to_be();
132    let partial_y = U64::from_low_high(low, high);
133
134    let pre_y = partial_y >> (u64::BITS - u32::from(K + PARAM_EXT) - partial_y_offset);
135    let pre_y = pre_y.as_u32();
136    // Mask for clearing the rest of bits of `pre_y`.
137    let pre_y_mask = (u32::MAX << PARAM_EXT) & (u32::MAX >> (u32::BITS - u32::from(K + PARAM_EXT)));
138
139    // Extract `PARAM_EXT` most significant bits from `x` and store in the final offset of
140    // eventual `y` with the rest of bits being zero (`x` is `0..2^K`)
141    let pre_ext = u32::from(x) >> (K - PARAM_EXT);
142
143    // Combine all of the bits together:
144    // [padding zero bits][`K` bits from `partial_y`][`PARAM_EXT` bits from `x`]
145    Y::from((pre_y & pre_y_mask) | pre_ext)
146}

148/// Compute Chia's `f1()` function for the whole table using the initial state of ChaCha8 cipher
149/// straight into buckets of the first table.
150///
151/// Buckets need to be sorted by position afterward due to concurrent writes that do not have
152/// deterministic order. Content of the bucket beyond the size specified in `bucket_sizes` is
153/// undefined.
154///
155/// # Safety
156/// `bucket_sizes` must be zero-initialized, which is the case by default in `wgpu`.
157#[spirv(compute(threads(256), entry_point_name = "compute_f1"))]
158pub unsafe fn compute_f1(
159    #[spirv(global_invocation_id)] global_invocation_id: UVec3,
160    #[spirv(num_workgroups)] num_workgroups: UVec3,
161    #[spirv(uniform, descriptor_set = 0, binding = 0)] initial_state: &UniformChaCha8Block,
162    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] bucket_sizes: &mut [u32; NUM_BUCKETS],
163    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE];
164             NUM_BUCKETS],
165) {
166    // TODO: Make a single input bounds check and use unsafe to avoid bounds check later
167    let global_invocation_id = global_invocation_id.x;
168    let num_workgroups = num_workgroups.x;
169
170    let global_size = WORKGROUP_SIZE * num_workgroups;
171    let initial_state = ChaCha8State::from_repr(ChaCha8Block::from(*initial_state));
172
173    // TODO: More idiomatic version currently doesn't compile:
174    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
175    for x_start in (global_invocation_id * ELEMENTS_PER_INVOCATION..MAX_TABLE_SIZE)
176        .step_by(global_size as usize * ELEMENTS_PER_INVOCATION as usize)
177    {
178        // TODO: More idiomatic version currently doesn't compile:
179        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
180        // let chacha8_keystream = [MaybeUninit<ChaCha8Block>; BLOCKS_PER_INVOCATION as usize];
181        let mut chacha8_keystream = [0u32; INVOCATION_KEYSTREAM_WORDS];
182        // TODO: More idiomatic version currently doesn't compile:
183        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
184        let first_block_counter = x_start * K as u32 / CHACHA8_BLOCK_BITS;
185        for block_index in 0..BLOCKS_PER_INVOCATION {
186            let counter = first_block_counter + block_index;
187
188            let block = initial_state.compute_block(counter);
189            // TODO: More idiomatic version currently doesn't compile:
190            //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
191            let block_write_offset = block_index as usize * CHACHA8_BLOCK_WORDS;
192            #[allow(clippy::manual_memcpy, reason = "Doesn't compile under rust-gpu")]
193            for offset in 0..CHACHA8_BLOCK_WORDS {
194                chacha8_keystream[block_write_offset + offset] = block[offset];
195            }
196        }
197
198        // TODO: More idiomatic version currently doesn't compile:
199        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
200        for x in x_start..(x_start + ELEMENTS_PER_INVOCATION).min(MAX_TABLE_SIZE) {
201            let y = compute_f1_impl(X::from(x), &chacha8_keystream);
202
203            let (bucket_index, r) = y.into_bucket_index_and_r();
204            // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
205            let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
206            // TODO: Probably should not be unsafe to begin with:
207            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
208            let bucket_offset = unsafe {
209                atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
210                    bucket_size,
211                )
212            };
213
214            // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
215            // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset` is
216            // also always within bounds.
217            unsafe {
218                buckets
219                    .get_unchecked_mut(bucket_index as usize)
220                    .get_unchecked_mut(bucket_offset as usize)
221            }
222            .write(PositionR {
223                position: Position::from_u32(x),
224                r,
225            });
226        }
227    }
