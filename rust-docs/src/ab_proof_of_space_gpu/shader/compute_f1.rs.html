18const fn gcd(a: u32, b: u32) -> u32 {
19    let mut x = a;
20    let mut y = b;
21    while y != 0 {
22        let t = y;
23        y = x % y;
24        x = t;
25    }
26    x
27}

29const fn lcm(a: u32, b: u32) -> u32 {
30    let g = gcd(a, b);
31    (a / g) * b
32}

106// TODO: Make unsafe and avoid bounds check
107// TODO: Reuse code from `ab-proof-of-space` after https://github.com/Rust-GPU/rust-gpu/pull/249 and
108//  https://github.com/Rust-GPU/rust-gpu/discussions/301
109/// `partial_y_offset` is in bits within `partial_y`
110#[inline(always)]
111fn compute_f1_impl(x: X, chacha8_keystream: &[u32; INVOCATION_KEYSTREAM_WORDS]) -> Y {
112    let skip_bits = (u32::from(K) * u32::from(x)) % BITS_PER_INVOCATION;
113    let skip_u32s = skip_bits / u32::BITS;
114    let partial_y_offset = skip_bits % u32::BITS;
115
116    let high = chacha8_keystream[skip_u32s as usize].to_be();
117    let low = chacha8_keystream[skip_u32s as usize + 1].to_be();
118    let partial_y = U32N::from_low_high(low, high);
119
120    let pre_y = partial_y >> (u64::BITS - u32::from(K + PARAM_EXT) - partial_y_offset);
121    let pre_y = pre_y.as_u32();
122    // Mask for clearing the rest of bits of `pre_y`.
123    let pre_y_mask = (u32::MAX << PARAM_EXT) & (u32::MAX >> (u32::BITS - u32::from(K + PARAM_EXT)));
124
125    // Extract `PARAM_EXT` most significant bits from `x` and store in the final offset of
126    // eventual `y` with the rest of bits being zero (`x` is `0..2^K`)
127    let pre_ext = u32::from(x) >> (K - PARAM_EXT);
128
129    // Combine all of the bits together:
130    // [padding zero bits][`K` bits from `partial_y`][`PARAM_EXT` bits from `x`]
131    Y::from((pre_y & pre_y_mask) | pre_ext)
132}

134/// Compute Chia's `f1()` function for the whole table using the initial state of ChaCha8 cipher
135/// straight into buckets of the first table.
136///
137/// Buckets need to be sorted by position afterward due to concurrent writes that do not have
138/// deterministic order. Content of the bucket beyond the size specified in `bucket_sizes` is
139/// undefined.
140///
141/// # Safety
142/// `bucket_sizes` must be zero-initialized, which is the case by default in `wgpu`.
143#[spirv(compute(threads(256), entry_point_name = "compute_f1"))]
144pub unsafe fn compute_f1(
145    #[spirv(global_invocation_id)] global_invocation_id: UVec3,
146    #[spirv(uniform, descriptor_set = 0, binding = 0)] initial_state: &UniformChaCha8Block,
147    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] bucket_sizes: &mut [u32; NUM_BUCKETS],
148    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE];
149             NUM_BUCKETS],
150) {
151    // TODO: Make a single input bounds check and use unsafe to avoid bounds check later
152    let global_invocation_id = global_invocation_id.x;
153
154    let initial_state = ChaCha8State::from_repr(ChaCha8Block::from(*initial_state));
155
156    let x_start = global_invocation_id * ELEMENTS_PER_INVOCATION;
157
158    // TODO: More idiomatic version currently doesn't compile:
159    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
160    // let chacha8_keystream = [MaybeUninit<ChaCha8Block>; BLOCKS_PER_INVOCATION as usize];
161    let mut chacha8_keystream = [0u32; INVOCATION_KEYSTREAM_WORDS];
162    // TODO: More idiomatic version currently doesn't compile:
163    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
164    let first_block_counter = x_start * K as u32 / CHACHA8_BLOCK_BITS;
165    for block_index in 0..BLOCKS_PER_INVOCATION {
166        let counter = first_block_counter + block_index;
167
168        let block = initial_state.compute_block(counter);
169        // TODO: More idiomatic version currently doesn't compile:
170        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
171        let block_write_offset = block_index as usize * CHACHA8_BLOCK_WORDS;
172        #[expect(clippy::manual_memcpy, reason = "Doesn't compile under rust-gpu")]
173        for offset in 0..CHACHA8_BLOCK_WORDS {
174            chacha8_keystream[block_write_offset + offset] = block[offset];
175        }
176    }
177
178    // TODO: More idiomatic version currently doesn't compile:
179    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
180    for x in x_start..(x_start + ELEMENTS_PER_INVOCATION).min(MAX_TABLE_SIZE) {
181        let y = compute_f1_impl(X::from(x), &chacha8_keystream);
182
183        let (bucket_index, r) = y.into_bucket_index_and_r();
184        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
185        let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
186        // SAFETY: TODO: Probably should not be unsafe to begin with:
187        //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
188        let bucket_offset = unsafe {
189            atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
190                bucket_size,
191            )
192        };
193
194        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
195        // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset` is
196        // also always within bounds.
197        unsafe {
198            buckets
199                .get_unchecked_mut(bucket_index as usize)
200                .get_unchecked_mut(bucket_offset as usize)
201        }
202        .write(PositionR {
203            position: Position::from_u32(x),
204            r,
205        });
206    }
