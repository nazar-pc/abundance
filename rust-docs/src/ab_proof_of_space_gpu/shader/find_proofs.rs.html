25pub const WORKGROUP_SIZE: u32 = 256;
26const PROOF_X_SOURCES: usize = 2_usize.pow(NUM_TABLES as u32 - 1);
27const PROOF_BITS: usize = PROOF_X_SOURCES * K as usize;
28const PROOF_BYTES: usize = PROOF_BITS.div_ceil(u8::BITS as usize);
29pub const PROOF_U32_WORDS: usize = PROOF_BYTES.div_ceil(size_of::<u32>());
30pub const FOUND_PROOFS_U32_WORDS: usize = {
31    assert!(NUM_S_BUCKETS.is_multiple_of(u32::BITS as usize));
32
33    NUM_S_BUCKETS / u32::BITS as usize
34};

74#[inline(always)]
75fn find_local_proof_targets<const SUBGROUP_SIZE: u32>(
76    local_invocation_id: u32,
77    subgroup_id: u32,
78    subgroup_local_invocation_id: u32,
79    positions_group_index: u32,
80    bucket_sizes: &mut [u32; NUM_S_BUCKETS],
81    buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET]; NUM_S_BUCKETS],
82    found_proofs: &mut [MaybeUninit<u32>; FOUND_PROOFS_U32_WORDS],
83    found_proofs_scratch: &mut [MaybeUninit<u32>; (WORKGROUP_SIZE / u32::BITS) as usize],
84) -> [Position; 2] {
85    let local_invocation_id = local_invocation_id as usize;
86    let base = positions_group_index * SUBGROUP_SIZE;
87
88    let mut min = [Position::SENTINEL; 2];
89
90    let local_bucket_size = {
91        let bucket_id = base + subgroup_local_invocation_id;
92
93        let local_bucket_size = bucket_sizes[bucket_id as usize];
94        bucket_sizes[bucket_id as usize] = 0;
95
96        local_bucket_size
97    };
98
99    for local_bucket_id in 0..SUBGROUP_SIZE {
100        let bucket_id = (base + local_bucket_id) as usize;
101        let bucket_size = subgroup_shuffle(local_bucket_size, local_bucket_id);
102        let bucket = &buckets[bucket_id];
103
104        // TODO: Can't use the struct due to this issue in Naga:
105        //  https://github.com/gfx-rs/wgpu/issues/8389#issuecomment-3430788603
106        // let mut local_min = ProofTargets {
107        //     absolute_position: u32::MAX,
108        //     positions: [Position::SENTINEL; 2],
109        // };
110        let mut local_min_absolute_position = [u32::MAX];
111        let mut local_min_positions = [Position::SENTINEL; 2];
112
113        for index in (subgroup_local_invocation_id..bucket_size).step_by(SUBGROUP_SIZE as usize) {
114            let proof_targets = bucket[index as usize];
115            if proof_targets.absolute_position < local_min_absolute_position[0] {
116                local_min_absolute_position[0] = proof_targets.absolute_position;
117                local_min_positions = proof_targets.positions;
118            }
119        }
120
121        let min_absolute_position = subgroup_u_min(local_min_absolute_position[0]);
122        let source_lane_mask =
123            subgroup_ballot(local_min_absolute_position[0] == min_absolute_position);
124        // TODO: This intrinsic is not supported by `wgpu` yet:
125        //  https://github.com/gfx-rs/wgpu/issues/8403
126        // let source_lane = subgroup_ballot_find_lsb(source_lane_mask);
127        let source_lane_mask = source_lane_mask.to_array();
128        let mut source_lane = u32::MAX;
129        for i in 0..SUBGROUP_SIZE.div_ceil(u32::BITS) {
130            let word = source_lane_mask[i as usize];
131            if word != 0 {
132                source_lane = word.trailing_zeros() + i * u32::BITS;
133                break;
134            }
135        }
136        let local_min = subgroup_shuffle(local_min_positions, source_lane);
137
138        if subgroup_local_invocation_id == local_bucket_id {
139            min = local_min;
140        }
141    }
142
143    let has_proof = local_bucket_size > 0;
144    let found_proofs_words = subgroup_ballot(has_proof);
145
146    if SUBGROUP_SIZE >= u32::BITS {
147        // For subgroup sizes that are multiple of `u32` words, results can be written directly into
148        // global memory
149        // TODO: should have been `subgroup_elect()`, but it is not implemented in `wgpu` yet:
150        //  https://github.com/gfx-rs/wgpu/issues/5555
151        if subgroup_local_invocation_id == 0 {
152            let start_word = (base / u32::BITS) as usize;
153            found_proofs[start_word].write(found_proofs_words.x);
154
155            if SUBGROUP_SIZE >= 2 * u32::BITS {
156                found_proofs[start_word + 1].write(found_proofs_words.y);
157
158                if SUBGROUP_SIZE >= 4 * u32::BITS {
159                    found_proofs[start_word + 2].write(found_proofs_words.z);
160                    found_proofs[start_word + 3].write(found_proofs_words.w);
161                }
162            }
163        }
164    } else {
165        if local_invocation_id < found_proofs_scratch.len() {
166            found_proofs_scratch[local_invocation_id].write(0);
167        }
168
169        workgroup_memory_barrier_with_group_sync();
170
171        // For subgroups of smaller sizes aggregate results in shared memory first, then write to
172        // global memory
173        // TODO: should have been `subgroup_elect()`, but it is not implemented in `wgpu` yet:
174        //  https://github.com/gfx-rs/wgpu/issues/5555
175        if subgroup_local_invocation_id == 0 {
176            let local_start_bit = subgroup_id * SUBGROUP_SIZE;
177            let local_word_index = (local_start_bit / u32::BITS) as usize;
178            let local_word_shift = local_start_bit % u32::BITS;
179
180            // SAFETY: Initialized above
181            let found_proofs_word =
182                unsafe { found_proofs_scratch[local_word_index].assume_init_mut() };
183            // SAFETY: TODO: Probably should not be unsafe to begin with:
184            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
185            unsafe {
186                atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
187                    found_proofs_word,
188                    found_proofs_words.x << local_word_shift,
189                );
190            }
191        }
192
193        workgroup_memory_barrier_with_group_sync();
194
195        if local_invocation_id < found_proofs_scratch.len() {
196            let workgroup_base_group_index = positions_group_index - subgroup_id;
197            let workgroup_start_bucket = workgroup_base_group_index * SUBGROUP_SIZE;
198            let global_start_word = (workgroup_start_bucket / u32::BITS) as usize;
199
200            // SAFETY: Initialized above
201            let found_proofs_word =
202                unsafe { found_proofs_scratch[local_invocation_id].assume_init() };
203            found_proofs[global_start_word + local_invocation_id].write(found_proofs_word);
204        }
205    }
206
207    min
208}

215fn find_proofs_impl<const SUBGROUP_SIZE: u32>(
216    local_invocation_id: u32,
217    subgroup_id: u32,
218    subgroup_local_invocation_id: u32,
219    positions_group_index: u32,
220    // TODO: This should have been `&[[[Position; 2]; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`,
221    //  but it currently doesn't compile if flattened:
222    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
223    table_2_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
224    table_3_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
225    table_4_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
226    table_5_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
227    table_6_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
228    bucket_sizes: &mut [u32; NUM_S_BUCKETS],
229    buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET]; NUM_S_BUCKETS],
230    found_proofs: &mut [MaybeUninit<u32>; FOUND_PROOFS_U32_WORDS],
231    // TODO: This should have been `&mut [[MaybeUninit<u32>; PROOF_U32_WORDS]; NUM_S_BUCKETS]`,
232    //  but it currently doesn't compile if flattened:
233    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
234    proofs: &mut [MaybeUninit<u32>; PROOF_U32_WORDS * NUM_S_BUCKETS],
235    found_proofs_scratch: &mut [MaybeUninit<u32>; (WORKGROUP_SIZE / u32::BITS) as usize],
236) where
237    [(); PROOF_X_SOURCES.div_ceil(SUBGROUP_SIZE as usize)]:,
238    [(); PROOF_U32_WORDS.div_ceil(SUBGROUP_SIZE as usize)]:,
239{
240    let table_6_proof_targets = find_local_proof_targets::<SUBGROUP_SIZE>(
241        local_invocation_id,
242        subgroup_id,
243        subgroup_local_invocation_id,
244        positions_group_index,
245        bucket_sizes,
246        buckets,
247        found_proofs,
248        found_proofs_scratch,
249    );
250
251    // TODO: This proof zeroing will not be needed once proofs are assembled in registers and no
252    //  longer use atomic writes
253    // Zero the proofs range with coalesced writes
254    {
255        let positions_group_words = SUBGROUP_SIZE as usize * PROOF_U32_WORDS;
256        let base_word = positions_group_index as usize * positions_group_words;
257
258        for offset in (subgroup_local_invocation_id as usize..positions_group_words)
259            .step_by(SUBGROUP_SIZE as usize)
260        {
261            proofs[base_word + offset].write(0);
262        }
263
264        subgroup_memory_barrier();
265    }
266
267    // `0` for left `1` for right
268    let left_right = (subgroup_local_invocation_id % 2) as usize;
269    // Otherwise `left_right` will not work as expected
270    const {
271        assert!(MIN_SUBGROUP_SIZE >= 2);
272    }
273
274    // `chunk_index` is used to emulate `for _ in 0..2` loops, while using a single variable for
275    // tracking the progress instead of a separate variable for each loop
276    let mut chunk_index = 0u32;
277    // Reading positions from table 6
278    loop {
279        let table_6_proof_targets = subgroup_shuffle(
280            table_6_proof_targets,
281            SUBGROUP_SIZE / 2 * (chunk_index & 1) + subgroup_local_invocation_id / 2,
282        );
283        let table_6_proof_target = table_6_proof_targets[left_right];
284
285        let table_5_proof_targets = if table_6_proof_target == Position::SENTINEL {
286            [Position::SENTINEL; 2]
287        } else {
288            table_6_positions[table_6_proof_target as usize]
289        };
290
291        // Reading positions from table 5
292        chunk_index <<= 1;
293        loop {
294            let table_5_proof_targets = subgroup_shuffle(
295                table_5_proof_targets,
296                SUBGROUP_SIZE / 2 * (chunk_index & 1) + subgroup_local_invocation_id / 2,
297            );
298            let table_5_proof_target = table_5_proof_targets[left_right];
299
300            let table_4_proof_targets = if table_5_proof_target == Position::SENTINEL {
301                [Position::SENTINEL; 2]
302            } else {
303                table_5_positions[table_5_proof_target as usize]
304            };
305
306            // Reading positions from table 4
307            chunk_index <<= 1;
308            loop {
309                let table_4_proof_targets = subgroup_shuffle(
310                    table_4_proof_targets,
311                    SUBGROUP_SIZE / 2 * (chunk_index & 1) + subgroup_local_invocation_id / 2,
312                );
313                let table_4_proof_target = table_4_proof_targets[left_right];
314
315                let table_3_proof_targets = if table_4_proof_target == Position::SENTINEL {
316                    [Position::SENTINEL; 2]
317                } else {
318                    table_4_positions[table_4_proof_target as usize]
319                };
320
321                // Reading positions from table 3
322                chunk_index <<= 1;
323                loop {
324                    let table_3_proof_targets = subgroup_shuffle(
325                        table_3_proof_targets,
326                        SUBGROUP_SIZE / 2 * (chunk_index & 1) + subgroup_local_invocation_id / 2,
327                    );
328                    let table_3_proof_target = table_3_proof_targets[left_right];
329
330                    let table_2_proof_targets = if table_3_proof_target == Position::SENTINEL {
331                        [Position::SENTINEL; 2]
332                    } else {
333                        table_3_positions[table_3_proof_target as usize]
334                    };
335
336                    // Reading positions from table 2
337                    chunk_index <<= 1;
338                    loop {
339                        let table_2_proof_targets = subgroup_shuffle(
340                            table_2_proof_targets,
341                            SUBGROUP_SIZE / 2 * (chunk_index & 1)
342                                + subgroup_local_invocation_id / 2,
343                        );
344                        let table_2_proof_target = table_2_proof_targets[left_right];
345
346                        let [x_left, x_right] = if table_2_proof_target == Position::SENTINEL {
347                            [Position::SENTINEL; 2]
348                        } else {
349                            table_2_positions[table_2_proof_target as usize]
350                        };
351
352                        let global_x_left_offset =
353                            subgroup_local_invocation_id * 2 + chunk_index * SUBGROUP_SIZE * 2;
354                        let group_proof_index = global_x_left_offset / PROOF_X_SOURCES as u32;
355                        let x_left_offset = global_x_left_offset % PROOF_X_SOURCES as u32;
356                        let global_proof_index =
357                            positions_group_index * SUBGROUP_SIZE + group_proof_index;
358
359                        let proof_base = global_proof_index as usize * PROOF_U32_WORDS;
360                        let first_proof_word_index =
361                            ((u32::from(K) * x_left_offset) / u32::BITS) as usize;
362
363                        // TODO: Writes below can be optimized by building the full proof into the
364                        //  registers first and only write final result without atomics to global
365                        //  memory
366
367                        let mut local_proof_words = [0u32; 3];
368                        let x_left_offset_in_bits = u32::from(K) * x_left_offset;
369                        {
370                            let bit_offset = x_left_offset_in_bits % u32::BITS;
371                            let x_shifted_to_start = x_left << (u32::BITS - u32::from(K));
372
373                            let first_word = x_shifted_to_start >> bit_offset;
374                            let second_word =
375                                x_shifted_to_start.unbounded_shl(u32::BITS - bit_offset);
376
377                            local_proof_words[0] = first_word;
378                            local_proof_words[1] = second_word;
379                        }
380
381                        let max_local_proof_word_index = {
382                            let x_right_offset_in_bits = x_left_offset_in_bits + u32::from(K);
383                            let local_proof_words_index = (x_right_offset_in_bits / u32::BITS)
384                                as usize
385                                - first_proof_word_index;
386                            let bit_offset = x_right_offset_in_bits % u32::BITS;
387                            let x_shifted_to_start = x_right << (u32::BITS - u32::from(K));
388
389                            let first_word = x_shifted_to_start >> bit_offset;
390                            let second_word =
391                                x_shifted_to_start.unbounded_shl(u32::BITS - bit_offset);
392
393                            local_proof_words[local_proof_words_index] |= first_word;
394                            local_proof_words[local_proof_words_index + 1] = second_word;
395
396                            local_proof_words_index + (second_word != 0) as usize
397                        };
398
399                        // The first word is written unconditionally
400                        {
401                            // SAFETY: The whole proof is initialized at the beginning of the
402                            // function
403                            let word = unsafe {
404                                proofs[proof_base + first_proof_word_index].assume_init_mut()
405                            };
406                            // SAFETY: TODO: Probably should not be unsafe to begin with:
407                            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
408                            unsafe {
409                                atomic_or::<
410                                    _,
411                                    { Scope::Subgroup as u32 },
412                                    { Semantics::NONE.bits() },
413                                >(
414                                    word, local_proof_words[0].to_be()
415                                );
416                            }
417                        }
418                        // Process remaining words, the loop is unrolled to save vector registers
419                        if max_local_proof_word_index > 0 {
420                            // SAFETY: The whole proof is initialized at the beginning of the
421                            // function
422                            let word = unsafe {
423                                proofs[proof_base + first_proof_word_index + 1].assume_init_mut()
424                            };
425                            // SAFETY: TODO: Probably should not be unsafe to begin with:
426                            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
427                            unsafe {
428                                atomic_or::<
429                                    _,
430                                    { Scope::Subgroup as u32 },
431                                    { Semantics::NONE.bits() },
432                                >(
433                                    word, local_proof_words[1].to_be()
434                                );
435                            }
436                        }
437                        if max_local_proof_word_index > 1 {
438                            // SAFETY: The whole proof is initialized at the beginning of the
439                            // function
440                            let word = unsafe {
441                                proofs[proof_base + first_proof_word_index + 2].assume_init_mut()
442                            };
443                            // SAFETY: TODO: Probably should not be unsafe to begin with:
444                            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
445                            unsafe {
446                                atomic_or::<
447                                    _,
448                                    { Scope::Subgroup as u32 },
449                                    { Semantics::NONE.bits() },
450                                >(
451                                    word, local_proof_words[2].to_be()
452                                );
453                            }
454                        }
455
456                        if chunk_index & 1 == 1 {
457                            break;
458                        }
459                        chunk_index += 1;
460                    }
461                    chunk_index >>= 1;
462
463                    if chunk_index & 1 == 1 {
464                        break;
465                    }
466                    chunk_index += 1;
467                }
468                chunk_index >>= 1;
469
470                if chunk_index & 1 == 1 {
471                    break;
472                }
473                chunk_index += 1;
474            }
475            chunk_index >>= 1;
476
477            if chunk_index & 1 == 1 {
478                break;
479            }
480            chunk_index += 1;
481        }
482        chunk_index >>= 1;
483
484        if chunk_index & 1 == 1 {
485            break;
486        }
487        chunk_index += 1;
488    }
489}

491/// NOTE: bucket sizes are zeroed after use
492// TODO: Maybe split `found_proofs` and `proofs` searching into separate shaders, such that less
493//  compute is wasted on searching proofs overall (right now up to half of the compute is wasted
494//  when computing proofs). It'll also be easier to add hashing after proof computation that way.
495#[spirv(compute(threads(256), entry_point_name = "find_proofs"))]
496#[expect(
497    clippy::too_many_arguments,
498    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
499)]
500pub fn find_proofs(
501    #[spirv(workgroup_id)] workgroup_id: UVec3,
502    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
503    #[spirv(subgroup_id)] subgroup_id: u32,
504    #[spirv(subgroup_size)] subgroup_size: u32,
505    #[spirv(num_subgroups)] num_subgroups: u32,
506    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
507    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)]
508    table_2_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
509    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)]
510    table_3_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
511    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)]
512    table_4_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
513    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)]
514    table_5_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
515    #[spirv(storage_buffer, descriptor_set = 0, binding = 4)]
516    table_6_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
517    #[spirv(storage_buffer, descriptor_set = 0, binding = 5)] bucket_sizes: &mut [u32;
518             NUM_S_BUCKETS],
519    #[spirv(storage_buffer, descriptor_set = 0, binding = 6)] buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET];
520         NUM_S_BUCKETS],
521    #[spirv(storage_buffer, descriptor_set = 0, binding = 7)] proofs: &mut Proofs,
522    #[spirv(workgroup)] found_proofs_scratch: &mut [MaybeUninit<u32>;
523             (WORKGROUP_SIZE / u32::BITS) as usize],
524) {
525    let local_invocation_id = local_invocation_id.x;
526    let workgroup_id = workgroup_id.x;
527
528    let global_subgroup_id = workgroup_id * num_subgroups + subgroup_id;
529
530    let positions_group_index = global_subgroup_id;
531    // Specify some common subgroup sizes so the driver can easily eliminate dead code. This is
