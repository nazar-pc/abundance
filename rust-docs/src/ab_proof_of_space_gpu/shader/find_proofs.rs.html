74#[inline(always)]
75fn find_local_proof_targets<const SUBGROUP_SIZE: u32>(
76    local_invocation_id: u32,
77    subgroup_id: u32,
78    subgroup_local_invocation_id: u32,
79    positions_group_index: u32,
80    bucket_sizes: &mut [u32; NUM_S_BUCKETS],
81    buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET]; NUM_S_BUCKETS],
82    found_proofs: &mut [MaybeUninit<u32>; FOUND_PROOFS_U32_WORDS],
83    found_proofs_scratch: &mut [MaybeUninit<u32>; (WORKGROUP_SIZE / u32::BITS) as usize],
84) -> [Position; 2] {
85    let local_invocation_id = local_invocation_id as usize;
86    let base = positions_group_index * SUBGROUP_SIZE;
87
88    let mut min = [Position::SENTINEL; 2];
89
90    let local_bucket_size = {
91        let bucket_id = base + subgroup_local_invocation_id;
92
93        let local_bucket_size = bucket_sizes[bucket_id as usize];
94        bucket_sizes[bucket_id as usize] = 0;
95
96        local_bucket_size
97    };
98
99    for local_bucket_id in 0..SUBGROUP_SIZE {
100        let bucket_id = (base + local_bucket_id) as usize;
101        let bucket_size = subgroup_shuffle(local_bucket_size, local_bucket_id);
102        let bucket = &buckets[bucket_id];
103
104        // TODO: Can't use the struct due to this issue in Naga:
105        //  https://github.com/gfx-rs/wgpu/issues/8389#issuecomment-3430788603
106        // let mut local_min = ProofTargets {
107        //     absolute_position: u32::MAX,
108        //     positions: [Position::SENTINEL; 2],
109        // };
110        let mut local_min_absolute_position = [u32::MAX];
111        let mut local_min_positions = [Position::SENTINEL; 2];
112
113        for index in (subgroup_local_invocation_id..bucket_size).step_by(SUBGROUP_SIZE as usize) {
114            let proof_targets = bucket[index as usize];
115            if proof_targets.absolute_position < local_min_absolute_position[0] {
116                local_min_absolute_position[0] = proof_targets.absolute_position;
117                local_min_positions = proof_targets.positions;
118            }
119        }
120
121        let min_absolute_position = subgroup_u_min(local_min_absolute_position[0]);
122        let source_lane_mask =
123            subgroup_ballot(local_min_absolute_position[0] == min_absolute_position);
124        // TODO: This intrinsic is not supported by `wgpu` yet:
125        //  https://github.com/gfx-rs/wgpu/issues/8403
126        // let source_lane = subgroup_ballot_find_lsb(source_lane_mask);
127        let source_lane_mask = source_lane_mask.0.to_array();
128        let mut source_lane = u32::MAX;
129        for i in 0..SUBGROUP_SIZE.div_ceil(u32::BITS) {
130            let word = source_lane_mask[i as usize];
131            if word != 0 {
132                source_lane = word.trailing_zeros() + i * u32::BITS;
133                break;
134            }
135        }
136        let local_min = [
137            subgroup_shuffle(local_min_positions[0], source_lane),
138            subgroup_shuffle(local_min_positions[1], source_lane),
139        ];
140
141        if subgroup_local_invocation_id == local_bucket_id {
142            min = local_min;
143        }
144    }
145
146    let has_proof = local_bucket_size > 0;
147    let found_proofs_words = subgroup_ballot(has_proof).0;
148
149    if SUBGROUP_SIZE >= u32::BITS {
150        // For subgroup sizes that are multiple of `u32` words, results can be written directly into
151        // global memory
152        // TODO: should have been `subgroup_elect()`, but it is not implemented in `wgpu` yet:
153        //  https://github.com/gfx-rs/wgpu/issues/5555
154        if subgroup_local_invocation_id == 0 {
155            let start_word = (base / u32::BITS) as usize;
156            found_proofs[start_word].write(found_proofs_words.x);
157
158            if SUBGROUP_SIZE >= 2 * u32::BITS {
159                found_proofs[start_word + 1].write(found_proofs_words.y);
160
161                if SUBGROUP_SIZE >= 4 * u32::BITS {
162                    found_proofs[start_word + 2].write(found_proofs_words.z);
163                    found_proofs[start_word + 3].write(found_proofs_words.w);
164                }
165            }
166        }
167    } else {
168        if local_invocation_id < found_proofs_scratch.len() {
169            found_proofs_scratch[local_invocation_id].write(0);
170        }
171
172        workgroup_memory_barrier_with_group_sync();
173
174        // For subgroups of smaller sizes aggregate results in shared memory first, then write to
175        // global memory
176        // TODO: should have been `subgroup_elect()`, but it is not implemented in `wgpu` yet:
177        //  https://github.com/gfx-rs/wgpu/issues/5555
178        if subgroup_local_invocation_id == 0 {
179            let local_start_bit = subgroup_id * SUBGROUP_SIZE;
180            let local_word_index = (local_start_bit / u32::BITS) as usize;
181            let local_word_shift = local_start_bit % u32::BITS;
182
183            // SAFETY: Initialized above
184            let found_proofs_word =
185                unsafe { found_proofs_scratch[local_word_index].assume_init_mut() };
186            // TODO: Probably should not be unsafe to begin with:
187            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
188            unsafe {
189                atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
190                    found_proofs_word,
191                    found_proofs_words.x << local_word_shift,
192                );
193            }
194        }
195
196        workgroup_memory_barrier_with_group_sync();
197
198        if local_invocation_id < found_proofs_scratch.len() {
199            let workgroup_base_group_index = positions_group_index - subgroup_id;
200            let workgroup_start_bucket = workgroup_base_group_index * SUBGROUP_SIZE;
201            let global_start_word = (workgroup_start_bucket / u32::BITS) as usize;
202
203            // SAFETY: Initialized above
204            let found_proofs_word =
205                unsafe { found_proofs_scratch[local_invocation_id].assume_init() };
206            found_proofs[global_start_word + local_invocation_id].write(found_proofs_word);
207        }
208    }
209
210    min
211}

218fn find_proofs_impl<const SUBGROUP_SIZE: u32>(
219    local_invocation_id: u32,
220    subgroup_id: u32,
221    subgroup_local_invocation_id: u32,
222    positions_group_index: u32,
223    // TODO: This should have been `&[[[Position; 2]; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`,
224    //  but it currently doesn't compile if flattened:
225    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
226    table_2_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
227    table_3_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
228    table_4_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
229    table_5_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
230    table_6_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
231    bucket_sizes: &mut [u32; NUM_S_BUCKETS],
232    buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET]; NUM_S_BUCKETS],
233    found_proofs: &mut [MaybeUninit<u32>; FOUND_PROOFS_U32_WORDS],
234    // TODO: This should have been `&mut [[MaybeUninit<u32>; PROOF_U32_WORDS]; NUM_S_BUCKETS]`,
235    //  but it currently doesn't compile if flattened:
236    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
237    proofs: &mut [MaybeUninit<u32>; PROOF_U32_WORDS * NUM_S_BUCKETS],
238    found_proofs_scratch: &mut [MaybeUninit<u32>; (WORKGROUP_SIZE / u32::BITS) as usize],
239) where
240    [(); PROOF_X_SOURCES.div_ceil(SUBGROUP_SIZE as usize)]:,
241    [(); PROOF_U32_WORDS.div_ceil(SUBGROUP_SIZE as usize)]:,
242{
243    let table_6_proof_targets = find_local_proof_targets::<SUBGROUP_SIZE>(
244        local_invocation_id,
245        subgroup_id,
246        subgroup_local_invocation_id,
247        positions_group_index,
248        bucket_sizes,
249        buckets,
250        found_proofs,
251        found_proofs_scratch,
252    );
253
254    // TODO: This proof zeroing will not be needed once proofs are assembled in registers and no
255    //  longer use atomic writes
256    // Zero the proofs range with coalesced writes
257    {
258        let positions_group_words = SUBGROUP_SIZE as usize * PROOF_U32_WORDS;
259        let base_word = positions_group_index as usize * positions_group_words;
260
261        for offset in (subgroup_local_invocation_id as usize..positions_group_words)
262            .step_by(SUBGROUP_SIZE as usize)
263        {
264            proofs[base_word + offset].write(0);
265        }
266
267        subgroup_memory_barrier();
268    }
269
270    // TODO: Here and below to/from `UVec2` conversion is only needed because the trait can't be
271    //  derived right now: https://github.com/Rust-GPU/rust-gpu/issues/410
272    let table_6_proof_targets = UVec2::from_array(table_6_proof_targets);
273
274    // `0` for left `1` for right
275    let left_right = (subgroup_local_invocation_id % 2) as usize;
276    // Otherwise `left_right` will not work as expected
277    const {
278        assert!(MIN_SUBGROUP_SIZE >= 2);
279    }
280
281    let mut group_left_x_index = subgroup_local_invocation_id * 2;
282
283    // TODO: This uses a lot of registers for all the loops and expressions, optimize it further
284    // Reading positions from table 6
285    for table_6_chunk in 0..2 {
286        let table_6_proof_targets = subgroup_shuffle(
287            table_6_proof_targets,
288            SUBGROUP_SIZE / 2 * table_6_chunk + subgroup_local_invocation_id / 2,
289        );
290        let table_6_proof_target = table_6_proof_targets.to_array()[left_right];
291
292        let table_5_proof_targets = if table_6_proof_target == Position::SENTINEL {
293            [Position::SENTINEL; 2]
294        } else {
295            table_6_positions[table_6_proof_target as usize]
296        };
297        let table_5_proof_targets = UVec2::from_array(table_5_proof_targets);
298
299        // Reading positions from table 5
300        for table_5_chunk in 0..2 {
301            let table_5_proof_targets = subgroup_shuffle(
302                table_5_proof_targets,
303                SUBGROUP_SIZE / 2 * table_5_chunk + subgroup_local_invocation_id / 2,
304            );
305            let table_5_proof_target = table_5_proof_targets.to_array()[left_right];
306
307            let table_4_proof_targets = if table_5_proof_target == Position::SENTINEL {
308                [Position::SENTINEL; 2]
309            } else {
310                table_5_positions[table_5_proof_target as usize]
311            };
312            let table_4_proof_targets = UVec2::from_array(table_4_proof_targets);
313
314            // Reading positions from table 4
315            for table_4_chunk in 0..2 {
316                let table_4_proof_targets = subgroup_shuffle(
317                    table_4_proof_targets,
318                    SUBGROUP_SIZE / 2 * table_4_chunk + subgroup_local_invocation_id / 2,
319                );
320                let table_4_proof_target = table_4_proof_targets.to_array()[left_right];
321
322                let table_3_proof_targets = if table_4_proof_target == Position::SENTINEL {
323                    [Position::SENTINEL; 2]
324                } else {
325                    table_4_positions[table_4_proof_target as usize]
326                };
327                let table_3_proof_targets = UVec2::from_array(table_3_proof_targets);
328
329                // Reading positions from table 3
330                for table_3_chunk in 0..2 {
331                    let table_3_proof_targets = subgroup_shuffle(
332                        table_3_proof_targets,
333                        SUBGROUP_SIZE / 2 * table_3_chunk + subgroup_local_invocation_id / 2,
334                    );
335                    let table_3_proof_target = table_3_proof_targets.to_array()[left_right];
336
337                    let table_2_proof_targets = if table_3_proof_target == Position::SENTINEL {
338                        [Position::SENTINEL; 2]
339                    } else {
340                        table_3_positions[table_3_proof_target as usize]
341                    };
342                    let table_2_proof_targets = UVec2::from_array(table_2_proof_targets);
343
344                    // Reading positions from table 2
345                    for table_2_chunk in 0..2 {
346                        let table_2_proof_targets = subgroup_shuffle(
347                            table_2_proof_targets,
348                            SUBGROUP_SIZE / 2 * table_2_chunk + subgroup_local_invocation_id / 2,
349                        );
350                        let table_2_proof_target = table_2_proof_targets.to_array()[left_right];
351
352                        let [x_left, x_right] = if table_2_proof_target == Position::SENTINEL {
353                            [Position::SENTINEL; 2]
354                        } else {
355                            table_2_positions[table_2_proof_target as usize]
356                        };
357
358                        let group_proof_index = group_left_x_index / PROOF_X_SOURCES as u32;
359                        let x_left_offset = group_left_x_index % PROOF_X_SOURCES as u32;
360                        let global_proof_index =
361                            positions_group_index * SUBGROUP_SIZE + group_proof_index;
362                        group_left_x_index += SUBGROUP_SIZE * 2;
363
364                        let proof_base = global_proof_index as usize * PROOF_U32_WORDS;
365                        let first_proof_word_index =
366                            ((u32::from(K) * x_left_offset) / u32::BITS) as usize;
367
368                        // TODO: Writes below can be optimized by building the full proof into the
369                        //  registers first and only write final result without atomics to global
370                        //  memory
371
372                        let mut local_proof_words = [0u32; 3];
373                        let x_left_offset_in_bits = u32::from(K) * x_left_offset;
374                        {
375                            let bit_offset = x_left_offset_in_bits % u32::BITS;
376                            let x_shifted_to_start = x_left << (u32::BITS - u32::from(K));
377
378                            let first_word = x_shifted_to_start >> bit_offset;
379                            let second_word =
380                                x_shifted_to_start.unbounded_shl(u32::BITS - bit_offset);
381
382                            local_proof_words[0] = first_word;
383                            local_proof_words[1] = second_word;
384                        }
385
386                        let max_local_proof_word_index = {
387                            let x_right_offset_in_bits = x_left_offset_in_bits + u32::from(K);
388                            let local_proof_words_index = (x_right_offset_in_bits / u32::BITS)
389                                as usize
390                                - first_proof_word_index;
391                            let bit_offset = x_right_offset_in_bits % u32::BITS;
392                            let x_shifted_to_start = x_right << (u32::BITS - u32::from(K));
393
394                            let first_word = x_shifted_to_start >> bit_offset;
395                            let second_word =
396                                x_shifted_to_start.unbounded_shl(u32::BITS - bit_offset);
397
398                            local_proof_words[local_proof_words_index] |= first_word;
399                            local_proof_words[local_proof_words_index + 1] = second_word;
400
401                            local_proof_words_index + (second_word != 0) as usize
402                        };
403
404                        // The first word is written unconditionally
405                        {
406                            // SAFETY: The whole proof is initialized at the beginning of the
407                            // function
408                            let word = unsafe {
409                                proofs[proof_base + first_proof_word_index].assume_init_mut()
410                            };
411                            // TODO: Probably should not be unsafe to begin with:
412                            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
413                            unsafe {
414                                atomic_or::<
415                                    _,
416                                    { Scope::Subgroup as u32 },
417                                    { Semantics::NONE.bits() },
418                                >(
419                                    word, local_proof_words[0].to_be()
420                                );
421                            }
422                        }
423                        // Process remaining words
424                        for i in 1..=max_local_proof_word_index {
425                            // SAFETY: The whole proof is initialized at the beginning of the
426                            // function
427                            let word = unsafe {
428                                proofs[proof_base + first_proof_word_index + i].assume_init_mut()
429                            };
430                            // TODO: Probably should not be unsafe to begin with:
431                            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
432                            unsafe {
433                                atomic_or::<
434                                    _,
435                                    { Scope::Subgroup as u32 },
436                                    { Semantics::NONE.bits() },
437                                >(
438                                    word, local_proof_words[i].to_be()
439                                );
440                            }
441                        }
442                    }
443                }
444            }
445        }
446    }
447}

449/// NOTE: bucket sizes are zeroed after use
450// TODO: Maybe split `found_proofs` and `proofs` searching into separate shaders, such that less
451//  compute is wasted on searching proofs overall (right now up to half of the compute is wasted
452//  when computing proofs). It'll also be easier to add hashing after proof computation that way.
453#[spirv(compute(threads(256), entry_point_name = "find_proofs"))]
454#[expect(
455    clippy::too_many_arguments,
456    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
457)]
458pub fn find_proofs(
459    #[spirv(workgroup_id)] workgroup_id: UVec3,
460    #[spirv(num_workgroups)] num_workgroups: UVec3,
461    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
462    #[spirv(subgroup_id)] subgroup_id: u32,
463    #[spirv(subgroup_size)] subgroup_size: u32,
464    #[spirv(num_subgroups)] num_subgroups: u32,
465    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
466    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)]
467    table_2_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
468    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)]
469    table_3_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
470    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)]
471    table_4_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
472    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)]
473    table_5_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
474    #[spirv(storage_buffer, descriptor_set = 0, binding = 4)]
475    table_6_positions: &[[Position; 2]; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
476    #[spirv(storage_buffer, descriptor_set = 0, binding = 5)] bucket_sizes: &mut [u32;
477             NUM_S_BUCKETS],
478    #[spirv(storage_buffer, descriptor_set = 0, binding = 6)] buckets: &[[ProofTargets; NUM_ELEMENTS_PER_S_BUCKET];
479         NUM_S_BUCKETS],
480    #[spirv(storage_buffer, descriptor_set = 0, binding = 7)] proofs: &mut Proofs,
481    #[spirv(workgroup)] found_proofs_scratch: &mut [MaybeUninit<u32>;
482             (WORKGROUP_SIZE / u32::BITS) as usize],
483) {
484    let local_invocation_id = local_invocation_id.x;
485    let workgroup_id = workgroup_id.x;
486    let num_workgroups = num_workgroups.x;
487
488    let total_subgroups = num_workgroups * num_subgroups;
489    let global_subgroup_id = workgroup_id * num_subgroups + subgroup_id;
490
491    // Process `subgroup_size` buckets per subgroup
492    // TODO: More idiomatic version currently doesn't compile:
493    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
494    for positions_group_index in
495        (global_subgroup_id..NUM_S_BUCKETS as u32 / subgroup_size).step_by(total_subgroups as usize)
496    {
497        // Specify some common subgroup sizes so the driver can easily eliminate dead code. This is
498        // important because `local_words` inside the function is generic and impacts the number of
499        // registers used, so we want to minimize them.
500        match subgroup_size {
501            // Hypothetically possible
502            1 => {
503                find_proofs_impl::<1>(
504                    local_invocation_id,
505                    subgroup_id,
