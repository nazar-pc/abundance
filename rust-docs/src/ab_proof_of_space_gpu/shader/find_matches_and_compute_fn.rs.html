46/// # Safety
47/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
48/// `matches` must be initialized, `matches` must have valid pointers into left/right buckets and
49/// `parent_metadatas`.
50#[inline(always)]
51#[expect(
52    clippy::too_many_arguments,
53    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
54)]
55unsafe fn compute_fn_into_buckets_inner<const TABLE_NUMBER: u8, const PARENT_TABLE_NUMBER: u8>(
56    index: u32,
57    left_bucket_base: u32,
58    metadatas_offset: u32,
59    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
60    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
61    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
62    matches: &[MaybeUninit<Match>; MAX_BUCKET_SIZE],
63    // TODO: This should have been `&[[Metadata; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`, but
64    //  it currently doesn't compile if flattened:
65    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
66    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
67    bucket_sizes: &mut [u32; NUM_BUCKETS],
68    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
69    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
70    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
71    bucket_scratch: &[PositionR; REDUCED_BUCKET_SIZE],
72) {
73    // SAFETY: Guaranteed by function contract
74    let (bucket_offset, r_target, positions_offset) =
75        unsafe { matches.get_unchecked(index as usize).assume_init() }.split();
76
77    // SAFETY: Guaranteed by function contract
78    let left_position_r = *unsafe { left_bucket.get_unchecked(bucket_offset as usize) };
79    let left_position = left_position_r.position;
80    let left_r = left_position_r.r.get();
81
82    // Repurpose variable for two purposes to save on registers
83    let mut right_position_or_skip = positions_offset;
84    // TODO: More idiomatic version currently doesn't compile:
85    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
86    #[expect(clippy::needless_range_loop)]
87    for offset in 0..REDUCED_BUCKET_SIZE {
88        let position_r = bucket_scratch[offset];
89        if position_r.r.get() == r_target {
90            if right_position_or_skip == 0 {
91                right_position_or_skip = position_r.position;
92                break;
93            } else {
94                right_position_or_skip -= 1;
95            }
96        }
97    }
98    let right_position = right_position_or_skip;
99
100    // TODO: Correct version currently doesn't compile:
101    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
102    // let left_metadata = parent_metadatas[usize::from(left_position)];
103    // let right_metadata = parent_metadatas[usize::from(right_position)];
104    // SAFETY: Guaranteed by function contract
105    let left_metadata = *unsafe { parent_metadatas.get_unchecked(left_position as usize) };
106    // SAFETY: Guaranteed by function contract
107    let right_metadata = *unsafe { parent_metadatas.get_unchecked(right_position as usize) };
108
109    let (y, metadata) = compute_fn_impl::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
110        Y::from(left_bucket_base + left_r),
111        left_metadata,
112        right_metadata,
113    );
114
115    let (bucket_index, r) = y.into_bucket_index_and_r();
116    // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
117    let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
118    // SAFETY: TODO: Probably should not be unsafe to begin with:
119    //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
120    let bucket_offset = unsafe {
121        atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
122            bucket_size,
123        )
124    };
125
126    // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
127    // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset`
128    // is also always within bounds.
129    unsafe {
130        buckets
131            .get_unchecked_mut(bucket_index as usize)
132            .get_unchecked_mut(bucket_offset as usize)
133    }
134    .write(PositionR {
135        position: Position::from_u32(metadatas_offset + index),
136        r,
137    });
138
139    positions[index as usize].write([left_position, right_position]);
140
141    // The last table doesn't have any metadata
142    if TABLE_NUMBER < 7 {
143        metadatas[index as usize].write(metadata);
144    }
145}

147/// # Safety
148/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
149/// `matches` must be initialized, `matches` must have valid pointers into left/right buckets and
150/// `parent_metadatas`.
151#[inline(always)]
152#[expect(
153    clippy::too_many_arguments,
154    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
155)]
156unsafe fn compute_fn_into_buckets<const TABLE_NUMBER: u8, const PARENT_TABLE_NUMBER: u8>(
157    local_invocation_id: u32,
158    left_bucket_index: u32,
159    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
160    right_bucket: &[PositionR; MAX_BUCKET_SIZE],
161    matches_count: usize,
162    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
163    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
164    matches: &[MaybeUninit<Match>; MAX_BUCKET_SIZE],
165    // TODO: This should have been `&[[Metadata; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`, but
166    //  it currently doesn't compile if flattened:
167    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
168    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
169    bucket_sizes: &mut [u32; NUM_BUCKETS],
170    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
171    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
172    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
173    bucket_scratch: &mut [PositionR; REDUCED_BUCKET_SIZE],
174) {
175    // Load the right bucket into shared memory for faster access
176    for bucket_offset in
177        (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
178    {
179        bucket_scratch[bucket_offset] = right_bucket[bucket_offset];
180    }
181
182    workgroup_memory_barrier_with_group_sync();
183
184    let left_bucket_base = left_bucket_index * u32::from(PARAM_BC);
185    let metadatas_offset = left_bucket_index * REDUCED_MATCHES_COUNT as u32;
186
187    const {
188        assert!(MAX_BUCKET_SIZE == WORKGROUP_SIZE as usize * 2);
189    }
190    // TODO: This should have been a loop, but register usage is too high, see:
191    //  https://github.com/Rust-GPU/rust-gpu/issues/462
192    // SAFETY: Guaranteed by function contract
193    unsafe {
194        if (local_invocation_id as usize) < matches_count {
195            compute_fn_into_buckets_inner::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
196                local_invocation_id,
197                left_bucket_base,
198                metadatas_offset,
199                left_bucket,
200                matches,
201                parent_metadatas,
202                bucket_sizes,
203                buckets,
204                positions,
205                metadatas,
206                bucket_scratch,
207            );
208        }
209        if ((local_invocation_id + WORKGROUP_SIZE) as usize) < matches_count {
210            compute_fn_into_buckets_inner::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
211                local_invocation_id + WORKGROUP_SIZE,
212                left_bucket_base,
213                metadatas_offset,
214                left_bucket,
215                matches,
216                parent_metadatas,
217                bucket_sizes,
218                buckets,
219                positions,
220                metadatas,
221                bucket_scratch,
222            );
223        }
224    }
225}

227/// # Safety
228/// Must be called from [`WORKGROUP_SIZE`] threads. All buckets must contain valid positions.
229#[expect(
230    clippy::too_many_arguments,
231    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
232)]
233pub unsafe fn find_matches_and_compute_fn<const TABLE_NUMBER: u8, const PARENT_TABLE_NUMBER: u8>(
234    local_invocation_id: UVec3,
235    workgroup_id: UVec3,
236    parent_buckets: &[[PositionR; MAX_BUCKET_SIZE]; NUM_BUCKETS],
237    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
238    bucket_sizes: &mut [u32; NUM_BUCKETS],
239    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
240    positions: &mut [[MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS],
241    metadatas: &mut [[MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS],
242    matches: &mut [MaybeUninit<Match>; MAX_BUCKET_SIZE],
243    shared: &mut FindMatchesAndComputeFnShared,
244) {
245    let local_invocation_id = local_invocation_id.x;
246    let workgroup_id = workgroup_id.x;
247
248    let left_bucket_index = workgroup_id as usize;
249    let left_bucket = &parent_buckets[left_bucket_index];
250    let right_bucket = &parent_buckets[left_bucket_index + 1];
251    let positions = &mut positions[left_bucket_index];
252    let metadatas = &mut metadatas[left_bucket_index];
253    let left_bucket_index = left_bucket_index as u32;
254
255    // TODO: Truncate buckets to reduced size here once it compiles:
256    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
257    // SAFETY: Guaranteed by function contract
258    let matches_count = unsafe {
259        find_matches_in_buckets_impl(
260            local_invocation_id,
261            left_bucket_index,
262            left_bucket,
263            right_bucket,
264            matches,
265            &mut shared.find_matches_shared,
266        )
267    };
268
269    // SAFETY: Guaranteed by function contract and call to `find_matches_in_buckets_impl`
270    unsafe {
271        compute_fn_into_buckets::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
272            local_invocation_id,
273            left_bucket_index,
274            left_bucket,
275            right_bucket,
276            matches_count as usize,
277            matches,
278            parent_metadatas,
279            bucket_sizes,
280            buckets,
281            positions,
282            metadatas,
283            &mut shared.bucket_scratch,
284        );
285    }
286}
