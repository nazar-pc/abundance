29// TODO: This is a polyfill to work around for this issue:
30//  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
31#[cfg(target_arch = "spirv")]
32trait ArrayIndexingPolyfill<T> {
33    /// The same as [`<[T]>::get_unchecked()`]
34    unsafe fn get_unchecked(&self, index: usize) -> &T;
35    /// The same as [`<[T]>::get_unchecked_mut()`]
36    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T;
37}
38
39#[cfg(target_arch = "spirv")]
40impl<const N: usize, T> ArrayIndexingPolyfill<T> for [T; N] {
41    #[inline(always)]
42    unsafe fn get_unchecked(&self, index: usize) -> &T {
43        &self[index]
44    }
45
46    #[inline(always)]
47    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
48        &mut self[index]
49    }
50}

52/// # Safety
53/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
54/// `matches` must be initialized, `matches` must have valid pointers into `parent_metadatas`.
55#[inline(always)]
56#[expect(
57    clippy::too_many_arguments,
58    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
59)]
60unsafe fn compute_fn_into_buckets<const TABLE_NUMBER: u8, const PARENT_TABLE_NUMBER: u8>(
61    local_invocation_id: u32,
62    left_bucket_index: u32,
63    matches_count: usize,
64    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
65    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
66    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
67    // TODO: This should have been `&[[Metadata; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`, but it
68    //  currently doesn't compile if flattened:
69    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
70    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
71    bucket_sizes: &mut [u32; NUM_BUCKETS],
72    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
73    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
74    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
75) {
76    let metadatas_offset = left_bucket_index * REDUCED_MATCHES_COUNT as u32;
77
78    // TODO: More idiomatic version currently doesn't compile:
79    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
80    for index in (local_invocation_id..matches_count as u32).step_by(WORKGROUP_SIZE as usize) {
81        // SAFETY: Guaranteed by function contract
82        let m = unsafe { matches.get_unchecked(index as usize).assume_init() };
83        // SAFETY: Guaranteed by function contract
84        let left_metadata = *unsafe { parent_metadatas.get_unchecked(m.left_position as usize) };
85        // SAFETY: Guaranteed by function contract
86        let right_metadata = *unsafe { parent_metadatas.get_unchecked(m.right_position as usize) };
87
88        let (y, metadata) = compute_fn_impl::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
89            m.left_y,
90            left_metadata,
91            right_metadata,
92        );
93
94        let (bucket_index, r) = y.into_bucket_index_and_r();
95        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
96        let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
97        // TODO: Probably should not be unsafe to begin with:
98        //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
99        let bucket_offset = unsafe {
100            atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
101                bucket_size,
102            )
103        };
104
105        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
106        // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset`
107        // is also always within bounds.
108        unsafe {
109            buckets
110                .get_unchecked_mut(bucket_index as usize)
111                .get_unchecked_mut(bucket_offset as usize)
112        }
113        .write(PositionR {
114            position: Position::from_u32(metadatas_offset + index),
115            r,
116        });
117
118        positions[index as usize].write([m.left_position, m.right_position]);
119
120        // The last table doesn't have any metadata
121        if TABLE_NUMBER < 7 {
122            metadatas[index as usize].write(metadata);
123        }
124    }
125}

127/// # Safety
128/// Must be called from [`WORKGROUP_SIZE`] threads. `num_subgroups` must be at most
129/// [`MAX_SUBGROUPS`]. All buckets must come from the `sort_buckets_with_rmap_details` shader.
130#[expect(
131    clippy::too_many_arguments,
132    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
133)]
134pub unsafe fn find_matches_and_compute_fn<const TABLE_NUMBER: u8, const PARENT_TABLE_NUMBER: u8>(
135    local_invocation_id: UVec3,
136    workgroup_id: UVec3,
137    num_workgroups: UVec3,
138    subgroup_local_invocation_id: u32,
139    subgroup_id: u32,
140    num_subgroups: u32,
141    parent_buckets: &[[PositionR; MAX_BUCKET_SIZE]; NUM_BUCKETS],
142    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
143    bucket_sizes: &mut [u32; NUM_BUCKETS],
144    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
145    positions: &mut [[MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS],
146    metadatas: &mut [[MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS],
147    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
148    scratch_space: &mut SharedScratchSpace,
149    rmap: &mut MaybeUninit<Rmap>,
150) {
151    let local_invocation_id = local_invocation_id.x;
152    let workgroup_id = workgroup_id.x;
153    let num_workgroups = num_workgroups.x;
154
155    // TODO: More idiomatic version currently doesn't compile:
156    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
157    // for (left_bucket_index, (([left_bucket, right_bucket], matches), matches_count)) in buckets
158    //     .array_windows::<2>()
159    //     .zip(matches)
160    //     .zip(matches_counts)
161    //     .enumerate()
162    //     .skip(workgroup_id as usize)
163    //     .step_by(num_workgroups as usize)
164    // {
165    for left_bucket_index in
166        (workgroup_id as usize..NUM_MATCH_BUCKETS).step_by(num_workgroups as usize)
167    {
168        let left_bucket = &parent_buckets[left_bucket_index];
169        let right_bucket = &parent_buckets[left_bucket_index + 1];
170        let positions = &mut positions[left_bucket_index];
171        let metadatas = &mut metadatas[left_bucket_index];
172        let left_bucket_index = left_bucket_index as u32;
173
174        // TODO: Truncate buckets to reduced size here once it compiles:
175        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
176        // SAFETY: Guaranteed by function contract
177        let matches_count = unsafe {
178            find_matches_in_buckets_impl(
179                subgroup_local_invocation_id,
180                subgroup_id,
181                num_subgroups,
182                local_invocation_id,
183                left_bucket_index,
184                left_bucket,
185                right_bucket,
186                matches,
187                scratch_space,
188                rmap,
189            )
190        };
191
192        workgroup_memory_barrier_with_group_sync();
193
194        unsafe {
195            compute_fn_into_buckets::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
196                local_invocation_id,
197                left_bucket_index,
198                matches_count as usize,
199                matches,
200                parent_metadatas,
201                bucket_sizes,
202                buckets,
203                positions,
204                metadatas,
205            );
206        }
207
208        // No need for explicit synchronization, `matches` will not be touched before extra
209        // synchronization in `find_matches_in_buckets_impl` again anyway
210    }
211}
