48/// # Safety
49/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
50/// `matches` must be initialized. `matches` must have valid pointers into left/right buckets and
51/// `parent_metadatas`.
52#[inline(always)]
53#[expect(
54    clippy::too_many_arguments,
55    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
56)]
57unsafe fn compute_f2_into_buckets_inner(
58    index: u32,
59    left_bucket_base: u32,
60    metadatas_offset: u32,
61    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
62    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
63    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
64    matches: &[MaybeUninit<Match>; MAX_BUCKET_SIZE],
65    bucket_sizes: &mut [u32; NUM_BUCKETS],
66    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
67    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
68    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
69    bucket_scratch: &[PositionR; REDUCED_BUCKET_SIZE],
70) {
71    // SAFETY: Guaranteed by function contract
72    let (bucket_offset, r_target, positions_offset) =
73        unsafe { matches.get_unchecked(index as usize).assume_init() }.split();
74
75    // SAFETY: Guaranteed by function contract
76    let left_position_r = *unsafe { left_bucket.get_unchecked(bucket_offset as usize) };
77    let left_position = left_position_r.position;
78    let left_r = left_position_r.r.get();
79
80    // Repurpose variable for two purposes to save on registers
81    let mut right_position_or_skip = positions_offset;
82    // TODO: More idiomatic version currently doesn't compile:
83    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
84    #[expect(clippy::needless_range_loop)]
85    for offset in 0..REDUCED_BUCKET_SIZE {
86        let position_r = bucket_scratch[offset];
87        if position_r.r.get() == r_target {
88            if right_position_or_skip == 0 {
89                right_position_or_skip = position_r.position;
90                break;
91            } else {
92                right_position_or_skip -= 1;
93            }
94        }
95    }
96    let right_position = right_position_or_skip;
97
98    let left_metadata = Metadata::from(left_position);
99    let right_metadata = Metadata::from(right_position);
100
101    let (y, metadata) = compute_fn_impl::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
102        Y::from(left_bucket_base + left_r),
103        left_metadata,
104        right_metadata,
105    );
106
107    let (bucket_index, r) = y.into_bucket_index_and_r();
108    // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
109    let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
110    // SAFETY: TODO: Probably should not be unsafe to begin with:
111    //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
112    let bucket_offset = unsafe {
113        atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
114            bucket_size,
115        )
116    };
117
118    // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
119    // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset`
120    // is also always within bounds.
121    unsafe {
122        buckets
123            .get_unchecked_mut(bucket_index as usize)
124            .get_unchecked_mut(bucket_offset as usize)
125    }
126    .write(PositionR {
127        position: Position::from_u32(metadatas_offset + index),
128        r,
129    });
130
131    positions[index as usize].write([left_position, right_position]);
132
133    metadatas[index as usize].write(metadata);
134}

136/// # Safety
137/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
138/// `matches` must be initialized. `matches` must have valid pointers into left/right buckets and
139/// `parent_metadatas`.
140#[expect(
141    clippy::too_many_arguments,
142    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
143)]
144unsafe fn compute_f2_into_buckets(
145    local_invocation_id: u32,
146    left_bucket_index: u32,
147    left_bucket: &[PositionR; MAX_BUCKET_SIZE],
148    right_bucket: &[PositionR; MAX_BUCKET_SIZE],
149    matches_count: usize,
150    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
151    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
152    matches: &[MaybeUninit<Match>; MAX_BUCKET_SIZE],
153    bucket_sizes: &mut [u32; NUM_BUCKETS],
154    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
155    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
156    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
157    bucket_scratch: &mut [PositionR; REDUCED_BUCKET_SIZE],
158) {
159    // Load the right bucket into shared memory for faster access
160    for bucket_offset in
161        (local_invocation_id as usize..REDUCED_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
162    {
163        bucket_scratch[bucket_offset] = right_bucket[bucket_offset];
164    }
165
166    workgroup_memory_barrier_with_group_sync();
167
168    let left_bucket_base = left_bucket_index * u32::from(PARAM_BC);
169    let metadatas_offset = left_bucket_index * REDUCED_MATCHES_COUNT as u32;
170
171    const {
172        assert!(MAX_BUCKET_SIZE == WORKGROUP_SIZE as usize * 2);
173    }
174    // TODO: This should have been a loop, but register usage is too high, see:
175    //  https://github.com/Rust-GPU/rust-gpu/issues/462
176    // SAFETY: Guaranteed by function contract
177    unsafe {
178        if (local_invocation_id as usize) < matches_count {
179            compute_f2_into_buckets_inner(
180                local_invocation_id,
181                left_bucket_base,
182                metadatas_offset,
183                left_bucket,
184                matches,
185                bucket_sizes,
186                buckets,
187                positions,
188                metadatas,
189                bucket_scratch,
190            );
191        }
192        if ((local_invocation_id + WORKGROUP_SIZE) as usize) < matches_count {
193            compute_f2_into_buckets_inner(
194                local_invocation_id + WORKGROUP_SIZE,
195                left_bucket_base,
196                metadatas_offset,
197                left_bucket,
198                matches,
199                bucket_sizes,
200                buckets,
201                positions,
202                metadatas,
203                bucket_scratch,
204            );
205        }
206    }
207}

209/// This is similar to `find_matches_and_compute_fn`, but it doesn't use any parent table metadata.
210///
211/// Buckets need to be sorted by position afterward due to concurrent writes that do not have
212/// deterministic order. Content of the bucket beyond the size specified in `bucket_sizes` is
213/// undefined.
214///
215/// # Safety
216/// Must be called from [`WORKGROUP_SIZE`] threads. All buckets must contain valid positions.
217#[spirv(compute(threads(256), entry_point_name = "find_matches_and_compute_f2"))]
218#[expect(
219    clippy::too_many_arguments,
220    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
221)]
222pub unsafe fn find_matches_and_compute_f2(
223    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
224    #[spirv(workgroup_id)] workgroup_id: UVec3,
225    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] parent_buckets: &[[PositionR; MAX_BUCKET_SIZE];
226         NUM_BUCKETS],
227    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] bucket_sizes: &mut [u32; NUM_BUCKETS],
228    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE];
229             NUM_BUCKETS],
230    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)] positions: &mut [[MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT];
231             NUM_MATCH_BUCKETS],
232    #[spirv(storage_buffer, descriptor_set = 0, binding = 4)] metadatas: &mut [[MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT];
233             NUM_MATCH_BUCKETS],
234    #[spirv(workgroup)] matches: &mut [MaybeUninit<Match>; MAX_BUCKET_SIZE],
235    #[spirv(workgroup)] shared: &mut FindMatchesAndComputeF2Shared,
236) {
237    let local_invocation_id = local_invocation_id.x;
238    let workgroup_id = workgroup_id.x;
239
240    let left_bucket_index = workgroup_id as usize;
241    let left_bucket = &parent_buckets[left_bucket_index];
242    let right_bucket = &parent_buckets[left_bucket_index + 1];
243    let positions = &mut positions[left_bucket_index];
244    let metadatas = &mut metadatas[left_bucket_index];
245    let left_bucket_index = left_bucket_index as u32;
246
247    // TODO: Truncate buckets to reduced size here once it compiles:
248    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
249    // SAFETY: Guaranteed by function contract
250    let matches_count = unsafe {
251        find_matches_in_buckets_impl(
252            local_invocation_id,
253            left_bucket_index,
254            left_bucket,
255            right_bucket,
256            matches,
257            &mut shared.find_matches_shared,
258        )
259    };
260
261    // SAFETY: Guaranteed by function contract and call to `find_matches_in_buckets_impl`
262    unsafe {
263        compute_f2_into_buckets(
264            local_invocation_id,
265            left_bucket_index,
266            left_bucket,
267            right_bucket,
268            matches_count as usize,
269            matches,
270            bucket_sizes,
271            buckets,
272            positions,
273            metadatas,
274            &mut shared.bucket_scratch,
275        );
276    }
277}
