33#[cfg(target_arch = "spirv")]
34trait ArrayIndexingPolyfill<T> {
35    /// The same as [`<[T]>::get_unchecked()`]
36    unsafe fn get_unchecked(&self, index: usize) -> &T;
37    /// The same as [`<[T]>::get_unchecked_mut()`]
38    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T;
39}

41#[cfg(target_arch = "spirv")]
42impl<const N: usize, T> ArrayIndexingPolyfill<T> for [T; N] {
43    #[inline(always)]
44    unsafe fn get_unchecked(&self, index: usize) -> &T {
45        &self[index]
46    }
47
48    #[inline(always)]
49    unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
50        &mut self[index]
51    }
52}

54/// # Safety
55/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
56/// `matches` must be initialized.
57#[inline(always)]
58#[expect(
59    clippy::too_many_arguments,
60    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
61)]
62unsafe fn compute_f2_into_buckets(
63    local_invocation_id: u32,
64    left_bucket_index: u32,
65    matches_count: usize,
66    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
67    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
68    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
69    bucket_sizes: &mut [u32; NUM_BUCKETS],
70    buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE]; NUM_BUCKETS],
71    positions: &mut [MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT],
72    metadatas: &mut [MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT],
73) {
74    let metadatas_offset = left_bucket_index * REDUCED_MATCHES_COUNT as u32;
75
76    // TODO: More idiomatic version currently doesn't compile:
77    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
78    for index in (local_invocation_id..matches_count as u32).step_by(WORKGROUP_SIZE as usize) {
79        // SAFETY: Guaranteed by function contract
80        let m = unsafe { matches.get_unchecked(index as usize).assume_init() };
81        let left_metadata = Metadata::from(m.left_position);
82        let right_metadata = Metadata::from(m.right_position);
83
84        let (y, metadata) = compute_fn_impl::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
85            m.left_y,
86            left_metadata,
87            right_metadata,
88        );
89
90        let (bucket_index, r) = y.into_bucket_index_and_r();
91        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition
92        let bucket_size = unsafe { bucket_sizes.get_unchecked_mut(bucket_index as usize) };
93        // TODO: Probably should not be unsafe to begin with:
94        //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
95        let bucket_offset = unsafe {
96            atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
97                bucket_size,
98            )
99        };
100
101        // SAFETY: Bucket is obtained using division by `PARAM_BC` and fits by definition. Bucket
102        // size upper bound is known statically to be [`MAX_BUCKET_SIZE`], so `bucket_offset`
103        // is also always within bounds.
104        unsafe {
105            buckets
106                .get_unchecked_mut(bucket_index as usize)
107                .get_unchecked_mut(bucket_offset as usize)
108        }
109        .write(PositionR {
110            position: Position::from_u32(metadatas_offset + index),
111            r,
112        });
113
114        positions[index as usize].write([m.left_position, m.right_position]);
115
116        metadatas[index as usize].write(metadata);
117    }
118}

120/// This is similar to `find_matches_and_compute_fn`, but it doesn't use any parent table metadata.
121///
122/// Buckets need to be sorted by position afterward due to concurrent writes that do not have
123/// deterministic order. Content of the bucket beyond the size specified in `bucket_sizes` is
124/// undefined.
125///
126/// # Safety
127/// Must be called from [`WORKGROUP_SIZE`] threads. `num_subgroups` must be at most
128/// [`MAX_SUBGROUPS`].  All buckets must come from the `sort_buckets_with_rmap_details` shader.
129#[spirv(compute(threads(256), entry_point_name = "find_matches_and_compute_f2"))]
130#[expect(
131    clippy::too_many_arguments,
132    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
133)]
134pub unsafe fn find_matches_and_compute_f2(
135    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
136    #[spirv(workgroup_id)] workgroup_id: UVec3,
137    #[spirv(num_workgroups)] num_workgroups: UVec3,
138    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
139    #[spirv(subgroup_id)] subgroup_id: u32,
140    #[spirv(num_subgroups)] num_subgroups: u32,
141    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] parent_buckets: &[[PositionR; MAX_BUCKET_SIZE];
142         NUM_BUCKETS],
143    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] bucket_sizes: &mut [u32; NUM_BUCKETS],
144    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)] buckets: &mut [[MaybeUninit<PositionR>; MAX_BUCKET_SIZE];
145             NUM_BUCKETS],
146    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)] positions: &mut [[MaybeUninit<[Position; 2]>; REDUCED_MATCHES_COUNT];
147             NUM_MATCH_BUCKETS],
148    #[spirv(storage_buffer, descriptor_set = 0, binding = 4)] metadatas: &mut [[MaybeUninit<Metadata>; REDUCED_MATCHES_COUNT];
149             NUM_MATCH_BUCKETS],
150    #[spirv(workgroup)] matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
151    #[spirv(workgroup)] scratch_space: &mut SharedScratchSpace,
152    // Non-modern GPUs do not have enough space in the shared memory
153    #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
154    #[spirv(workgroup)]
155    rmap: &mut MaybeUninit<Rmap>,
156    #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
157    #[spirv(storage_buffer, descriptor_set = 0, binding = 5)]
158    rmap: &mut [MaybeUninit<Rmap>; MAX_SUBGROUPS],
159) {
160    let local_invocation_id = local_invocation_id.x;
161    let workgroup_id = workgroup_id.x;
162    let num_workgroups = num_workgroups.x;
163
164    // TODO: More idiomatic version currently doesn't compile:
165    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
166    // for (left_bucket_index, (([left_bucket, right_bucket], matches), matches_count)) in buckets
167    //     .array_windows::<2>()
168    //     .zip(matches)
169    //     .zip(matches_counts)
170    //     .enumerate()
171    //     .skip(workgroup_id as usize)
172    //     .step_by(num_workgroups as usize)
173    // {
174    for left_bucket_index in
175        (workgroup_id as usize..NUM_MATCH_BUCKETS).step_by(num_workgroups as usize)
176    {
177        let left_bucket = &parent_buckets[left_bucket_index];
178        let right_bucket = &parent_buckets[left_bucket_index + 1];
179        let positions = &mut positions[left_bucket_index];
180        let metadatas = &mut metadatas[left_bucket_index];
181        let left_bucket_index = left_bucket_index as u32;
182
183        // TODO: Truncate buckets to reduced size here once it compiles:
184        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
185        // SAFETY: Guaranteed by function contract
186        let matches_count = unsafe {
187            find_matches_in_buckets_impl(
188                subgroup_local_invocation_id,
189                subgroup_id,
190                num_subgroups,
191                local_invocation_id,
192                left_bucket_index,
193                left_bucket,
194                right_bucket,
195                matches,
196                scratch_space,
197                #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
198                rmap,
199                #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
200                &mut rmap[subgroup_id as usize],
201            )
202        };
203
204        workgroup_memory_barrier_with_group_sync();
205
206        unsafe {
207            compute_f2_into_buckets(
208                local_invocation_id,
209                left_bucket_index,
210                matches_count as usize,
211                matches,
212                bucket_sizes,
213                buckets,
214                positions,
215                metadatas,
216            );
217        }
218
219        // No need for explicit synchronization, `matches` will not be touched before extra
220        // synchronization in `find_matches_in_buckets_impl` again anyway
221    }
222}
