10// TODO: Benchmark on different GPUs to see if the complexity of dealing with 9-bit pointers is
11//  worth it or maybe using u16s would be better despite using more shared memory
12/// Number of bits necessary to address a single pair of positions in the rmap
13const POINTER_BITS: u32 = REDUCED_BUCKET_SIZE.next_power_of_two().ilog2();
14const POINTERS_BITS: usize = PARAM_BC as usize * POINTER_BITS as usize;
15const POINTERS_WORDS: usize = POINTERS_BITS.div_ceil(u32::BITS as usize);
16
17// Ensure `u32` is sufficiently large as a container
18const _: () = assert!(POINTER_BITS <= u32::BITS);

25impl NextPhysicalPointer {
26    /// Increments next physical pointer and returns previous value
27    #[inline(always)]
28    fn inc(&mut self) -> u32 {
29        let physical_pointer = self.next_physical_pointer;
30        self.next_physical_pointer += 1;
31        physical_pointer
32    }
33}

102#[derive(Debug, Copy, Clone)]
103pub struct Rmap {
104    /// `0` is a sentinel value indicating no virtual pointer is stored yet.
105    ///
106    /// Physical pointer must be increased by `1` to get a virtual pointer before storing. Virtual
107    /// pointer must be decreased by `1` before reading to get a physical pointer.
108    virtual_pointers: [u32; POINTERS_WORDS],
109    positions: [[Position; 2]; REDUCED_BUCKET_SIZE],
110}
111
112impl Rmap {
113    #[cfg(test)]
114    #[inline(always)]
115    fn new() -> Self {
116        Self {
117            virtual_pointers: [0; _],
118            positions: [[Position::ZERO; 2]; _],
119        }
120    }
121
122    /// # Safety
123    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted. `NextPhysicalPointer` and
124    /// `Rmap` must have 1:1 mapping and not mixed with anything else.
125    #[inline(always)]
126    fn insertion_item_physical_pointer(
127        &mut self,
128        rmap_bit_position: RmapBitPosition,
129        next_physical_pointer: &mut NextPhysicalPointer,
130    ) -> u32 {
131        let bit_position = rmap_bit_position.get();
132        let word_offset = (bit_position / u32::BITS) as usize;
133        let bit_offset = bit_position % u32::BITS;
134
135        // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
136        let mut word = *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) };
137
138        if bit_offset + POINTER_BITS > u32::BITS {
139            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
140            let mut word_next =
141                *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset + 1) };
142            {
143                let value = (word >> bit_offset) | (word_next << (u32::BITS - bit_offset));
144                let virtual_pointer = value & (u32::MAX >> (u32::BITS - POINTER_BITS));
145
146                if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
147                    return physical_pointer;
148                }
149            }
150
151            let physical_pointer = next_physical_pointer.inc();
152            let virtual_pointer = physical_pointer + 1;
153
154            word |= virtual_pointer << bit_offset;
155            word_next |= virtual_pointer >> (u32::BITS - bit_offset);
156
157            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) } = word;
158            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset + 1) } = word_next;
159
160            physical_pointer
161        } else {
162            {
163                let virtual_pointer =
164                    (word >> bit_offset) & (u32::MAX >> (u32::BITS - POINTER_BITS));
165
166                if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
167                    return physical_pointer;
168                }
169            }
170
171            let physical_pointer = next_physical_pointer.inc();
172            let virtual_pointer = physical_pointer + 1;
173
174            word |= virtual_pointer << bit_offset;
175
176            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) } = word;
177
178            physical_pointer
179        }
180    }
181
182    /// # Safety
183    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted. `NextPhysicalPointer` and
184    /// `Rmap` must have 1:1 mapping and not mixed with anything else.
185    #[inline(always)]
186    unsafe fn insertion_item(
187        &mut self,
188        rmap_bit_position: RmapBitPosition,
189        next_physical_pointer: &mut NextPhysicalPointer,
190    ) -> &mut [Position; 2] {
191        let physical_pointer =
192            self.insertion_item_physical_pointer(rmap_bit_position, next_physical_pointer);
193        // SAFETY: Internal pointers are always valid
194        unsafe { self.positions.get_unchecked_mut(physical_pointer as usize) }
195    }
196
197    /// Note that `position == Position::ZERO` is effectively ignored here, supporting it cost too
198    /// much in terms of performance and not required for correctness.
199    ///
200    /// # Safety
201    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted. `NextPhysicalPointer` and
202    /// `Rmap` must have 1:1 mapping and not mixed with anything else.
203    #[inline(always)]
204    pub(super) unsafe fn add(
205        &mut self,
206        rmap_bit_position: RmapBitPosition,
207        position: Position,
208        next_physical_pointer: &mut NextPhysicalPointer,
209    ) {
210        // SAFETY: Guaranteed by function contract
211        let rmap_item = unsafe { self.insertion_item(rmap_bit_position, next_physical_pointer) };
212
213        // The same `r` can appear in the table multiple times, one duplicate is supported here
214        if rmap_item[0] == Position::ZERO {
215            rmap_item[0] = position;
216        } else if rmap_item[1] == Position::ZERO {
217            rmap_item[1] = position;
218        }
219    }
220
221    #[inline(always)]
222    pub(super) fn get(&self, rmap_bit_position: RmapBitPosition) -> [Position; 2] {
223        let bit_position = rmap_bit_position.get();
224        let word_offset = (bit_position / u32::BITS) as usize;
225        let bit_offset = bit_position % u32::BITS;
226
227        let virtual_pointer = if bit_offset + POINTER_BITS > u32::BITS {
228            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
229            let word = unsafe { *self.virtual_pointers.get_unchecked(word_offset) };
230            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
231            let word_next = unsafe { *self.virtual_pointers.get_unchecked(word_offset + 1) };
232
233            let value = (word >> bit_offset) | (word_next << (u32::BITS - bit_offset));
234            value & (u32::MAX >> (u32::BITS - POINTER_BITS))
235        } else {
236            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
237            let word = unsafe { *self.virtual_pointers.get_unchecked(word_offset) };
238
239            (word >> bit_offset) & (u32::MAX >> (u32::BITS - POINTER_BITS))
240        };
241
242        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
243            // SAFETY: Internal pointers are always valid
244            *unsafe { self.positions.get_unchecked(physical_pointer as usize) }
245        } else {
246            [Position::ZERO; 2]
247        }
248    }
249
250    // TODO: Remove as soon as non-hacky version compiles
251    #[inline(always)]
252    pub(super) fn zeroing_hack(rmap: &mut MaybeUninit<Self>) {
253        let rmap = unsafe { rmap.assume_init_mut() };
254        for index in 0..rmap.positions.len() {
255            rmap.positions[index][0] = Position::ZERO;
256            rmap.positions[index][1] = Position::ZERO;
257        }
258        for index in 0..rmap.virtual_pointers.len() {
259            rmap.virtual_pointers[index] = 0;
260        }
261    }
