16/// Number of bits necessary to address a single pair of positions in the rmap.
17///
18/// `+1` is used because virtual pointers used for storage are increased by `1`
19const VIRTUAL_POINTER_BITS: u32 = (REDUCED_BUCKET_SIZE + 1).bit_width();
20const POINTERS_BITS: usize = PARAM_BC as usize * VIRTUAL_POINTER_BITS as usize;
21const POINTERS_WORDS: usize = POINTERS_BITS.div_ceil(u32::BITS as usize);

24const _: () = assert!(VIRTUAL_POINTER_BITS <= u32::BITS);

91/// Preparation state that combines a bunch of separate tracking values into a compact
92/// representation to minimize registers usage to maintain reasonable occupancy
93#[derive(Debug)]
94struct RAccumulator {
95    /// Accumulates `r`, virtual pointer, and a flag whether `r` duplicate was found
96    r_accumulator: u32,
97}
98
99impl Default for RAccumulator {
100    #[inline(always)]
101    fn default() -> Self {
102        Self {
103            // TODO: `const {}` is a workaround for https://github.com/Rust-GPU/rust-gpu/issues/322 and
104            //  shouldn't be necessary otherwise
105            r_accumulator: const { u32::MAX >> (u32::BITS - (PARAM_BC - 1).bit_width()) },
106        }
107    }
108}
109
110impl RAccumulator {
111    #[inline(always)]
112    fn get_additional_data(&self) -> u32 {
113        self.r_accumulator >> (u32::BITS - (VIRTUAL_POINTER_BITS + 1))
114    }
115
116    #[inline(always)]
117    fn set_r_duplicate(&mut self) {
118        let r_duplicate = 1 << (u32::BITS - (VIRTUAL_POINTER_BITS + 1));
119        self.r_accumulator |= r_duplicate;
120    }
121
122    #[inline(always)]
123    fn has_r_duplicate(&self) -> bool {
124        let r_duplicate = 1 << (u32::BITS - (VIRTUAL_POINTER_BITS + 1));
125        (self.r_accumulator & r_duplicate) != 0
126    }
127
128    #[inline(always)]
129    fn set_r(&mut self, r: u32) {
130        // Statically ensure all 3 components fit into `u32`
131        const {
132            assert!(VIRTUAL_POINTER_BITS + 1 + (PARAM_BC - 1).bit_width() <= u32::BITS);
133        }
134        // Clear everything except the virtual pointer
135        self.r_accumulator &= u32::MAX << (u32::BITS - VIRTUAL_POINTER_BITS);
136        // Increment virtual pointer
137        let virtual_pointer_increment = 1 << (u32::BITS - VIRTUAL_POINTER_BITS);
138        self.r_accumulator += virtual_pointer_increment;
139        // Set `r`
140        self.r_accumulator |= r;
141    }
142
143    #[inline(always)]
144    fn get_r(&self) -> u32 {
145        // TODO: `const {}` is a workaround for https://github.com/Rust-GPU/rust-gpu/issues/322 and
146        //  shouldn't be necessary otherwise
147        self.r_accumulator & const { u32::MAX >> (u32::BITS - (PARAM_BC - 1).bit_width()) }
148    }
149}

153    /// `0` is a sentinel value indicating no virtual pointer is stored yet.
154    ///
155    /// Physical pointer must be increased by `1` to get a virtual pointer before storing. Virtual
156    /// pointer must be decreased by `1` before reading to get a physical pointer.
157    virtual_pointers: [u32; POINTERS_WORDS],
158    positions: [[Position; 2]; REDUCED_BUCKET_SIZE],
159}

161impl Rmap {
162    #[cfg(test)]
163    #[inline(always)]
164    pub(in super::super) fn new() -> Self {
165        Self {
166            virtual_pointers: [0; _],
167            positions: [[Position::ZERO; 2]; _],
168        }

171    /// # Safety
172    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted. `NextPhysicalPointer` and
173    /// `Rmap` must have 1:1 mapping and not mixed with anything else.
174    #[cfg(test)]
175    #[inline(always)]
176    fn insertion_item_physical_pointer(
177        &mut self,
178        rmap_bit_position: RmapBitPosition,
179        next_physical_pointer: &mut NextPhysicalPointer,
180    ) -> u32 {
181        let bit_position = rmap_bit_position.get();
182        let word_offset = (bit_position / u32::BITS) as usize;
183        let bit_offset = bit_position % u32::BITS;
184
185        // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
186        let mut word = *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) };
187
188        if bit_offset + VIRTUAL_POINTER_BITS > u32::BITS {
189            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
190            let mut word_next =
191                *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset + 1) };
192            {
193                let value = (word >> bit_offset) | (word_next << (u32::BITS - bit_offset));
194                let virtual_pointer = value & (u32::MAX >> (u32::BITS - VIRTUAL_POINTER_BITS));
195
196                if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
197                    return physical_pointer;
198                }
199            }
200
201            let physical_pointer = next_physical_pointer.inc();
202            let virtual_pointer = physical_pointer + 1;
203
204            word |= virtual_pointer << bit_offset;
205            word_next |= virtual_pointer >> (u32::BITS - bit_offset);
206
207            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
208            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) } = word;
209            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
210            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset + 1) } = word_next;
211
212            physical_pointer
213        } else {
214            {
215                let virtual_pointer =
216                    (word >> bit_offset) & (u32::MAX >> (u32::BITS - VIRTUAL_POINTER_BITS));
217
218                if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
219                    return physical_pointer;
220                }
221            }
222
223            let physical_pointer = next_physical_pointer.inc();
224            let virtual_pointer = physical_pointer + 1;
225
226            word |= virtual_pointer << bit_offset;
227
228            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
229            *unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) } = word;
230
231            physical_pointer
232        }
233    }

235    /// Note that `position == Position::ZERO` is effectively ignored here, supporting it cost too
236    /// much in terms of performance and not required for correctness.
237    ///
238    /// # Safety
239    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted. `NextPhysicalPointer` and
240    /// `Rmap` must have 1:1 mapping and not mixed with anything else.
241    #[cfg(test)]
242    #[inline(always)]
243    pub(super) unsafe fn add(
244        &mut self,
245        rmap_bit_position: RmapBitPosition,
246        position: Position,
247        next_physical_pointer: &mut NextPhysicalPointer,
248    ) {
249        let physical_pointer =
250            self.insertion_item_physical_pointer(rmap_bit_position, next_physical_pointer);
251        // SAFETY: Internal pointers are always valid
252        let rmap_item = unsafe { self.positions.get_unchecked_mut(physical_pointer as usize) };
253
254        // The same `r` can appear in the table multiple times, one duplicate is supported here
255        if rmap_item[0] == Position::ZERO {
256            rmap_item[0] = position;
257        } else if rmap_item[1] == Position::ZERO {
258            rmap_item[1] = position;
259        }
260    }

262    /// Add using `r` that was previously updated using [`Self::update_local_bucket_r_data()`] in a
263    /// fully parallel way (all threads can contribute at once).
264    ///
265    /// # Safety
266    /// `r` elements must be updated using [`Self::update_local_bucket_r_data()`].
267    pub(super) unsafe fn add_with_data_parallel(&mut self, r: R, position: Position) {
268        let (r, data) = r.split();
269        if data == 0 {
270            // No virtual pointer here, hence this is a duplicate that should be ignored
271            return;
272        }
273        let rmap_offset = data & 1;
274        let virtual_pointer = data >> 1;
275
276        // SAFETY: `r` is obtained from the `R` instance and thus must be valid
277        let rmap_bit_position = unsafe { RmapBitPosition::new(r) };
278
279        let bit_position = rmap_bit_position.get();
280        let word_offset = (bit_position / u32::BITS) as usize;
281        let bit_offset = bit_position % u32::BITS;
282
283        // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
284        let word = unsafe { self.virtual_pointers.get_unchecked_mut(word_offset) };
285        // TODO: Probably should not be unsafe to begin with:
286        //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
287        #[cfg(target_arch = "spirv")]
288        unsafe {
289            atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
290                word,
291                virtual_pointer << bit_offset,
292            );
293        }
294        #[cfg(not(target_arch = "spirv"))]
295        {
296            *word |= virtual_pointer << bit_offset;
297        }
298
299        if bit_offset + VIRTUAL_POINTER_BITS > u32::BITS {
300            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
301            let word_next = unsafe { self.virtual_pointers.get_unchecked_mut(word_offset + 1) };
302            // TODO: Probably should not be unsafe to begin with:
303            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
304            #[cfg(target_arch = "spirv")]
305            unsafe {
306                atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
307                    word_next,
308                    virtual_pointer >> (u32::BITS - bit_offset),
309                );
310            }
311            #[cfg(not(target_arch = "spirv"))]
312            {
313                *word_next |= virtual_pointer >> (u32::BITS - bit_offset);
314            }
315        }
316
317        let physical_pointer = virtual_pointer - 1;
318
319        // SAFETY: Internal pointers are always valid
320        let rmap_item = unsafe { self.positions.get_unchecked_mut(physical_pointer as usize) };
321        rmap_item[rmap_offset as usize] = position;
322    }

324    #[inline(always)]
325    pub(super) fn get(&self, rmap_bit_position: RmapBitPosition) -> [Position; 2] {
326        let bit_position = rmap_bit_position.get();
327        let word_offset = (bit_position / u32::BITS) as usize;
328        let bit_offset = bit_position % u32::BITS;
329
330        let virtual_pointer = if bit_offset + VIRTUAL_POINTER_BITS > u32::BITS {
331            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
332            let word = unsafe { *self.virtual_pointers.get_unchecked(word_offset) };
333            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
334            let word_next = unsafe { *self.virtual_pointers.get_unchecked(word_offset + 1) };
335
336            let value = (word >> bit_offset) | (word_next << (u32::BITS - bit_offset));
337            value & (u32::MAX >> (u32::BITS - VIRTUAL_POINTER_BITS))
338        } else {
339            // SAFETY: Offset comes from `RmapBitPosition`, whose constructor guarantees bounds
340            let word = unsafe { *self.virtual_pointers.get_unchecked(word_offset) };
341
342            (word >> bit_offset) & (u32::MAX >> (u32::BITS - VIRTUAL_POINTER_BITS))
343        };
344
345        if let Some(physical_pointer) = virtual_pointer.checked_sub(1) {
346            // SAFETY: Internal pointers are always valid
347            *unsafe { self.positions.get_unchecked(physical_pointer as usize) }
348        } else {
349            [Position::ZERO; 2]
350        }
351    }

354    #[inline(always)]
355    pub(super) fn zeroing_hack(
356        rmap: &mut MaybeUninit<Self>,
357        local_invocation_id: u32,
358        workgroup_size: u32,
359    ) {
360        let rmap = unsafe { rmap.assume_init_mut() };
361
362        for i in (local_invocation_id..POINTERS_WORDS as u32).step_by(workgroup_size as usize) {
363            rmap.virtual_pointers[i as usize] = 0;
364        }
365
366        const {
367            assert!(REDUCED_BUCKET_SIZE.is_multiple_of(2));
368        }
369        let pair_id = local_invocation_id / 2;
370        let pair_offset = local_invocation_id % 2;
371        for bucket in (pair_id..REDUCED_BUCKET_SIZE as u32).step_by((workgroup_size / 2) as usize) {
372            rmap.positions[bucket as usize][pair_offset as usize] = Position::ZERO;
373        }
374    }

376    // TODO: It should be possible to optimize this further with parallelism instead of (currently)
377    //  sequential version
378    /// This prepares the local bucket by appending extra information in `R`'s data field, such that
379    /// `Rmap` can later be constructed in parallel rather than sequentially.
380    ///
381    /// Each `local_bucket` stores its slice of elements of the whole bucket. `position` field is
382    /// checked for zero and sentinel values (checking functions are generic so that extra bits can
383    /// be repurposed for additional information), but otherwise is not interpreted or modified.
384    ///
385    /// NOTE: For this to work correctly, all local buckets together must be sorted by `r` and
386    /// `position` among `r` duplicates. `r` must not store additional data in it yet.
387    ///
388    /// # Safety
389    /// There must be at most [`REDUCED_BUCKET_SIZE`] items inserted.
390    pub(in super::super) unsafe fn update_local_bucket_r_data<
391        const ELEMENTS_PER_THREAD: usize,
392        IsZeroPosition,
393        IsSentinelPosition,
394    >(
395        lane_id: u32,
396        subgroup_size: u32,
397        local_bucket: &mut [PositionR; ELEMENTS_PER_THREAD],
398        is_zero_position: IsZeroPosition,
399        is_sentinel_position: IsSentinelPosition,
400    ) where
401        IsZeroPosition: Fn(Position) -> bool,
402        IsSentinelPosition: Fn(Position) -> bool,
403    {
404        let mut preparation_state = RAccumulator::default();
405
406        // TODO: More idiomatic version currently doesn't compile:
407        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
408        #[expect(
409            clippy::needless_range_loop,
410            reason = "rust-gpu can't compile idiomatic version"
411        )]
412        for source_lane in 0..subgroup_size {
413            for local_offset in 0..ELEMENTS_PER_THREAD {
414                #[cfg(target_arch = "spirv")]
415                let position = subgroup_shuffle(local_bucket[local_offset].position, source_lane);
416                #[cfg(not(target_arch = "spirv"))]
417                let position = {
418                    assert_eq!(lane_id, 0);
419                    assert_eq!(subgroup_size, 1);
420
421                    local_bucket[local_offset].position
422                };
423                if is_sentinel_position(position) {
424                    return;
425                }
426                if is_zero_position(position) {
427                    // This is to match the sequential version
428                    continue;
429                }
430
431                #[cfg(target_arch = "spirv")]
432                let r = subgroup_shuffle(local_bucket[local_offset].r.get_inner(), source_lane);
433                #[cfg(not(target_arch = "spirv"))]
434                let r = {
435                    assert_eq!(lane_id, 0);
436                    assert_eq!(subgroup_size, 1);
437
438                    local_bucket[local_offset].r.get_inner()
439                };
440
441                if preparation_state.get_r() == r {
442                    if preparation_state.has_r_duplicate() {
443                        // One `r` duplicate was already processed, all others are skipped
444                        continue;
445                    }
446                    preparation_state.set_r_duplicate();
447                } else {
448                    preparation_state.set_r(r);
449                }
450
451                if lane_id == source_lane {
452                    #[expect(clippy::int_plus_one, reason = "This describes the invariant exactly")]
453                    const {
454                        assert!(VIRTUAL_POINTER_BITS + 1 <= u32::BITS - (PARAM_BC - 1).bit_width());
455                    }
456                    // `r` is valid according to the function contract, `data` part is statically
457                    // asserted to fit above
458                    local_bucket[local_offset].r =
459                        unsafe { R::new_with_data(r, preparation_state.get_additional_data()) };
460
461                    #[cfg(not(target_arch = "spirv"))]
462                    {
463                        assert_eq!(r, local_bucket[local_offset].r.split().0);
464                        assert_eq!(
465                            preparation_state.get_additional_data(),
466                            local_bucket[local_offset].r.split().1
467                        );
468                    }
469                }
470            }
471        }
472    }
