11const PRESENCE_FLAGS_WORDS: usize = (PARAM_BC as usize * 2).div_ceil(u32::BITS as usize);
12
13#[derive(Debug, Copy, Clone)]
14pub(super) struct Rmap {
15    /// Store two bits per target, indicating whether matching elements are present (two at most)
16    presence_flags: [u32; PRESENCE_FLAGS_WORDS],
17}
18
19impl Rmap {
20    #[cfg(test)]
21    #[inline(always)]
22    fn new() -> Self {
23        Self {
24            presence_flags: [0; _],
25        }
26    }
27
28    /// Returns `0`, `1` or `2` depending on whether `r` was present and whether there were
29    /// duplicates.
30    ///
31    /// `0` means not present, `1` means exactly one value. `2` means two or more values, but
32    /// doesn't track how many exactly.
33    pub(super) fn num_r_items(&self, r_target: R) -> u32 {
34        let bit_position = r_target.get() * 2;
35        let word_offset = (bit_position / u32::BITS) as usize;
36        let bit_offset = bit_position % u32::BITS;
37        // SAFETY: `bit_position` is within bounds of presence flags
38        let word = *unsafe { self.presence_flags.get_unchecked(word_offset) };
39
40        ((word >> bit_offset) & 0b11).count_ones()
41    }
42
43    pub(super) fn add_with_data_parallel(&mut self, r: R) {
44        let bit_position = r.get() * 2;
45        let word_offset = (bit_position / u32::BITS) as usize;
46        let bit_offset = bit_position % u32::BITS;
47        // SAFETY: `bit_position` is within bounds of presence flags
48        let word = unsafe { self.presence_flags.get_unchecked_mut(word_offset) };
49        let mask = 1 << bit_offset;
50
51        if cfg!(target_arch = "spirv") {
52            // SAFETY: TODO: Probably should not be unsafe to begin with:
53            //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
54            let prev_word_value = unsafe {
55                atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(word, mask)
56            };
57
58            if prev_word_value & mask != 0 {
59                // Bit was already set, so this is not the first such `r` value, set a flag
60                // indicating there was a duplicate
61                // SAFETY: TODO: Probably should not be unsafe to begin with:
62                //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
63                unsafe {
64                    atomic_or::<_, { Scope::Workgroup as u32 }, { Semantics::NONE.bits() }>(
65                        word,
66                        mask << 1,
67                    )
68                };
69            }
70        } else if *word & mask != 0 {
71            // Bit was already set, so this is not the first such `r` value, set a flag
72            // indicating there was a duplicate
73            *word |= mask << 1;
74        } else {
75            *word |= mask;
76        }
77    }
78}
