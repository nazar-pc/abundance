14#[inline(always)]
15fn perform_local_compare_swap<const ELEMENTS_PER_THREAD: usize>(
16    lane_id: u32,
17    bit_position: u32,
18    block_size: usize,
19    local_data: &mut [PositionY; ELEMENTS_PER_THREAD],
20) {
21    // For local swaps within a thread's register data, iterate over half the elements to form pairs
22    // `(a_offset, b_offset)` where indices differ only at `bit_position` and swaps them in-place
23    for pair_id in 0..ELEMENTS_PER_THREAD / 2 {
24        // Compute pair indices using bit manipulation to map `pair_id`
25        // (`0..ELEMENTS_PER_THREAD / 2`) to sparse indices in `local_data`
26        // (`0..ELEMENTS_PER_THREAD`) where bit_position is `0` or `1`
27
28        // Bits above `bit_position
29        let high = (pair_id & ((u32::MAX as usize) << bit_position)) << 1;
30        // Bits below `bit_position`
31        let low = pair_id & (u32::MAX as usize).unbounded_shr(u32::BITS - bit_position);
32        let a_offset = high | low;
33        let b_offset = a_offset | (1 << bit_position);
34
35        let a = local_data[a_offset];
36        let b = local_data[b_offset];
37
38        // Determine the sort direction: ascending if `a_offset`'s bit at `block_size` is `0`.
39        // This alternates direction in bitonic merges to create sorted sequences.
40        let select_smaller =
41            ((lane_id as usize * ELEMENTS_PER_THREAD + a_offset) & block_size) == 0;
42
43        let (final_a, final_b) = if (a.position <= b.position) == select_smaller {
44            (a, b)
45        } else {
46            (b, a)
47        };
48
49        local_data[a_offset] = final_a;
50        local_data[b_offset] = final_b;
51    }
52}

54#[inline(always)]
55fn perform_cross_compare_swap<const ELEMENTS_PER_THREAD: usize>(
56    lane_id: u32,
57    bit_position: u32,
58    block_size: usize,
59    local_data: &mut [PositionY; ELEMENTS_PER_THREAD],
60) {
61    // For cross-subgroup swaps, compute partner lane differing at `lane_bit_position`
62    let lane_bit_position = bit_position - ELEMENTS_PER_THREAD.ilog2();
63    let partner_lane_id = lane_id ^ (1 << lane_bit_position);
64    // When this lane's bit at `lane_bit_position` is `0`, then this is a lower lane (when compared
65    // to `partner_lane_id`)
66    let is_low = (lane_id as usize & (1 << lane_bit_position)) == 0;
67    // If `block_size` bit is `0`, setting sort direction for bitonic merge.
68    let ascending = ((lane_id as usize * ELEMENTS_PER_THREAD) & block_size) == 0;
69    let select_smaller = is_low == ascending;
70
71    // TODO: More idiomatic version currently doesn't compile:
72    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
73    #[expect(
74        clippy::needless_range_loop,
75        reason = "rust-gpu can't compile idiomatic version"
76    )]
77    // Iterate over all elements in `local_data`, swapping with partner lane
78    for a_offset in 0..ELEMENTS_PER_THREAD {
79        let a = local_data[a_offset];
80        let b = {
81            let position = subgroup_shuffle(a.position, partner_lane_id);
82            let y = Y::from(subgroup_shuffle(u32::from(a.y), partner_lane_id));
83
84            PositionY { position, y }
85        };
86
87        let selected_value = if (a.position <= b.position) == select_smaller {
88            a
89        } else {
90            b
91        };
92
93        local_data[a_offset] = selected_value;
94    }
95}

98/// Sort a bucket using bitonic sort
99#[inline(always)]
100fn sort_bucket_impl<const ELEMENTS_PER_THREAD: usize>(
101    lane_id: u32,
102    bucket_size: u32,
103    bucket: &mut [PositionY; MAX_BUCKET_SIZE],
104) {
105    let mut local_data = [PositionY::EMPTY; ELEMENTS_PER_THREAD];
106
107    // TODO: More idiomatic version currently doesn't compile:
108    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
109    #[expect(
110        clippy::needless_range_loop,
111        reason = "rust-gpu can't compile idiomatic version"
112    )]
113    for local_offset in 0..ELEMENTS_PER_THREAD {
114        let bucket_offset = lane_id as usize * ELEMENTS_PER_THREAD + local_offset;
115        local_data[local_offset] = if bucket_offset < bucket_size as usize {
116            bucket[bucket_offset]
117        } else {
118            PositionY {
119                position: Position::SENTINEL,
120                y: Y::SENTINEL,
121            }
122        };
123    }
124
125    // Iterate over merger stages, doubling block_size each time
126    let mut block_size = 2;
127    let mut merger_stage = 1;
128    while block_size <= MAX_BUCKET_SIZE {
129        // For each stage, process bit positions in reverse for bitonic comparisons
130        for bit_position in (0..merger_stage).rev() {
131            if bit_position < ELEMENTS_PER_THREAD.ilog2() {
132                // Local swaps within thread's registers, no synchronization needed
133                perform_local_compare_swap(lane_id, bit_position, block_size, &mut local_data);
134            } else {
135                // Cross-lane swaps using subgroup shuffles for communication
136                perform_cross_compare_swap(lane_id, bit_position, block_size, &mut local_data);
137            }
138        }
139        block_size *= 2;
140        merger_stage += 1;
141    }
142
143    // TODO: More idiomatic version currently doesn't compile:
144    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
145    #[expect(
146        clippy::needless_range_loop,
147        reason = "rust-gpu can't compile idiomatic version"
148    )]
149    for local_offset in 0..ELEMENTS_PER_THREAD {
150        let bucket_offset = lane_id as usize * ELEMENTS_PER_THREAD + local_offset;
151        bucket[bucket_offset] = local_data[local_offset];
152    }
153}

155#[spirv(compute(threads(256), entry_point_name = "sort_buckets"))]
156#[expect(
157    clippy::too_many_arguments,
158    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
159)]
160pub fn sort_buckets(
161    #[spirv(workgroup_id)] workgroup_id: UVec3,
162    #[spirv(num_workgroups)] num_workgroups: UVec3,
163    #[spirv(subgroup_id)] subgroup_id: u32,
164    #[spirv(subgroup_size)] subgroup_size: u32,
165    #[spirv(num_subgroups)] num_subgroups: u32,
166    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
167    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] bucket_sizes: &[u32; NUM_BUCKETS],
168    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] buckets: &mut [[PositionY; MAX_BUCKET_SIZE];
169             NUM_BUCKETS],
170) {
171    let workgroup_id = workgroup_id.x;
172    let num_workgroups = num_workgroups.x;
173
174    let total_subgroups = num_workgroups * num_subgroups;
175    let global_subgroup_id = workgroup_id * num_subgroups + subgroup_id;
176
177    // Process one bucket per subgroup
178    // TODO: More idiomatic version currently doesn't compile:
179    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
180    for bucket_index in (global_subgroup_id..NUM_BUCKETS as u32).step_by(total_subgroups as usize) {
181        // Specify some common subgroup sizes so the driver can easily eliminate dead code. This is
182        // important because `local_data` inside the function is generic and impacts the number of
183        // registers used, so we want to minimize them.
184        match subgroup_size {
185            // Hypothetically possible
186            1 => {
187                sort_bucket_impl::<MAX_BUCKET_SIZE>(
188                    subgroup_local_invocation_id,
189                    bucket_sizes[bucket_index as usize],
190                    &mut buckets[bucket_index as usize],
191                );
192            }
193            // Hypothetically possible
194            2 => {
195                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 2 }>(
196                    subgroup_local_invocation_id,
197                    bucket_sizes[bucket_index as usize],
198                    &mut buckets[bucket_index as usize],
199                );
200            }
201            // LLVMpipe (Mesa 24, SSE)
202            4 => {
203                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 4 }>(
204                    subgroup_local_invocation_id,
205                    bucket_sizes[bucket_index as usize],
206                    &mut buckets[bucket_index as usize],
207                );
208            }
209            // LLVMpipe (Mesa 25, AVX/AVX2)
210            8 => {
211                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 8 }>(
212                    subgroup_local_invocation_id,
213                    bucket_sizes[bucket_index as usize],
214                    &mut buckets[bucket_index as usize],
215                );
216            }
217            // Raspberry PI 5
218            16 => {
219                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 16 }>(
220                    subgroup_local_invocation_id,
221                    bucket_sizes[bucket_index as usize],
222                    &mut buckets[bucket_index as usize],
223                );
224            }
225            // Intel/Nvidia
226            32 => {
227                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 32 }>(
228                    subgroup_local_invocation_id,
229                    bucket_sizes[bucket_index as usize],
230                    &mut buckets[bucket_index as usize],
231                );
232            }
233            // AMD
234            64 => {
235                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 64 }>(
236                    subgroup_local_invocation_id,
237                    bucket_sizes[bucket_index as usize],
238                    &mut buckets[bucket_index as usize],
239                );
240            }
241            // Hypothetically possible
242            128 => {
243                sort_bucket_impl::<{ MAX_BUCKET_SIZE / 128 }>(
244                    subgroup_local_invocation_id,
245                    bucket_sizes[bucket_index as usize],
246                    &mut buckets[bucket_index as usize],
247                );
248            }
249            _ => {
250                // https://registry.khronos.org/vulkan/specs/latest/man/html/SubgroupSize.html
251                unreachable!("All Vulkan targets use power of two and subgroup size <= 128")
252            }
253        }
254    }
