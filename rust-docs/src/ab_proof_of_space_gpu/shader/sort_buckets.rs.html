19#[inline(always)]
20fn perform_compare_swap<T, LessOrEqual>(
21    local_invocation_id: u32,
22    bit_position: u32,
23    block_size: usize,
24    shared_bucket: &mut [T; MAX_BUCKET_SIZE],
25    // TODO: Should have been just `fn()`, but https://github.com/Rust-GPU/rust-gpu/issues/452
26    less_or_equal: LessOrEqual,
27) where
28    T: Copy,
29    LessOrEqual: Fn(&T, &T) -> bool,
30{
31    // Take a pair `(a_offset, b_offset)` where indices differ only at `bit_position` and swaps them
32    let pair_id = local_invocation_id as usize;
33    // Bits above `bit_position`
34    let high = (pair_id >> bit_position) << (bit_position + 1);
35    // Bits below `bit_position`
36    let low = pair_id & (u32::MAX as usize).unbounded_shr(u32::BITS - bit_position);
37    let a_offset = high | low;
38    let b_offset = a_offset | (1 << bit_position);
39
40    // Determine the sort direction: ascending if `a_offset`'s bit at `block_size` is `0`.
41    // This alternates direction in bitonic merges to create sorted sequences.
42    let ascending = (a_offset & block_size) == 0;
43
44    let a = shared_bucket[a_offset];
45    let b = shared_bucket[b_offset];
46
47    // Only update when necessary
48    if less_or_equal(&a, &b) != ascending {
49        shared_bucket[a_offset] = b;
50        shared_bucket[b_offset] = a;
51    }
52}

54#[inline(always)]
55pub(super) fn sort_shared_bucket<T, LessOrEqual>(
56    local_invocation_id: u32,
57    shared_bucket: &mut [T; MAX_BUCKET_SIZE],
58    // TODO: Should have been just `fn()`, but https://github.com/Rust-GPU/rust-gpu/issues/452
59    less_or_equal: LessOrEqual,
60) where
61    T: Copy,
62    LessOrEqual: Fn(&T, &T) -> bool,
63{
64    // Iterate over merger stages, doubling block_size each time
65    for merger_stage in 1..=MAX_BUCKET_SIZE.ilog2() {
66        let block_size = 1 << merger_stage;
67        // For each stage, process bit positions in reverse for bitonic comparisons
68        for bit_position in (0..merger_stage).rev() {
69            perform_compare_swap(
70                local_invocation_id,
71                bit_position,
72                block_size,
73                shared_bucket,
74                &less_or_equal,
75            );
76
77            workgroup_memory_barrier_with_group_sync();
78        }
79    }
80}

82#[inline(always)]
83fn load_into_shared_bucket(
84    local_invocation_id: u32,
85    bucket_size: u32,
86    bucket: &[PositionR; MAX_BUCKET_SIZE],
87    shared_bucket: &mut [PositionR; MAX_BUCKET_SIZE],
88) {
89    for bucket_offset in
90        (local_invocation_id as usize..MAX_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
91    {
92        shared_bucket[bucket_offset] = if bucket_offset < bucket_size as usize {
93            bucket[bucket_offset]
94        } else {
95            PositionR::SENTINEL
96        };
97    }
98}

100#[inline(always)]
101fn store_from_shared_bucket(
102    local_invocation_id: u32,
103    bucket: &mut [PositionR; MAX_BUCKET_SIZE],
104    shared_bucket: &[PositionR; MAX_BUCKET_SIZE],
105) {
106    for bucket_offset in
107        (local_invocation_id as usize..MAX_BUCKET_SIZE).step_by(WORKGROUP_SIZE as usize)
108    {
109        bucket[bucket_offset] = shared_bucket[bucket_offset];
110    }
111}

113// TODO: Make unsafe and avoid bounds check
114/// Sort a bucket using bitonic sort
115fn sort_bucket_impl(
116    local_invocation_id: u32,
117    bucket_size: u32,
118    bucket: &mut [PositionR; MAX_BUCKET_SIZE],
119    shared_bucket: &mut [PositionR; MAX_BUCKET_SIZE],
120) {
121    load_into_shared_bucket(local_invocation_id, bucket_size, bucket, shared_bucket);
122
123    workgroup_memory_barrier_with_group_sync();
124
125    sort_shared_bucket(
126        local_invocation_id,
127        shared_bucket,
128        #[inline(always)]
129        |a, b| a.position <= b.position,
130    );
131
132    store_from_shared_bucket(local_invocation_id, bucket, shared_bucket);
133}

135/// NOTE: bucket sizes are zeroed after use
136#[spirv(compute(threads(256), entry_point_name = "sort_buckets"))]
137pub fn sort_buckets(
138    #[spirv(workgroup_id)] workgroup_id: UVec3,
139    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
140    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] bucket_sizes: &mut [u32; NUM_BUCKETS],
141    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)] buckets: &mut [[PositionR; MAX_BUCKET_SIZE];
142             NUM_BUCKETS],
143    #[spirv(workgroup)] shared_bucket: &mut [PositionR; MAX_BUCKET_SIZE],
144) {
145    let local_invocation_id = local_invocation_id.x;
146    let workgroup_id = workgroup_id.x;
147
148    // Process one bucket per workgroup
149    // TODO: More idiomatic version currently doesn't compile:
150    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
151    let bucket_size = bucket_sizes[workgroup_id as usize];
152    let bucket = &mut buckets[workgroup_id as usize];
153
154    sort_bucket_impl(local_invocation_id, bucket_size, bucket, shared_bucket);
155
156    if local_invocation_id == 0 {
157        bucket_sizes[workgroup_id as usize] = 0;
158    }
159}
