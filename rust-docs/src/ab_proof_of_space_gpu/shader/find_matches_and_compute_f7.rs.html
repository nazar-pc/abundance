33pub const NUM_ELEMENTS_PER_S_BUCKET: usize =

42    pub absolute_position: u32,

44    pub positions: [Position; 2],
45}

47/// Upper-bound estimation of the number of matched elements per s-bucket.
48///
49/// Buckets are defined by the lower `NUM_S_BUCKETS.ilog2()` bits of the values. This is based on a
50/// Chernoff bound for the Poisson distribution with mean `lambda = 1`, ensuring the probability
51/// that any bucket exceeds the bound is less than `2^{-security_bits}`. The bound is
52/// `lambda + ceil(sqrt(3 * lambda * (NUM_S_BUCKETS.ilog2() + security_bits) * ln(2)))`.
53/// Accounts for the filter to values in `0..NUM_S_BUCKETS-1` by using the expected number of
54/// remaining elements ~`NUM_S_BUCKETS`, distributed uniformly across all `NUM_S_BUCKETS` buckets.
55const fn proofs_bucket_upper_bound(security_bits: u8) -> u64 {
56    // Lambda is the expected number of entries in a bucket:
57    // ~`NUM_S_BUCKETS / NUM_S_BUCKETS = 1`
58    const LAMBDA: u64 = 1;
59    // Approximation of ln(2) as a fraction: `ln(2) â‰ˆ LN2_NUM / LN2_DEN`.
60    // This allows integer-only computation of the square root term involving ln(2).
61    const LN2_NUM: u128 = 693147;
62    const LN2_DEN: u128 = 1000000;
63
64    // `log2(NUM_S_BUCKETS) + security_bits` for the union bound over `NUM_S_BUCKETS` buckets
65    let ks = NUM_S_BUCKETS.ilog2() as u128 + security_bits as u128;
66    // Compute numerator for the expression under the square root:
67    // `3 * lambda * ks * LN2_NUM`
68    let num = 3u128 * LAMBDA as u128 * ks * LN2_NUM;
69    // Denominator for ln(2): `LN2_DEN`
70    let den = LN2_DEN;
71
72    let ceil_div = num.div_ceil(den);
73
74    // Binary search to find the smallest `x` such that `x * x >= ceil_div`,
75    // which computes `ceil(sqrt(num / den))` without floating-point.
76    // We use a custom binary search over `u64` range because binary search in the standard library
77    // operates on sorted slices, not directly on integer ranges for solving inequalities like this.
78    let mut low = 0u64;
79    let mut high = u64::MAX;
80    while low < high {
81        let mid = low + (high - low) / 2;
82        let left = (mid as u128) * (mid as u128);
83        if left >= ceil_div {
84            high = mid;
85        } else {
86            low = mid + 1;
87        }
88    }
89    let add_term = low;
90
91    LAMBDA + add_term
92}

108        &self[index]
109    }

117/// # Safety
118/// `bucket_index` must be within range `0..REDUCED_MATCHES_COUNT`. `matches_count` elements in
119/// `matches` must be initialized, `matches` must have valid pointers into `parent_metadatas`.
120#[inline(always)]
121unsafe fn compute_f7_into_buckets(
122    local_invocation_id: u32,
123    left_bucket_index: u32,
124    matches_count: usize,
125    // TODO: `&[Match]` would have been nicer, but it currently doesn't compile:
126    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
127    matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
128    // TODO: This should have been `&[[Metadata; REDUCED_MATCHES_COUNT]; NUM_MATCH_BUCKETS]`, but it
129    //  currently doesn't compile if flattened:
130    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
131    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
132    table_6_proof_targets_sizes: &mut [u32; NUM_S_BUCKETS],
133    table_6_proof_targets: &mut [[MaybeUninit<ProofTargets>; NUM_ELEMENTS_PER_S_BUCKET];
134             NUM_S_BUCKETS],
135) {
136    let absolute_position_base = left_bucket_index * REDUCED_MATCHES_COUNT as u32;
137
138    // TODO: More idiomatic version currently doesn't compile:
139    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
140    for index in (local_invocation_id..matches_count as u32).step_by(WORKGROUP_SIZE as usize) {
141        // SAFETY: Guaranteed by function contract
142        let m = unsafe { matches.get_unchecked(index as usize).assume_init() };
143        // TODO: Correct version currently doesn't compile:
144        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
145        // let left_metadata = parent_metadatas[usize::from(m.left_position)];
146        // let right_metadata = parent_metadatas[usize::from(m.right_position)];
147        // SAFETY: Guaranteed by function contract
148        let left_metadata = *unsafe { parent_metadatas.get_unchecked(m.left_position as usize) };
149        // SAFETY: Guaranteed by function contract
150        let right_metadata = *unsafe { parent_metadatas.get_unchecked(m.right_position as usize) };
151
152        let (y, _) = compute_fn_impl::<TABLE_NUMBER, PARENT_TABLE_NUMBER>(
153            m.left_y,
154            left_metadata,
155            right_metadata,
156        );
157
158        let s_bucket = y.first_k_bits() as usize;
159        // TODO: More idiomatic version currently doesn't compile:
160        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
161        // let Some(bucket_count) = bucket_sizes.get_mut(s_bucket) else {
162        //     continue;
163        // };
164        if s_bucket >= NUM_S_BUCKETS {
165            continue;
166        }
167        let bucket_size = &mut table_6_proof_targets_sizes[s_bucket];
168        // TODO: Probably should not be unsafe to begin with:
169        //  https://github.com/Rust-GPU/rust-gpu/pull/394#issuecomment-3316594485
170        let bucket_offset = unsafe {
171            atomic_i_increment::<_, { Scope::QueueFamily as u32 }, { Semantics::NONE.bits() }>(
172                bucket_size,
173            )
174        };
175
176        // TODO: Maybe store `absolute_position` separately from more densely populated positions,
177        //  then per-s-bucket `atomic_u_min()` can be done to store the minimum pointer. This all
178        //  will use less memory overall
179        // SAFETY: `s_bucket` is checked above to be correct. Bucket size upper bound is known
180        // statically to be [`NUM_ELEMENTS_PER_S_BUCKET`], so `bucket_offset` is also always within
181        // bounds.
182        unsafe {
183            table_6_proof_targets
184                .get_unchecked_mut(s_bucket)
185                .get_unchecked_mut(bucket_offset as usize)
186        }
187        .write(ProofTargets {
188            absolute_position: absolute_position_base + index,
189            positions: [m.left_position, m.right_position],
190        });
191    }
192}

194/// This is similar to `find_matches_and_compute_fn`, but it stores results in buckets grouped by
195/// s-buckets, which is how proofs can later be found efficiently.
196///
197/// Buckets need to be sorted by position afterward due to concurrent writes that do not have
198/// deterministic order. Content of the bucket beyond the size specified in `bucket_sizes` is
199/// undefined.
200///
201/// # Safety
202/// Must be called from [`WORKGROUP_SIZE`] threads. `num_subgroups` must be at most
203/// [`MAX_SUBGROUPS`]. All buckets must come from the `sort_buckets_with_rmap_details` shader.
204#[spirv(compute(threads(256), entry_point_name = "find_matches_and_compute_f7"))]
205#[expect(
206    clippy::too_many_arguments,
207    reason = "Both I/O and Vulkan stuff together take a lot of arguments"
208)]
209pub unsafe fn find_matches_and_compute_f7(
210    #[spirv(local_invocation_id)] local_invocation_id: UVec3,
211    #[spirv(workgroup_id)] workgroup_id: UVec3,
212    #[spirv(num_workgroups)] num_workgroups: UVec3,
213    #[spirv(subgroup_local_invocation_id)] subgroup_local_invocation_id: u32,
214    #[spirv(subgroup_id)] subgroup_id: u32,
215    #[spirv(num_subgroups)] num_subgroups: u32,
216    #[spirv(storage_buffer, descriptor_set = 0, binding = 0)] parent_buckets: &[[PositionR; MAX_BUCKET_SIZE];
217         NUM_BUCKETS],
218    #[spirv(storage_buffer, descriptor_set = 0, binding = 1)]
219    parent_metadatas: &[Metadata; REDUCED_MATCHES_COUNT * NUM_MATCH_BUCKETS],
220    #[spirv(storage_buffer, descriptor_set = 0, binding = 2)]
221    table_6_proof_targets_sizes: &mut [u32; NUM_S_BUCKETS],
222    #[spirv(storage_buffer, descriptor_set = 0, binding = 3)]
223    table_6_proof_targets: &mut [[MaybeUninit<ProofTargets>; NUM_ELEMENTS_PER_S_BUCKET];
224             NUM_S_BUCKETS],
225    #[spirv(workgroup)] matches: &mut [MaybeUninit<Match>; REDUCED_MATCHES_COUNT],
226    #[spirv(workgroup)] scratch_space: &mut SharedScratchSpace,
227    // Non-modern GPUs do not have enough space in the shared memory
228    #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
229    #[spirv(workgroup)]
230    rmap: &mut MaybeUninit<Rmap>,
231    #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
232    #[spirv(storage_buffer, descriptor_set = 0, binding = 4)]
233    rmap: &mut [MaybeUninit<Rmap>; MAX_SUBGROUPS],
234) {
235    let local_invocation_id = local_invocation_id.x;
236    let workgroup_id = workgroup_id.x;
237    let num_workgroups = num_workgroups.x;
238
239    // TODO: More idiomatic version currently doesn't compile:
240    //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
241    // for (left_bucket_index, (([left_bucket, right_bucket], matches), matches_count)) in buckets
242    //     .array_windows::<2>()
243    //     .zip(matches)
244    //     .zip(matches_counts)
245    //     .enumerate()
246    //     .skip(workgroup_id as usize)
247    //     .step_by(num_workgroups as usize)
248    // {
249    for left_bucket_index in
250        (workgroup_id as usize..NUM_MATCH_BUCKETS).step_by(num_workgroups as usize)
251    {
252        let left_bucket = &parent_buckets[left_bucket_index];
253        let right_bucket = &parent_buckets[left_bucket_index + 1];
254        let left_bucket_index = left_bucket_index as u32;
255
256        // TODO: Truncate buckets to reduced size here once it compiles:
257        //  https://github.com/Rust-GPU/rust-gpu/issues/241#issuecomment-3005693043
258        // SAFETY: Guaranteed by function contract
259        let matches_count = unsafe {
260            find_matches_in_buckets_impl(
261                subgroup_local_invocation_id,
262                subgroup_id,
263                num_subgroups,
264                local_invocation_id,
265                left_bucket_index,
266                left_bucket,
267                right_bucket,
268                matches,
269                scratch_space,
270                #[cfg(all(target_arch = "spirv", feature = "__modern-gpu"))]
271                rmap,
272                #[cfg(not(all(target_arch = "spirv", feature = "__modern-gpu")))]
273                &mut rmap[subgroup_id as usize],
274            )
275        };
276
277        workgroup_memory_barrier_with_group_sync();
278
279        unsafe {
280            compute_f7_into_buckets(
281                local_invocation_id,
282                left_bucket_index,
283                matches_count as usize,
284                matches,
285                parent_metadatas,
286                table_6_proof_targets_sizes,
287                table_6_proof_targets,
288            );
289        }
290
291        // No need for explicit synchronization, `matches` will not be touched before extra
292        // synchronization in `find_matches_in_buckets_impl` again anyway
293    }
