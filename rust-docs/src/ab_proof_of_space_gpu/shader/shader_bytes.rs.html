1use std::borrow::Cow;
2use std::slice;
3use wgpu::{ShaderModuleDescriptor, ShaderSource};
4
5#[repr(align(4))]
6pub(super) struct ShaderBytes<T>(pub(super) T)
7where
8    T: ?Sized;
9
10#[cfg(not(target_arch = "spirv"))]
11impl ShaderBytes<[u8]> {
12    pub(super) const fn to_module(&self) -> ShaderModuleDescriptor<'_> {
13        assert!(
14            u16::from_ne_bytes(1u16.to_le_bytes()) == 1u16,
15            "Only little-endian platform is supported"
16        );
17
18        assert!(align_of_val(self) == align_of::<u32>());
19        let shader_bytes = &self.0;
20
21        // SAFETY: Correctly aligned, all bit patterns are valid, lifetime is static before and after
22        let shader_bytes = unsafe {
23            slice::from_raw_parts(
24                shader_bytes.as_ptr().cast::<u32>(),
25                shader_bytes.len() / size_of::<u32>(),
26            )
27        };
28
29        ShaderModuleDescriptor {
30            label: Some(env!("CARGO_PKG_NAME")),
31            source: ShaderSource::SpirV(Cow::Borrowed(shader_bytes)),
32        }
33    }
34}
