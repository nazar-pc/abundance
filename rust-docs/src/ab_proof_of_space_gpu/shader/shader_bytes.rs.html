1use std::borrow::Cow;
2use std::slice;
3use wgpu::{ShaderModuleDescriptor, ShaderSource};
4
5#[repr(align(4))]
6pub(super) struct ShaderBytes<T>(pub(super) T)
7where
8    T: ?Sized;
9
10#[cfg(not(target_arch = "spirv"))]
11impl ShaderBytes<[u8]> {
12    pub(super) const fn to_module(&self) -> ShaderModuleDescriptor<'_> {
13        assert!(align_of_val(self) == align_of::<u32>());
14        let shader_bytes = &self.0;
15
16        // SAFETY: Correctly aligned, all bit patterns are valid, lifetime is static before and after
17        let shader_bytes = unsafe {
18            slice::from_raw_parts(
19                shader_bytes.as_ptr().cast::<u32>(),
20                shader_bytes.len() / size_of::<u32>(),
21            )
22        };
23
24        ShaderModuleDescriptor {
25            label: Some(env!("CARGO_PKG_NAME")),
26            source: ShaderSource::SpirV(Cow::Borrowed(shader_bytes)),
27        }
28    }
29}
