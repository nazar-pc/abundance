24pub(super) fn collect_enum_csr_impls_from_dependencies()
25-> impl Iterator<Item = anyhow::Result<(ItemImpl, Rc<Path>)>> {
26    // Collect exported instruction enums from dependencies
27    env::vars().filter_map(|(key, value)| {
28        if !key.ends_with(ENUM_CSR_IMPL_ENV_VAR_SUFFIX) {
29            return None;
30        }
31
32        let result = try {
33            let mut item_enum_contents = fs::read_to_string(&value).with_context(|| {
34                format!(
35                    "Failed to read Rust file `{value}` that is expected to contain instruction \
36                    enum csr implementation"
37                )
38            })?;
39            pre_process_rust_code(&mut item_enum_contents);
40            let item_impl = parse_str::<ItemImpl>(&item_enum_contents).with_context(|| {
41                format!(
42                    "Failed to parse Rust file `{value}` that is expected to contain instruction \
43                    enum csr implementation"
44                )
45            })?;
46
47            (item_impl, Rc::from(Path::new(&value)))
48        };
49
50        Some(result)
51    })
52}

156fn output_processed_enum_operands_impl(
157    enum_name: &Ident,
158    item_impl: ItemImpl,
159    out_dir: &Path,
160) -> anyhow::Result<()> {
161    let enum_file_path = out_dir.join(format!("{enum_name}_operands_impl.rs"));
162    let code = item_impl.to_token_stream().to_string();
163    // Format
164    let mut code = unparse(&parse_file(&code).expect("Generated code is valid; qed"));
165    post_process_rust_code(&mut code);
166    // This is a hack for pulling generics from decoding impl, which is `const`
167    let code = code.replace("[const] ", "");
168
169    // Avoid extra file truncation/override if it didn't change
170    if fs::read_to_string(&enum_file_path).ok().as_ref() != Some(&code) {
171        fs::write(&enum_file_path, code).with_context(|| {
172            format!(
173                "Failed to write generated Rust file with instruction operands implementation for \
174                `{enum_name}`",
175            )
176        })?;
177    }
178
179    Ok(())
180}

246pub(super) fn process_execution_impl(
247    mut item_impl: ItemImpl,
248    out_dir: &Path,
249    state: &mut State,
250) -> Option<anyhow::Result<()>> {
251    let attribute_index = item_impl
252        .attrs
253        .iter()
254        .enumerate()
255        .find_map(|(index, attr)| {
256            attr.meta
257                .path()
258                .is_ident("instruction_execution")
259                .then_some(index)
260        })?;
261    item_impl.attrs.remove(attribute_index);
262
263    let Some((_, trait_path, _)) = &item_impl.trait_ else {
264        return Some(Err(anyhow::anyhow!(
265            "Expected  `#[instruction_execution] impl ExecutableInstructionOperands for {0}` or \
266            `#[instruction_execution] impl ExecutableInstructionCsr for {0}` or \
267            `#[instruction_execution] impl ExecutableInstruction for {0}`, but no trait was found",
268            item_impl.self_ty.to_token_stream()
269        )));
270    };
271
272    let last_trait_segment_path = trait_path
273        .segments
274        .last()
275        .expect("Path is never empty; qed");
276
277    Some(
278        if last_trait_segment_path.ident == "ExecutableInstructionOperands" {
279            process_enum_operands_impl(item_impl, out_dir, state)
280        } else if last_trait_segment_path.ident == "ExecutableInstructionCsr" {
281            process_enum_csr_impl(item_impl, out_dir, state)
282        } else if last_trait_segment_path.ident == "ExecutableInstruction" {
283            process_enum_execution_impl(item_impl, out_dir, state)
284        } else {
285            Err(anyhow::anyhow!(
286                "Expected `impl` for `{}`, `#[instruction_execution]` attribute must be added to a \
287                trait implementation, but trait `{}` is not supported",
288                item_impl.self_ty.to_token_stream(),
289                last_trait_segment_path.ident
290            ))
291        },
292    )
293}

295pub(super) fn process_enum_operands_impl(
296    mut item_impl: ItemImpl,
297    out_dir: &Path,
298    state: &mut State,
299) -> anyhow::Result<()> {
300    let enum_name = enum_name_from_impl(&item_impl);
301
302    let Some(enum_definition) = state.get_known_enum_definition(&enum_name) else {
303        eprintln!("{enum_name} operands is waiting on its definition");
304        state.add_pending_enum_operands_impl(PendingEnumOperandsImpl { item_impl });
305        return Ok(());
306    };
307
308    let all_dependencies = match collect_all_dependencies(
309        state,
310        enum_definition.direct_dependencies.iter().cloned(),
311    ) {
312        Ok(all_dependencies) => all_dependencies,
313        Err(dependency_enum_name) => {
314            eprintln!("{enum_name} operands is waiting on {dependency_enum_name} definition");
315            state.add_pending_enum_operands_impl(PendingEnumOperandsImpl { item_impl });
316            return Ok(());
317        }
318    };
319
320    let mut all_where_predicates = Vec::new();
321
322    for (dependency_enum_name, _dependency_enum_definition) in all_dependencies {
323        let Some(dependency_enum_operands_impl) =
324            state.get_known_original_enum_decoding_impl(&dependency_enum_name)
325        else {
326            eprintln!(
327                "{enum_name} operands is waiting on {dependency_enum_name} decoding implementation"
328            );
329            state.add_pending_enum_operands_impl(PendingEnumOperandsImpl { item_impl });
330            return Ok(());
331        };
332
333        if let Some(where_clause) = &dependency_enum_operands_impl
334            .item_impl
335            .generics
336            .where_clause
337        {
338            all_where_predicates.extend(where_clause.predicates.iter().cloned());
339        }
340    }
341
342    {
343        let where_clause = item_impl
344            .generics
345            .where_clause
346            .get_or_insert(parse_quote! { where });
347        let mut already_inserted = where_clause
348            .predicates
349            .iter()
350            .cloned()
351            .collect::<HashSet<_>>();
352        for predicate in all_where_predicates {
353            if already_inserted.contains(&predicate) {
354                continue;
355            }
356            already_inserted.insert(predicate.clone());
357            where_clause.predicates.push(predicate);
358        }
359    }
360
361    item_impl.items.push({
362        let all_instructions = enum_definition
363            .instructions
364            .iter()
365            .map(|variant| &variant.ident);
366
367        parse_quote! {
368            #[inline(always)]
369            fn get_rs1_rs2_operands(self) -> Rs1Rs2Operands<Self::Reg> {
370                match self {
371                    #( Self::#all_instructions { rs1, rs2, .. } => Rs1Rs2Operands { rs1, rs2 }, )*
372                }
373            }
374        }
375    });
376
377    output_processed_enum_operands_impl(&enum_name, item_impl, out_dir)
378}

380pub(super) fn process_enum_csr_impl(
381    mut item_impl: ItemImpl,
382    out_dir: &Path,
383    state: &mut State,
384) -> anyhow::Result<()> {
385    let enum_name = enum_name_from_impl(&item_impl);
386
387    let Some(enum_definition) = state.get_known_enum_definition(&enum_name) else {
388        eprintln!("{enum_name} CSR is waiting on its definition");
389        state.add_pending_enum_csr_impl(PendingEnumCsrImpl { item_impl });
390        return Ok(());
391    };
392
393    let all_dependencies = match collect_all_dependencies(
394        state,
395        enum_definition.direct_dependencies.iter().cloned(),
396    ) {
397        Ok(all_dependencies) => all_dependencies,
398        Err(dependency_enum_name) => {
399            eprintln!("{enum_name} CSR is waiting on {dependency_enum_name} definition");
400            state.add_pending_enum_csr_impl(PendingEnumCsrImpl { item_impl });
401            return Ok(());
402        }
403    };
404
405    let mut all_prepare_csr_read_fns = Vec::new();
406    let mut all_prepare_csr_write_fns = Vec::new();
407    let mut all_where_predicates = Vec::new();
408
409    // TODO: This should be changed to recursively combine all fundamental extensions instead of
410    //  combined ones instead, such that extension D that depends two extensions B and C that both
411    //  depend on the same extension A will get A included in D once rather than twice. This is
412    //  caused by `get_known_enum_csr_impl` returning final extension implementation and not
413    //  its original state as defined in the source code. Essentially, in addition to the final
414    //  version, the original body of the implementation needs to be retained and used.
415    for (dependency_enum_name, _dependency_enum_definition) in all_dependencies {
416        let Some(dependency_enum_csr_impl) = state.get_known_enum_csr_impl(&dependency_enum_name)
417        else {
418            eprintln!("{enum_name} CSR is waiting on {dependency_enum_name} CSR implementation");
419            state.add_pending_enum_csr_impl(PendingEnumCsrImpl { item_impl });
420            return Ok(());
421        };
422
423        all_prepare_csr_read_fns.extend(extract_prepare_csr_read_fn(
424            &dependency_enum_csr_impl.item_impl,
425        ));
426        all_prepare_csr_write_fns.extend(extract_prepare_csr_write_fn(
427            &dependency_enum_csr_impl.item_impl,
428        ));
429
430        if let Some(where_clause) = &dependency_enum_csr_impl.item_impl.generics.where_clause {
431            all_where_predicates.extend(where_clause.predicates.iter().cloned());
432        }
433    }
434
435    {
436        let where_clause = item_impl
437            .generics
438            .where_clause
439            .get_or_insert(parse_quote! { where });
440        let mut already_inserted = where_clause
441            .predicates
442            .iter()
443            .cloned()
444            .collect::<HashSet<_>>();
445        for predicate in all_where_predicates {
446            if already_inserted.contains(&predicate) {
447                continue;
448            }
449            already_inserted.insert(predicate.clone());
450            where_clause.predicates.push(predicate);
451        }
452    }
453
454    // Composition for `prepare_csr_read` method
455    let maybe_prepare_csr_read_fn = extract_prepare_csr_read_fn_mut(&mut item_impl);
456    if let Some(prepare_csr_read_fn) = maybe_prepare_csr_read_fn {
457        (!block_contains_forbidden_syntax(&prepare_csr_read_fn.block)).ok_or_else(|| {
458            anyhow::anyhow!(
459                "Expected `#[instruction_execution] impl ExecutableInstructionCsr for {enum_name}` \
460                must not have `return` in `prepare_csr_read` method"
461            )
462        })?;
463
464        let all_prepare_csr_read_blocks = all_prepare_csr_read_fns
465            .iter()
466            .map(|impl_item_fn| &impl_item_fn.block)
467            .chain(iter::once(&prepare_csr_read_fn.block));
468        prepare_csr_read_fn.attrs.push(parse_quote! {
469            #[expect(clippy::allow_attributes, reason = "Attribute below")]
470        });
471        prepare_csr_read_fn.attrs.push(parse_quote! {
472            #[allow(clippy::try_err, reason = "Macro requirement")]
473        });
474        prepare_csr_read_fn.block = parse_quote! {{
475            let mut accepted_by_at_least_one = false;
476            #( if #all_prepare_csr_read_blocks? { accepted_by_at_least_one = true; } )*
477
478            Ok(accepted_by_at_least_one)
479        }};
480    } else if let Some(&first_prepare_csr_read_fn) = all_prepare_csr_read_fns.first() {
481        let all_prepare_csr_read_blocks = all_prepare_csr_read_fns
482            .iter()
483            .map(|impl_item_fn| &impl_item_fn.block);
484        let mut base_fn = first_prepare_csr_read_fn.clone();
485        base_fn.block = parse_quote! {{
486            let mut accepted_by_at_least_one = false;
487            #( if #all_prepare_csr_read_blocks? { accepted_by_at_least_one = true; } )*
488
489            Ok(accepted_by_at_least_one)
490        }};
491        item_impl.items.push(ImplItem::Fn(base_fn));
492    }
493
494    // Composition for `prepare_csr_write` method
495    let maybe_prepare_csr_write_fn = extract_prepare_csr_write_fn_mut(&mut item_impl);
496    if let Some(prepare_csr_write_fn) = maybe_prepare_csr_write_fn {
497        (!block_contains_forbidden_syntax(&prepare_csr_write_fn.block)).ok_or_else(|| {
498            anyhow::anyhow!(
499                "Expected `#[instruction_execution] impl ExecutableInstructionCsr for {enum_name}` \
500                must not have `return` in `prepare_csr_write` method",
501            )
502        })?;
503
504        let all_prepare_csr_write_blocks = all_prepare_csr_write_fns
505            .iter()
506            .map(|impl_item_fn| &impl_item_fn.block)
507            .chain(iter::once(&prepare_csr_write_fn.block));
508        prepare_csr_write_fn.attrs.push(parse_quote! {
509            #[expect(clippy::allow_attributes, reason = "Attribute below")]
510        });
511        prepare_csr_write_fn.attrs.push(parse_quote! {
512            #[allow(clippy::try_err, reason = "Macro requirement")]
513        });
514        prepare_csr_write_fn.block = parse_quote! {{
515            let mut accepted_by_at_least_one = false;
516            #( if #all_prepare_csr_write_blocks? { accepted_by_at_least_one = true; } )*
517
518            Ok(accepted_by_at_least_one)
519        }};
520    } else if let Some(&first_prepare_csr_write_fn) = all_prepare_csr_write_fns.first() {
521        let all_prepare_csr_write_blocks = all_prepare_csr_write_fns
522            .iter()
523            .map(|impl_item_fn| &impl_item_fn.block);
524        let mut base_fn = first_prepare_csr_write_fn.clone();
525        base_fn.block = parse_quote! {{
526            let mut accepted_by_at_least_one = false;
527            #( if #all_prepare_csr_write_blocks? { accepted_by_at_least_one = true; } )*
528
529            Ok(accepted_by_at_least_one)
530        }};
531        item_impl.items.push(ImplItem::Fn(base_fn));
532    }
533
534    // Comments will be stripped, this will suppress some of the lints that are caused by it
535    item_impl.attrs.extend([
536        parse_quote! { #[expect(clippy::allow_attributes, reason = "Attribute below")] },
537        parse_quote! { #[allow(
538            clippy::undocumented_unsafe_blocks,
539            reason = "Comments will be stripped, this will suppress some of the lints that are \
540            caused by it"
541        )] },
542    ]);
543
544    output_processed_enum_csr_impl(&enum_name, item_impl, out_dir, state)
545}

547pub(super) fn process_enum_execution_impl(
548    mut item_impl: ItemImpl,
549    out_dir: &Path,
550    state: &mut State,
551) -> anyhow::Result<()> {
552    let enum_name = enum_name_from_impl(&item_impl);
553
554    let Some(execute_fn) = extract_execute_fn_from_impl_mut(&mut item_impl.items) else {
555        return Err(anyhow::anyhow!(
556            "Unexpected `impl` for `{}`, `#[instruction_execution]` attribute must be added to a \
557            trait implementation, but no `execute` method was found",
558            item_impl.self_ty.to_token_stream()
559        ));
560    };
561    let execute_block = &mut execute_fn.block;
562
563    let Some(enum_definition) = state.get_known_enum_definition(&enum_name) else {
564        eprintln!("{enum_name} execution is waiting on its definition");
565        state.add_pending_enum_execution_impl(PendingEnumExecutionImpl { item_impl });
566        return Ok(());
567    };
568
569    let all_dependencies = match collect_all_dependencies(
570        state,
571        enum_definition.direct_dependencies.iter().cloned(),
572    ) {
573        Ok(all_dependencies) => all_dependencies,
574        Err(dependency_enum_name) => {
575            eprintln!("{enum_name} execution is waiting on {dependency_enum_name} definition");
576            state.add_pending_enum_execution_impl(PendingEnumExecutionImpl { item_impl });
577            return Ok(());
578        }
579    };
580
581    let mut all_execute_blocks = Vec::new();
582    let mut all_where_predicates = Vec::new();
583
584    for (dependency_enum_name, _dependency_enum_definition) in all_dependencies {
585        let Some(dependency_enum_execution_impl) =
586            state.get_known_enum_execution_impl(&dependency_enum_name)
587        else {
588            eprintln!(
589                "{enum_name} execution is waiting on {dependency_enum_name} execution \
590                implementation"
591            );
592            state.add_pending_enum_execution_impl(PendingEnumExecutionImpl { item_impl });
593            return Ok(());
594        };
595
596        all_execute_blocks.push(
597            &extract_execute_fn(&dependency_enum_execution_impl.item_impl)
598                .expect("Dependencies are all valid; qed")
599                .block,
600        );
601        if let Some(where_clause) = &dependency_enum_execution_impl
602            .item_impl
603            .generics
604            .where_clause
605        {
606            all_where_predicates.extend(where_clause.predicates.iter().cloned());
607        }
608    }
609
610    if let Some(where_clause) = &mut item_impl.generics.where_clause {
611        let mut already_inserted = where_clause
612            .predicates
613            .iter()
614            .cloned()
615            .collect::<HashSet<_>>();
616        for predicate in all_where_predicates {
617            if already_inserted.contains(&predicate) {
618                continue;
619            }
620            already_inserted.insert(predicate.clone());
621            where_clause.predicates.push(predicate);
622        }
623    } else {
624        return Err(anyhow::anyhow!(
625            "Missing where clause on `#[instruction_execution] impl ExecutableInstruction for \
626            {enum_name}`"
627        ));
628    }
629
630    let all_execute_blocks = all_execute_blocks
631        .into_iter()
632        .chain(iter::once(&*execute_block));
633
634    let mut all_instructions = HashMap::new();
635    for block in all_execute_blocks {
636        for maybe_instruction in extract_variant_arms(block)? {
637            let (variant_name, instruction_arm) = maybe_instruction?;
638            all_instructions.insert(variant_name, instruction_arm);
639        }
640    }
641
642    let all_instructions = enum_definition
643        .instructions
644        .iter()
645        .map(|variant| {
646            let ident = &variant.ident;
647            all_instructions.remove(ident).with_context(|| {
648                format!(
649                    "Instruction `{ident}` not found in all_instructions for enum `{enum_name}`"
650                )
651            })
652        })
653        .collect::<Result<Vec<_>, _>>()?;
654
655    *execute_block = parse_quote! {{
656        match self {
657            #( #all_instructions )*
658        }
659    }};
660
661    // Comments will be stripped, this will suppress some of the lints that are caused by it
662    item_impl.attrs.extend([
663        parse_quote! { #[expect(clippy::allow_attributes, reason = "Attribute below")] },
664        parse_quote! { #[allow(
665            clippy::undocumented_unsafe_blocks,
666            reason = "Comments will be stripped, this will suppress some of the lints that are \
667            caused by it"
668        )] },
669    ]);
670
671    output_processed_enum_execution_impl(&enum_name, item_impl, out_dir, state)
672}

674/// Process remaining enums that were waiting for dependencies
675pub(super) fn process_pending_enum_execution_impls(
676    out_dir: &Path,
677    state: &mut State,
678) -> anyhow::Result<()> {
679    {
680        let mut last_pending_enums_count = 0;
681        loop {
682            let pending_enums = state.take_pending_enum_operands_impls();
683
684            if pending_enums.is_empty() {
685                break;
686            }
687
688            if pending_enums.len() == last_pending_enums_count {
689                return Err(anyhow::anyhow!(
690                    "Failed to process instruction `#[instruction_execution]` macro on \
691                    `ExecutableInstructionOperands`, circular dependency detected, pending_enums: \
692                    {:?}",
693                    pending_enums
694                        .iter()
695                        .map(|pending_enum| enum_name_from_impl(&pending_enum.item_impl))
696                        .collect::<Vec<_>>()
697                ));
698            }
699            last_pending_enums_count = pending_enums.len();
700
701            for PendingEnumOperandsImpl { item_impl } in pending_enums {
702                process_enum_operands_impl(item_impl, out_dir, state)?;
703            }
704        }
705    }
706    {
707        let mut last_pending_enums_count = 0;
708        loop {
709            let pending_enums = state.take_pending_enum_csr_impls();
710
711            if pending_enums.is_empty() {
712                break;
713            }
714
715            if pending_enums.len() == last_pending_enums_count {
716                return Err(anyhow::anyhow!(
717                    "Failed to process instruction `#[instruction_execution]` macro on \
718                    `ExecutableInstructionCsr`, circular dependency detected, pending_enums: {:?}",
719                    pending_enums
720                        .iter()
721                        .map(|pending_enum| enum_name_from_impl(&pending_enum.item_impl))
722                        .collect::<Vec<_>>()
723                ));
724            }
725            last_pending_enums_count = pending_enums.len();
726
727            for PendingEnumCsrImpl { item_impl } in pending_enums {
728                process_enum_csr_impl(item_impl, out_dir, state)?;
729            }
730        }
731    }
732    {
733        let mut last_pending_enums_count = 0;
734        loop {
735            let pending_enums = state.take_pending_enum_execution_impls();
736
737            if pending_enums.is_empty() {
738                break;
739            }
740
741            if pending_enums.len() == last_pending_enums_count {
742                return Err(anyhow::anyhow!(
743                    "Failed to process instruction `#[instruction_execution]` macro on \
744                    `ExecutableInstruction`, circular dependency detected, pending_enums: {:?}",
745                    pending_enums
746                        .iter()
747                        .map(|pending_enum| enum_name_from_impl(&pending_enum.item_impl))
748                        .collect::<Vec<_>>()
749                ));
750            }
751            last_pending_enums_count = pending_enums.len();
752
753            for PendingEnumExecutionImpl { item_impl } in pending_enums {
754                process_enum_execution_impl(item_impl, out_dir, state)?;
755            }
756        }
757    }
758
759    Ok(())
760}
