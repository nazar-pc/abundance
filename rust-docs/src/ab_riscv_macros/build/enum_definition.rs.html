18/// Attribute for `#[instruction]` macro used on enum definition
19#[derive(Debug, Default)]
20pub(super) struct InstructionDefinition {
21    pub(super) items: Punctuated<InstructionDefinitionItem, Token![,]>,
22}
23
24impl Parse for InstructionDefinition {
25    fn parse(input: ParseStream<'_>) -> Result<Self, Error> {
26        Ok(Self {
27            items: input.parse_terminated(InstructionDefinitionItem::parse, Token![,])?,
28        })
29    }
30}

32/// Attribute item for `#[instruction]` macro used on enum definition
33#[derive(Debug)]
34pub(super) enum InstructionDefinitionItem {
35    /// Specifies the order of instruction variants
36    Reorder(Vec<Ident>),
37    /// Specifies ignored instruction variants or whole enums
38    Ignore(Vec<Ident>),
39    /// Specifies the inherited instruction variants
40    Inherit(Vec<Ident>),
41}
42
43impl InstructionDefinitionItem {
44    fn parse(input: ParseStream<'_>) -> Result<Self, Error> {
45        let key: Ident = input.parse()?;
46        input.parse::<Token![=]>()?;
47
48        let content;
49        bracketed!(content in input);
50
51        let values: Punctuated<Ident, Token![,]> =
52            content.parse_terminated(Ident::parse, Token![,])?;
53
54        match key.to_string().as_str() {
55            "reorder" => Ok(Self::Reorder(values.into_iter().collect())),
56            "ignore" => Ok(Self::Ignore(values.into_iter().collect())),
57            "inherit" => Ok(Self::Inherit(values.into_iter().collect())),
58            _ => Err(Error::new_spanned(key, "unknown instruction attribute key")),
59        }
60    }
61}

68pub(super) fn collect_enum_definitions_from_dependencies()
69-> impl Iterator<Item = anyhow::Result<(ItemEnum, Vec<Ident>, Rc<Path>)>> {
70    // Collect exported instruction enums from dependencies
71    env::vars().filter_map(|(key, source)| {
72        if !key.ends_with(ENUM_DEFINITION_ENV_VAR_SUFFIX) {
73            return None;
74        }
75
76        let result = try {
77            let dependencies_key = format!(
78                "{}{ENUM_DEPENDENCIES_ENV_VAR_SUFFIX}",
79                &key[..key.len() - ENUM_DEFINITION_ENV_VAR_SUFFIX.len()]
80            );
81            let dependencies_string = env::var(&dependencies_key).with_context(|| {
82                format!(
83                    "Failed to read environment variable `{}` that is expected to contain \
84                    instruction enum dependencies",
85                    dependencies_key
86                )
87            })?;
88            let dependencies = dependencies_string
89                .split('=')
90                .nth(1)
91                .with_context(|| {
92                    format!(
93                        "Dependency must follow the pattern \
94                        `Instruction=InstructionA,InstructionB`: {dependencies_string}"
95                    )
96                })?
97                .split(',')
98                .filter(|dependency| !dependency.is_empty())
99                .map(|dependency| format_ident!("{dependency}"))
100                .collect::<Vec<_>>();
101
102            let source = Rc::from(Path::new(&source));
103
104            let mut item_enum_contents = fs::read_to_string(&source).with_context(|| {
105                format!(
106                    "Failed to read Rust file `{}` that is expected to contain instruction \
107                    enum definition",
108                    source.display()
109                )
110            })?;
111            pre_process_rust_code(&mut item_enum_contents);
112            let item_enum = parse_str::<ItemEnum>(&item_enum_contents).with_context(|| {
113                format!(
114                    "Failed to parse Rust file `{}` that is expected to contain instruction \
115                    enum definition",
116                    source.display()
117                )
118            })?;
119
120            // Re-export metadata so it is available for downstream crates even if they don't depend
121            // directly on a crate where upstream instruction is defined
122            println!(
123                "cargo::metadata={}{ENUM_DEFINITION_ENV_VAR_SUFFIX}={}",
124                item_enum.ident,
125                source.display()
126            );
127            println!(
128                "cargo::metadata={}{ENUM_DEPENDENCIES_ENV_VAR_SUFFIX}={dependencies_string}",
129                item_enum.ident
130            );
131
132            (item_enum, dependencies, source)
133        };
134
135        Some(result)
136    })
137}

139fn output_processed_enum_definition(
140    item_enum: ItemEnum,
141    dependencies: Vec<Ident>,
142    out_dir: &Path,
143    state: &mut State,
144) -> anyhow::Result<()> {
145    let enum_name = item_enum.ident.clone();
146    let enum_file_path = out_dir.join(format!("{}_definition.rs", enum_name));
147    let code = item_enum.to_token_stream().to_string();
148    // Format
149    let code = unparse(&parse_file(&code).expect("Generated code is valid; qed"));
150    // Normalize source
151    let item_enum = parse_str(&code).expect("Generated code is valid; qed");
152
153    // Avoid extra file truncation/override if it didn't change
154    if fs::read_to_string(&enum_file_path).ok().as_ref() != Some(&code) {
155        fs::write(&enum_file_path, code).with_context(|| {
156            format!(
157                "Failed to write generated Rust file with instruction enum `{}`",
158                enum_name,
159            )
160        })?;
161    }
162    println!(
163        "cargo::metadata={enum_name}{ENUM_DEFINITION_ENV_VAR_SUFFIX}={}",
164        enum_file_path.display()
165    );
166    println!(
167        "cargo::metadata={enum_name}{ENUM_DEPENDENCIES_ENV_VAR_SUFFIX}={enum_name}={}",
168        dependencies
169            .iter()
170            .map(ToString::to_string)
171            .collect::<Vec<_>>()
172            .join(","),
173    );
174
175    state.insert_known_enum_definition(item_enum, dependencies, Rc::from(enum_file_path))
176}

178/// Identify enums with an `#[instruction]` attribute and generate Rust files with the enum
179/// definition after processing attribute contents and propagate the environment variable with the
180/// path do dependent crates
181pub(super) fn process_enum_definition(
182    mut item_enum: ItemEnum,
183    out_dir: &Path,
184    state: &mut State,
185) -> anyhow::Result<()> {
186    let Some(attribute_index) = item_enum
187        .attrs
188        .iter()
189        .enumerate()
190        .find_map(|(index, attr)| attr.meta.path().is_ident("instruction").then_some(index))
191    else {
192        // Enum without `#[instruction]` attribute, skip it
193        return Ok(());
194    };
195
196    // We'll store the whole enum in the Rust file, so remove any attributes before `#[instruction]`
197    // attribute to avoid conflicts
198    let mut removed_attributes = item_enum.attrs.drain(..=attribute_index);
199    let attribute = removed_attributes
200        .next_back()
201        .expect("`#[instruction]` attribute is found; qed");
202
203    for removed_attribute in removed_attributes {
204        if removed_attribute.path().is_ident("derive") {
205            return Err(anyhow::anyhow!(
206                "All `#[derive(...)]` attributes must be after `#[instruction(...)]` attribute, found: {}",
207                removed_attribute.to_token_stream()
208            ));
209        }
210    }
211
212    let dependencies = match attribute.meta {
213        Meta::Path(_) => {
214            process_enum_definition_with_variants(&mut item_enum, state)?;
215            Vec::new()
216        }
217        Meta::List(meta_list) => {
218            let instruction_definition = parse2::<InstructionDefinition>(meta_list.tokens)
219                .context("Failed to parse `#[instruction(...)]` attribute")?;
220
221            if instruction_definition.items.iter().any(|item| match item {
222                InstructionDefinitionItem::Reorder(_) | InstructionDefinitionItem::Ignore(_) => {
223                    false
224                }
225                InstructionDefinitionItem::Inherit(enums) => enums
226                    .iter()
227                    .any(|enum_name| state.get_known_enum_definition(enum_name).is_none()),
228            }) {
229                state.add_pending_enum_definition(PendingEnumDefinition {
230                    instruction_definition,
231                    item_enum,
232                });
233                return Ok(());
234            }
235
236            let Some(result) =
237                process_enum_definition_inherited(instruction_definition, item_enum, state)?
238            else {
239                return Ok(());
240            };
241            let dependencies;
242            (item_enum, dependencies) = result;
243            dependencies
244        }
245        Meta::NameValue(meta_name_value) => {
246            return Err(anyhow::anyhow!(
247                "Unexpected `#[instruction = {}]` attribute",
248                Meta::NameValue(meta_name_value).to_token_stream()
249            ));
250        }
251    };
252
253    output_processed_enum_definition(item_enum, dependencies, out_dir, state)
254}

264fn process_enum_definition_inherited(
265    instruction_definition: InstructionDefinition,
266    mut item_enum: ItemEnum,
267    state: &mut State,
268) -> anyhow::Result<Option<(ItemEnum, Vec<Ident>)>> {
269    let mut all_known_instructions = HashMap::<Ident, KnownInstruction>::new();
270    let mut dependencies = Vec::new();
271
272    for item in &instruction_definition.items {
273        match item {
274            InstructionDefinitionItem::Reorder(_) | InstructionDefinitionItem::Ignore(_) => {
275                // Ignore for now
276            }
277            InstructionDefinitionItem::Inherit(inherit_enums) => {
278                for inherit_enum in inherit_enums {
279                    dependencies.push(inherit_enum.clone());
280                    let Some(known_enum) = state.get_known_enum_definition(inherit_enum) else {
281                        return Err(anyhow::anyhow!(
282                            "Unknown inherit enum `{}` in `#[instruction(...)]` attribute",
283                            inherit_enum
284                        ));
285                    };
286
287                    for known_instruction in &known_enum.instructions {
288                        match all_known_instructions.entry(known_instruction.ident.clone()) {
289                            Entry::Occupied(entry) => {
290                                let existing_known_instruction = entry.get();
291                                if &existing_known_instruction.instruction != known_instruction {
292                                    return Err(anyhow::anyhow!(
293                                        "Duplicate instruction definition that is not identical to \
294                                        an existing one: `{}` ({}) != `{}` ({})",
295                                        existing_known_instruction.instruction.to_token_stream(),
296                                        existing_known_instruction.source.display(),
297                                        known_instruction.to_token_stream(),
298                                        known_enum.source.display(),
299                                    ));
300                                }
301                            }
302                            Entry::Vacant(entry) => {
303                                let instruction = Rc::new(known_instruction);
304
305                                entry.insert(KnownInstruction {
306                                    instruction: Rc::clone(&instruction),
307                                    source: Rc::clone(&known_enum.source),
308                                });
309                            }
310                        }
311                    }
312                }
313            }
314        }
315    }
316
317    let mut included_instructions = HashSet::new();
318    let mut instructions = Vec::new();
319
320    let mut own_instructions = item_enum
321        .variants
322        .iter()
323        .map(|variant| (variant.ident.clone(), Rc::new(variant.clone())))
324        .collect::<HashMap<_, _>>();
325
326    for item in &instruction_definition.items {
327        match item {
328            InstructionDefinitionItem::Reorder(reorder_variants) => {
329                for reorder_variant in reorder_variants {
330                    if included_instructions.contains(reorder_variant) {
331                        return Err(anyhow::anyhow!(
332                            "Instruction `{}` in `#[instruction(...)]`'s `reorder` attribute is \
333                            already included earlier",
334                            reorder_variant
335                        ));
336                    }
337                    included_instructions.insert(reorder_variant.clone());
338
339                    let instruction = if let Some(instruction) =
340                        own_instructions.remove(reorder_variant)
341                    {
342                        instruction
343                    } else if let Some(instruction) = all_known_instructions.get(reorder_variant) {
344                        Rc::clone(&instruction.instruction)
345                    } else {
346                        return Err(anyhow::anyhow!(
347                            "Unknown reorder instruction `{}` in `#[instruction(...)]` attribute",
348                            reorder_variant
349                        ));
350                    };
351
352                    instructions.push(instruction);
353                }
354            }
355            InstructionDefinitionItem::Ignore(ignore_items) => {
356                for ignore_item in ignore_items {
357                    if own_instructions.contains_key(ignore_item)
358                        || all_known_instructions.contains_key(ignore_item)
359                    {
360                        included_instructions.insert(ignore_item.clone());
361                    } else if let Some(known_enum) = state.get_known_enum_definition(ignore_item) {
362                        for known_instruction in &known_enum.instructions {
363                            included_instructions.insert(known_instruction.ident.clone());
364                        }
365                    } else {
366                        state.add_pending_enum_definition(PendingEnumDefinition {
367                            instruction_definition,
368                            item_enum,
369                        });
370                        return Ok(None);
371                    }
372                }
373            }
374            InstructionDefinitionItem::Inherit(inherit_enums) => {
375                for inherit_enum in inherit_enums {
376                    let Some(known_enum) = state.get_known_enum_definition(inherit_enum) else {
377                        return Err(anyhow::anyhow!(
378                            "Unknown inherit enum `{}` in `#[instruction(...)]` attribute",
379                            inherit_enum
380                        ));
381                    };
382
383                    for known_instruction in &known_enum.instructions {
384                        if !included_instructions.contains(&known_instruction.ident) {
385                            included_instructions.insert(known_instruction.ident.clone());
386                            instructions.push(Rc::clone(known_instruction));
387                        }
388                    }
389                }
390            }
391        }
392    }
393
394    for own_instruction in &item_enum.variants {
395        if !included_instructions.contains(&own_instruction.ident) {
396            instructions.push(Rc::new(own_instruction.clone()));
397        }
398    }
399
400    item_enum.variants = Punctuated::from_iter(
401        instructions
402            .into_iter()
403            .map(|instruction| instruction.as_ref().clone()),
404    );
405
406    Ok(Some((item_enum, dependencies)))
407}

409/// Process remaining enums that were waiting for dependencies
410pub(super) fn process_pending_enum_definitions(
411    out_dir: &Path,
412    state: &mut State,
413) -> anyhow::Result<()> {
414    let mut last_pending_enums_count = 0;
415    loop {
416        let pending_enums = state.take_pending_enum_definitions();
417
418        if pending_enums.is_empty() {
419            break;
420        }
421
422        if pending_enums.len() == last_pending_enums_count {
423            return Err(anyhow::anyhow!(
424                "Failed to process instruction macros, circular dependency detected, \
425                pending_enums: {:?}",
426                pending_enums
427                    .iter()
428                    .map(|pending_enum| &pending_enum.item_enum.ident)
429                    .collect::<Vec<_>>()
430            ));
431        }
432        last_pending_enums_count = pending_enums.len();
433
434        for PendingEnumDefinition {
435            instruction_definition,
436            item_enum,
437        } in pending_enums
438        {
439            if let Some((item_enum, dependencies)) =
440                process_enum_definition_inherited(instruction_definition, item_enum, state)?
441            {
442                output_processed_enum_definition(item_enum, dependencies, out_dir, state)?;
443            }
444        }
445    }
446
447    Ok(())
