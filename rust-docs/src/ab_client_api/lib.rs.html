69/// Error for [`ChainInfoWrite::persist_block()`]
70#[derive(Debug, thiserror::Error)]
71pub enum PersistBlockError {
72    /// Missing parent
73    #[error("Missing parent")]
74    MissingParent,
75    /// Block is outside the acceptable range
76    #[error("Block is outside the acceptable range")]
77    OutsideAcceptableRange,
78    /// Storage item write error
79    #[error("Storage item write error")]
80    StorageItemWriteError {
81        /// Low-level error
82        #[from]

84    },

87// TODO: Split this into different more narrow traits
88/// Chain info.
89///
90/// NOTE:
91/// <div class="warning">
92/// Blocks or their parts returned from these APIs are reference-counted and cheap to clone.
93/// However, it is not expected that they will be retained in memory for a long time. Blocks and
94/// headers will not be pruned until their reference count goes down to one. This is imported when
95/// there is an ongoing block import happening and its parent must exist until the import
96/// finishes.
97/// </div>
98pub trait ChainInfo<Block>: Clone + Send + Sync
99where
100    Block: GenericOwnedBlock,
101{
102    /// Best block root
103    fn best_root(&self) -> BlockRoot;
104
105    // TODO: Uncomment if/when necessary
106    // /// Find root of ancestor block number for descendant block root
107    // fn ancestor_root(
108    //     &self,
109    //     ancestor_block_number: BlockNumber,
110    //     descendant_block_root: &BlockRoot,
111    // ) -> Option<BlockRoot>;
112
113    /// Best block header
114    fn best_header(&self) -> Block::Header;
115
116    /// Returns the best block header like [`Self::best_header()`] with additional block details
117    fn best_header_with_details(&self) -> (Block::Header, BlockDetails);
118
119    /// Get header of ancestor block number for descendant block root
120    fn ancestor_header(
121        &self,
122        ancestor_block_number: BlockNumber,
123        descendant_block_root: &BlockRoot,
124    ) -> Option<Block::Header>;
125
126    /// Block header
127    fn header(&self, block_root: &BlockRoot) -> Option<Block::Header>;
128
129    /// Returns a block header like [`Self::header()`] with additional block details
130    fn header_with_details(&self, block_root: &BlockRoot) -> Option<(Block::Header, BlockDetails)>;
131}

133/// [`ChainInfo`] extension for writing information
134pub trait ChainInfoWrite<Block>: ChainInfo<Block>
135where
136    Block: GenericOwnedBlock,
137{
138    /// Persist newly imported block
139    fn persist_block(
140        &self,
141        block: Block,
142        block_details: BlockDetails,
143    ) -> impl Future<Output = Result<(), PersistBlockError>> + Send;
144}

146/// Chain sync status
147pub trait ChainSyncStatus: Clone + Send + Sync + 'static {
148    /// The block number that the sync process is targeting right now.
149    ///
150    /// Can be zero if not syncing actively.
151    fn target_block_number(&self) -> BlockNumber;
152
153    /// Returns `true` if the chain is currently syncing
154    fn is_syncing(&self) -> bool;
155
156    /// Returns `true` if the node is currently offline
157    fn is_offline(&self) -> bool;
158}
