69/// Error for [`ChainInfo::block()`]
70#[derive(Debug, thiserror::Error)]
71pub enum ReadBlockError {
72    /// Unknown block root
73    #[error("Unknown block root")]
74    UnknownBlockRoot,
75    /// Failed to decode the block
76    #[error("Failed to decode the block")]
77    FailedToDecode,
78    /// Storage item read error
79    #[error("Storage item read error")]
80    StorageItemReadError {
81        /// Low-level error
82        #[from]
83        error: io::Error,
84    },
85}

105// TODO: Split this into different more narrow traits
106/// Chain info.
107///
108/// NOTE:
109/// <div class="warning">
110/// Blocks or their parts returned from these APIs are reference-counted and cheap to clone.
111/// However, it is not expected that they will be retained in memory for a long time. Blocks and
112/// headers will not be pruned until their reference count goes down to one. This is imported when
113/// there is an ongoing block import happening and its parent must exist until the import
114/// finishes.
115/// </div>
116pub trait ChainInfo<Block>: Clone + Send + Sync
117where
118    Block: GenericOwnedBlock,
119{
120    /// Best block root
121    fn best_root(&self) -> BlockRoot;
122
123    // TODO: Uncomment if/when necessary
124    // /// Find root of ancestor block number for descendant block root
125    // fn ancestor_root(
126    //     &self,
127    //     ancestor_block_number: BlockNumber,
128    //     descendant_block_root: &BlockRoot,
129    // ) -> Option<BlockRoot>;
130
131    /// Best block header
132    fn best_header(&self) -> Block::Header;
133
134    /// Returns the best block header like [`Self::best_header()`] with additional block details
135    fn best_header_with_details(&self) -> (Block::Header, BlockDetails);
136
137    /// Get header of ancestor block number for descendant block root
138    fn ancestor_header(
139        &self,
140        ancestor_block_number: BlockNumber,
141        descendant_block_root: &BlockRoot,
142    ) -> Option<Block::Header>;
143
144    /// Block header
145    fn header(&self, block_root: &BlockRoot) -> Option<Block::Header>;
146
147    /// Returns a block header like [`Self::header()`] with additional block details
148    fn header_with_details(&self, block_root: &BlockRoot) -> Option<(Block::Header, BlockDetails)>;
149
150    fn block(
151        &self,
152        block_root: &BlockRoot,
153    ) -> impl Future<Output = Result<Block, ReadBlockError>> + Send;
154}

156/// [`ChainInfo`] extension for writing information
157pub trait ChainInfoWrite<Block>: ChainInfo<Block>
158where
159    Block: GenericOwnedBlock,
160{
161    /// Persist newly imported block
162    fn persist_block(
163        &self,
164        block: Block,
165        block_details: BlockDetails,
166    ) -> impl Future<Output = Result<(), PersistBlockError>> + Send;
167}

169/// Chain sync status
170pub trait ChainSyncStatus: Clone + Send + Sync + 'static {
171    /// The block number that the sync process is targeting right now.
172    ///
173    /// Can be zero if not syncing actively.
174    fn target_block_number(&self) -> BlockNumber;
175
176    /// Returns `true` if the chain is currently syncing
177    fn is_syncing(&self) -> bool;
178
179    /// Returns `true` if the node is currently offline
180    fn is_offline(&self) -> bool;
181}
