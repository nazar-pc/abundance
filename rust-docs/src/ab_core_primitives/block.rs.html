16/// Block number
17#[derive(
18    Debug,
19    Display,
20    Default,
21    Copy,
22    Clone,
23    Ord,
24    PartialOrd,
25    Eq,
26    PartialEq,
27    Hash,
28    From,
29    Into,
30    Add,
31    AddAssign,
32    Sub,
33    SubAssign,
34    TrivialType,
35)]
36#[cfg_attr(
37    feature = "scale-codec",
38    derive(Encode, Decode, TypeInfo, MaxEncodedLen)
39)]
40#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
41#[cfg_attr(feature = "serde", serde(transparent))]
42#[repr(C)]
43pub struct BlockNumber(u64);
44
45impl Step for BlockNumber {
46    #[inline]
47    fn steps_between(start: &Self, end: &Self) -> (usize, Option<usize>) {
48        u64::steps_between(&start.0, &end.0)
49    }
50
51    #[inline]
52    fn forward_checked(start: Self, count: usize) -> Option<Self> {
53        u64::forward_checked(start.0, count).map(Self)
54    }
55
56    #[inline]
57    fn backward_checked(start: Self, count: usize) -> Option<Self> {
58        u64::backward_checked(start.0, count).map(Self)
59    }
60}
61
62impl BlockNumber {
63    /// Size in bytes
64    pub const SIZE: usize = size_of::<u64>();
65    /// Genesis block number
66    pub const ZERO: BlockNumber = BlockNumber(0);
67    /// First block number
68    pub const ONE: BlockNumber = BlockNumber(1);
69    /// Max block number
70    pub const MAX: BlockNumber = BlockNumber(u64::MAX);
71
72    /// Create new instance
73    #[inline]
74    pub const fn new(n: u64) -> Self {
75        Self(n)
76    }
77
78    /// Get internal representation
79    #[inline(always)]
80    pub const fn as_u64(self) -> u64 {
81        self.0
82    }
83
84    /// Create block number from bytes
85    #[inline]
86    pub const fn from_bytes(bytes: [u8; Self::SIZE]) -> Self {
87        Self(u64::from_le_bytes(bytes))
88    }
89
90    /// Convert block number to bytes
91    #[inline]
92    pub const fn to_bytes(self) -> [u8; Self::SIZE] {
93        self.0.to_le_bytes()
94    }
95
96    /// Checked subtraction, returns `None` on underflow
97    pub const fn checked_sub(self, rhs: Self) -> Option<Self> {
98        if let Some(n) = self.0.checked_sub(rhs.0) {
99            Some(Self(n))
100        } else {
101            None
102        }
103    }
104
105    /// Saturating subtraction
106    pub const fn saturating_sub(self, rhs: Self) -> Self {
107        Self(self.0.saturating_sub(rhs.0))
108    }
109}
