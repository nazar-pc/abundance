3// Minimal `u256` implementation that is needed for sectors and optimized for producing the
4// remainder of division by `u64`
5#[derive(Clone, Copy, PartialEq, Eq)]
6pub(crate) struct NanoU256 {
7    lo: u128,
8    hi: u128,
9}
10
11impl NanoU256 {
12    #[inline(always)]
13    pub(crate) const fn from_le_bytes(bytes: [u8; 32]) -> Self {
14        let lo = u128::from_le_bytes([
15            bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
16            bytes[8], bytes[9], bytes[10], bytes[11], bytes[12], bytes[13], bytes[14], bytes[15],
17        ]);
18        let hi = u128::from_le_bytes([
19            bytes[16], bytes[17], bytes[18], bytes[19], bytes[20], bytes[21], bytes[22], bytes[23],
20            bytes[24], bytes[25], bytes[26], bytes[27], bytes[28], bytes[29], bytes[30], bytes[31],
21        ]);
22
23        Self { lo, hi }
24    }
25}
26
27impl Rem<u64> for NanoU256 {
28    type Output = u64;
29
30    #[inline(always)]
31    fn rem(self, rhs: u64) -> u64 {
32        assert_ne!(rhs, 0, "division by zero");
33
34        let rhs = u128::from(rhs);
35
36        // If `hi` is 0, we can directly compute the remainder from `lo`
37        if self.hi == 0 {
38            return (self.lo % rhs) as u64;
39        }
40
41        // Process the high 128 bits first
42        let hi_rem = self.hi % rhs;
43
44        // Combine a high remainder with low 128 bits
45        // hi_rem * 2^128 + lo
46        let combined = (hi_rem << 64u8) | (self.lo >> 64u8);
47        let combined_rem = combined % rhs;
48
49        // Process the remaining low 64 bits
50        let low = self.lo & 0xffff_ffff_ffff_ffff;
51        let final_rem = ((combined_rem << 64u8) | low) % rhs;
52
53        final_rem as u64
54    }
55}
