72        u64::forward_checked(start.0, count).map(Self)
73    }

77        u64::backward_checked(start.0, count).map(Self)
78    }

81impl SegmentIndex {
82    /// Segment index 0.
83    pub const ZERO: SegmentIndex = SegmentIndex(0);
84    /// Segment index 1.
85    pub const ONE: SegmentIndex = SegmentIndex(1);
86
87    /// Create new instance
88    #[inline]
89    pub const fn new(n: u64) -> Self {
90        Self(n)
91    }
92
93    /// Get internal representation
94    #[inline(always)]
95    pub const fn as_u64(self) -> u64 {
96        self.0
97    }
98
99    /// Get the first piece index in this segment.
100    #[inline]
101    pub const fn first_piece_index(&self) -> PieceIndex {
102        PieceIndex::new(self.0 * RecordedHistorySegment::NUM_PIECES as u64)
103    }
104
105    /// Get the last piece index in this segment.
106    #[inline]
107    pub const fn last_piece_index(&self) -> PieceIndex {
108        PieceIndex::new((self.0 + 1) * RecordedHistorySegment::NUM_PIECES as u64 - 1)
109    }
110
111    /// List of piece indexes that belong to this segment.
112    #[inline]
113    pub fn segment_piece_indexes(&self) -> [PieceIndex; RecordedHistorySegment::NUM_PIECES] {
114        let mut piece_indices = [PieceIndex::ZERO; RecordedHistorySegment::NUM_PIECES];
115        (self.first_piece_index()..=self.last_piece_index())
116            .zip(&mut piece_indices)
117            .for_each(|(input, output)| {
118                *output = input;
119            });
120
121        piece_indices
122    }
123
124    /// Checked integer subtraction. Computes `self - rhs`, returning `None` if overflow occurred.
125    #[inline]
126    pub fn checked_sub(self, rhs: Self) -> Option<Self> {
127        self.0.checked_sub(rhs.0).map(Self)
128    }
129}

250impl HistorySize {
251    /// History size of one
252    pub const ONE: Self = Self(SegmentIndex::ZERO);
253
254    /// Create new instance
255    #[inline(always)]
256    pub const fn new(value: NonZeroU64) -> Self {
257        Self(SegmentIndex::new(value.get() - 1))
258    }
259
260    /// Get internal representation
261    pub const fn as_segment_index(&self) -> SegmentIndex {
262        self.0
263    }
264
265    /// Get internal representation
266    pub const fn as_non_zero_u64(&self) -> NonZeroU64 {
267        NonZeroU64::new(self.0.as_u64().saturating_add(1)).expect("Not zero; qed")
268    }
269
270    /// Size of blockchain history in pieces.
271    #[inline(always)]
272    pub const fn in_pieces(&self) -> NonZeroU64 {
273        NonZeroU64::new(
274            self.0
275                .as_u64()
276                .saturating_add(1)
277                .saturating_mul(RecordedHistorySegment::NUM_PIECES as u64),
278        )
279        .expect("Not zero; qed")
280    }
281
282    /// Segment index that corresponds to this history size.
283    #[inline(always)]
284    pub fn segment_index(&self) -> SegmentIndex {
285        self.0
286    }
287
288    /// History size at which expiration check for sector happens.
289    ///
290    /// Returns `None` on overflow.
291    #[inline(always)]
292    pub fn sector_expiration_check(&self, min_sector_lifetime: Self) -> Option<Self> {
293        self.as_non_zero_u64()
294            .checked_add(min_sector_lifetime.as_non_zero_u64().get())
295            .map(Self::new)
296    }

299/// Progress of an archived block.
300#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd, Hash, TrivialType)]
301#[cfg_attr(feature = "scale-codec", derive(Encode, Decode, TypeInfo))]
302#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
303#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
304#[repr(C)]
305pub struct ArchivedBlockProgress {
306    /// Number of partially archived bytes of a block, `0` for full block
307    bytes: u32,
308}
309
310impl Default for ArchivedBlockProgress {
311    /// We assume a block can always fit into the segment initially, but it is definitely possible
312    /// to be transitioned into the partial state after some overflow checking.
313    #[inline(always)]
314    fn default() -> Self {
315        Self::new_complete()
316    }
317}
318
319impl ArchivedBlockProgress {
320    /// Block is archived fully
321    #[inline(always)]
322    pub const fn new_complete() -> Self {
323        Self { bytes: 0 }
324    }
325
326    /// Block is partially archived with provided number of bytes
327    #[inline(always)]
328    pub const fn new_partial(new_partial: NonZeroU32) -> Self {
329        Self {
330            bytes: new_partial.get(),
331        }
332    }
333
334    /// Return the number of partially archived bytes if the progress is not complete
335    #[inline(always)]
336    pub const fn partial(&self) -> Option<NonZeroU32> {
337        NonZeroU32::new(self.bytes)
338    }
339}

341/// Last archived block
342#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd, Hash, TrivialType)]
343#[cfg_attr(feature = "scale-codec", derive(Encode, Decode, TypeInfo))]
344#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
345#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
346#[repr(C)]
347pub struct LastArchivedBlock {
348    /// Block number
349    pub number: Unaligned<BlockNumber>,
350    /// Progress of an archived block.
351    pub archived_progress: ArchivedBlockProgress,
352}
353
354impl LastArchivedBlock {
355    /// Returns the number of partially archived bytes for a block.
356    #[inline(always)]
357    pub fn partial_archived(&self) -> Option<NonZeroU32> {
358        self.archived_progress.partial()
359    }
360
361    /// Sets the number of partially archived bytes if block progress was archived partially
362    #[inline(always)]
363    pub fn set_partial_archived(&mut self, new_partial: NonZeroU32) {
364        self.archived_progress = ArchivedBlockProgress::new_partial(new_partial);
365    }
366
367    /// Indicate last archived block was archived fully
368    #[inline(always)]
369    pub fn set_complete(&mut self) {
370        self.archived_progress = ArchivedBlockProgress::new_complete();
371    }
372
373    /// Get block number (unwrap `Unaligned`)
374    pub const fn number(&self) -> BlockNumber {
375        self.number.as_inner()
376    }

379/// Segment header for a specific segment.
380///
381/// Each segment will have corresponding [`SegmentHeader`] included as the first item in the next
382/// segment. Each `SegmentHeader` includes hash of the previous one and all together form a chain of
383/// segment headers that is used for quick and efficient verification that some `Piece`
384/// corresponds to the actual archival history of the blockchain.
385#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, TrivialType)]
386#[cfg_attr(feature = "scale-codec", derive(Encode, Decode, TypeInfo))]
387#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
388#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
389#[repr(C)]
390pub struct SegmentHeader {
391    /// Segment index
392    pub segment_index: Unaligned<SegmentIndex>,
393    /// Root of roots of all records in a segment.
394    pub segment_root: SegmentRoot,
395    /// Hash of the segment header of the previous segment
396    pub prev_segment_header_hash: Blake3Hash,
397    /// Last archived block
398    pub last_archived_block: LastArchivedBlock,
399}
400
401impl SegmentHeader {
402    /// Hash of the whole segment header
403    #[inline(always)]
404    pub fn hash(&self) -> Blake3Hash {
405        blake3_hash(self.as_bytes())
406    }
407
408    /// Get segment index (unwrap `Unaligned`)
409    pub const fn segment_index(&self) -> SegmentIndex {
410        self.segment_index.as_inner()
411    }
412}

414/// Recorded history segment before archiving is applied.
415///
416/// NOTE: This is a stack-allocated data structure and can cause stack overflow!
417#[derive(Copy, Clone, Eq, PartialEq, Deref, DerefMut)]
418#[repr(C)]
419pub struct RecordedHistorySegment([Record; Self::NUM_RAW_RECORDS]);
420
421impl fmt::Debug for RecordedHistorySegment {
422    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
423        f.debug_struct("RecordedHistorySegment")
424            .finish_non_exhaustive()
425    }
426}

430    fn default() -> Self {

432    }
433}

451impl RecordedHistorySegment {
452    /// Number of raw records in one segment of recorded history.
453    pub const NUM_RAW_RECORDS: usize = 128;
454    /// Erasure coding rate for records during archiving process.
455    pub const ERASURE_CODING_RATE: (usize, usize) = (1, 2);
456    /// Number of pieces in one segment of archived history (taking erasure coding rate into
457    /// account)
458    pub const NUM_PIECES: usize =
459        Self::NUM_RAW_RECORDS * Self::ERASURE_CODING_RATE.1 / Self::ERASURE_CODING_RATE.0;
460    /// Size of recorded history segment in bytes.
461    ///
462    /// It includes half of the records (just source records) that will later be erasure coded and
463    /// together with corresponding roots and proofs will result in
464    /// [`Self::NUM_PIECES`] `Piece`s of archival history.
465    pub const SIZE: usize = Record::SIZE * Self::NUM_RAW_RECORDS;
466
467    /// Create boxed value without hitting stack overflow
468    #[inline]
469    #[cfg(feature = "alloc")]
470    pub fn new_boxed() -> Box<Self> {
471        // TODO: Should have been just `::new()`, but https://github.com/rust-lang/rust/issues/53827
472        // SAFETY: Data structure filled with zeroes is a valid invariant
473        unsafe { Box::<Self>::new_zeroed().assume_init() }
474    }
