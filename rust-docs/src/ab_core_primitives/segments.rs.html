264impl SegmentIndex {
265    /// Segment index 0
266    pub const ZERO: Self = Self(0);
267    /// Segment index 1
268    pub const ONE: Self = Self(1);
269
270    /// Get the first piece index in this segment
271    #[inline]
272    pub const fn first_piece_index(&self) -> PieceIndex {
273        PieceIndex::from(self.0 * RecordedHistorySegment::NUM_PIECES as u64)
274    }
275
276    /// Get the last piece index in this segment
277    #[inline]
278    pub const fn last_piece_index(&self) -> PieceIndex {
279        PieceIndex::from((self.0 + 1) * RecordedHistorySegment::NUM_PIECES as u64 - 1)
280    }
281
282    /// List of piece indexes that belong to this segment
283    #[inline]
284    pub fn segment_piece_indexes(&self) -> [PieceIndex; RecordedHistorySegment::NUM_PIECES] {
285        let mut piece_indices = [PieceIndex::ZERO; RecordedHistorySegment::NUM_PIECES];
286        (self.first_piece_index()..=self.last_piece_index())
287            .zip(&mut piece_indices)
288            .for_each(|(input, output)| {
289                *output = input;
290            });
291
292        piece_indices
293    }
294
295    /// Checked integer subtraction. Computes `self - rhs`, returning `None` if underflow occurred
296    #[inline]
297    pub fn checked_sub(self, rhs: Self) -> Option<Self> {
298        self.0.checked_sub(rhs.0).map(Self)
299    }
300
301    /// Saturating integer subtraction. Computes `self - rhs`, returning zero if underflow
302    /// occurred
303    #[inline]
304    pub const fn saturating_sub(self, rhs: Self) -> Self {
305        Self(self.0.saturating_sub(rhs.0))
306    }
307}

427impl HistorySize {
428    /// History size of one
429    pub const ONE: Self = Self(SegmentIndex::ZERO);
430
431    /// Create a new instance
432    #[inline(always)]
433    pub const fn new(value: NonZeroU64) -> Self {
434        Self(SegmentIndex::from(value.get() - 1))
435    }
436
437    /// Get internal representation
438    pub const fn as_segment_index(&self) -> SegmentIndex {
439        self.0
440    }
441
442    /// Get internal representation
443    pub const fn as_non_zero_u64(&self) -> NonZeroU64 {
444        NonZeroU64::new(u64::from(self.0).saturating_add(1)).expect("Not zero; qed")
445    }
446
447    /// Size of blockchain history in pieces
448    #[inline(always)]
449    pub const fn in_pieces(&self) -> NonZeroU64 {
450        NonZeroU64::new(
451            u64::from(self.0)
452                .saturating_add(1)
453                .saturating_mul(RecordedHistorySegment::NUM_PIECES as u64),
454        )
455        .expect("Not zero; qed")
456    }
457
458    /// Segment index that corresponds to this history size
459    #[inline(always)]
460    pub fn segment_index(&self) -> SegmentIndex {
461        self.0
462    }
463
464    /// History size at which expiration check for a sector happens.
465    ///
466    /// Returns `None` on overflow.
467    #[inline(always)]
468    pub fn sector_expiration_check(&self, min_sector_lifetime: Self) -> Option<Self> {
469        self.as_non_zero_u64()
470            .checked_add(min_sector_lifetime.as_non_zero_u64().get())
471            .map(Self::new)
472    }

475/// Progress of an archived block.
476#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd, Hash, TrivialType)]
477#[cfg_attr(feature = "scale-codec", derive(Encode, Decode))]
478#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
479#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
480#[repr(C)]
481pub struct ArchivedBlockProgress {
482    /// Number of partially archived bytes of a block, `0` for a full block
483    bytes: u32,
484}
485
486impl Default for ArchivedBlockProgress {
487    /// We assume a block can always fit into the segment initially, but it is definitely possible
488    /// to be transitioned into the partial state after some overflow checking.
489    #[inline(always)]
490    fn default() -> Self {
491        Self::new_complete()
492    }
493}
494
495impl ArchivedBlockProgress {
496    /// Block is archived fully
497    #[inline(always)]
498    pub const fn new_complete() -> Self {
499        Self { bytes: 0 }
500    }
501
502    /// Block is partially archived with the provided number of bytes
503    #[inline(always)]
504    pub const fn new_partial(new_partial: NonZeroU32) -> Self {
505        Self {
506            bytes: new_partial.get(),
507        }
508    }
509
510    /// Return the number of partially archived bytes if the progress is not complete
511    #[inline(always)]
512    pub const fn partial(&self) -> Option<NonZeroU32> {
513        NonZeroU32::new(self.bytes)
514    }
515}

556#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, TrivialType)]
557#[cfg_attr(feature = "scale-codec", derive(Encode, Decode))]
558#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
559#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
560#[repr(C)]
561pub struct SegmentHeader {
562    /// Local segment index
563    pub segment_index: Unaligned<LocalSegmentIndex>,
564    /// Root of roots of all records in a segment.
565    pub segment_root: SegmentRoot,
566    /// Hash of the segment header of the previous segment
567    pub prev_segment_header_hash: Blake3Hash,
568    /// Last archived block
569    pub last_archived_block: LastArchivedBlock,
570}

572impl SegmentHeader {
573    /// Hash of the whole segment header
574    #[inline(always)]
575    pub fn hash(&self) -> Blake3Hash {
576        const {
577            assert!(size_of::<Self>() <= CHUNK_LEN);
578        }
579        Blake3Hash::new(
580            single_chunk_hash(self.as_bytes())
581                .expect("Less than a single chunk worth of bytes; qed"),
582        )
583    }
584
585    /// Get local segment index (unwrap `Unaligned`)
586    #[inline(always)]
587    pub const fn local_segment_index(&self) -> LocalSegmentIndex {
588        self.segment_index.as_inner()
589    }

629impl RecordedHistorySegment {
630    /// Number of raw records in one segment of recorded history
631    pub const NUM_RAW_RECORDS: usize = 128;
632    /// Erasure coding rate for records during the archiving process
633    pub const ERASURE_CODING_RATE: (usize, usize) = (1, 2);
634    /// Number of pieces in one segment of archived history (taking erasure coding rate into
635    /// account)
636    pub const NUM_PIECES: usize =
637        Self::NUM_RAW_RECORDS * Self::ERASURE_CODING_RATE.1 / Self::ERASURE_CODING_RATE.0;
638    /// Size of recorded history segment in bytes.
639    ///
640    /// It includes half of the records (just source records) that will later be erasure coded and
641    /// together with corresponding roots and proofs will result in
642    /// [`Self::NUM_PIECES`] `Piece`s of archival history.
643    pub const SIZE: usize = Record::SIZE * Self::NUM_RAW_RECORDS;
644
645    /// Create boxed value without hitting stack overflow
646    #[inline]
647    #[cfg(feature = "alloc")]
648    pub fn new_boxed() -> Box<Self> {
649        // TODO: Should have been just `::new()`, but https://github.com/rust-lang/rust/issues/53827
650        // SAFETY: Data structure filled with zeroes is a valid invariant
651        unsafe { Box::<Self>::new_zeroed().assume_init() }
652    }
