292#[cfg(feature = "alloc")]
293impl SuperSegment {
294    /// Create a new instance and derive super segment root.
295    ///
296    /// Returns `None` if the list of segment roots is empty or there are too many of them.
297    pub fn new(
298        previous_header: &SuperSegmentHeader,
299        target_beacon_chain_block_number: BlockNumber,
300        segment_roots: StdArc<[ShardSegmentRootWithPosition]>,
301    ) -> Option<Self> {
302        let num_segments = u32::try_from(segment_roots.len()).ok()?;
303        let max_segment_index = SegmentIndex::from(
304            u64::from(previous_header.max_segment_index.as_inner()) + u64::from(num_segments),
305        );
306
307        // TODO: This is a workaround for https://github.com/rust-lang/rust/issues/139866 that
308        //  allows the code to compile. Constant 1_048_575 is hardcoded here and below for
309        //  compilation to succeed.
310        const {
311            assert!(SuperSegmentRoot::MAX_SEGMENTS == 1_048_575);
312        }
313        // TODO: Keyed hash
314        let maybe_super_segment_root = UnbalancedMerkleTree::compute_root_only::<1_048_575, _, _>(
315            segment_roots.iter().map(ShardSegmentRootWithPosition::hash),
316        )?;
317
318        Some(Self {
319            header: SuperSegmentHeader {
320                index: (previous_header.index.as_inner() + SuperSegmentIndex::ONE).into(),
321                root: SuperSegmentRoot::from(maybe_super_segment_root),
322                prev_super_segment_header_hash: Blake3Hash::from(
323                    single_chunk_hash(previous_header.as_bytes())
324                        .expect("Less than a single chunk worth of bytes; qed"),
325                ),
326                max_segment_index: max_segment_index.into(),
327                target_beacon_chain_block_number: target_beacon_chain_block_number.into(),
328                num_segments,
329            },
330            segment_roots,
331        })
332    }
333
334    /// Produce a proof for a segment in the super segment at a given position
335    pub fn proof_for_segment(&self, segment_position: SegmentPosition) -> Option<SegmentProof> {
336        // TODO: This is a workaround for https://github.com/rust-lang/rust/issues/139866 that
337        //  allows the code to compile. Constant 1_048_575 is hardcoded here and below for
338        //  compilation to succeed.
339        const {
340            assert!(SuperSegmentRoot::MAX_SEGMENTS == 1_048_575);
341        }
342        // TODO: Keyed hash
343        let mut segment_proof = SegmentProof::default();
344        UnbalancedMerkleTree::compute_root_and_proof_in::<1_048_575, _, _>(
345            self.segment_roots.iter().map(|shard_segment_root| {
346                single_block_hash(shard_segment_root.as_bytes())
347                    .expect("Less than a single block worth of bytes; qed")
348            }),
349            u32::from(segment_position) as usize,
350            segment_proof.as_uninit_repr(),
351        )?;
352
353        Some(segment_proof)
354    }
355}

515        (self.first_piece_index()..=self.last_piece_index())
516            .zip(&mut piece_indices)
517            .for_each(|(input, output)| {
518                *output = input;
519            });
520
521        piece_indices
522    }

631
632    /// Check whether a segment root is a part of the super segment
633    pub fn is_valid(
634        &self,
635        shard_index: ShardIndex,
636        local_segment_index: LocalSegmentIndex,
637        segment_position: SegmentPosition,
638        segment_proof: &SegmentProof,
639        num_segments: u32,
640        super_segment_root: &SuperSegmentRoot,
641    ) -> bool {
642        let shard_segment_root = ShardSegmentRootWithPosition {
643            shard_index,
644            segment_position,
645            local_segment_index,
646            segment_root: *self,
647        };
648        // The proof is fixed size and contains zero padding elements, which must be skipped for
649        // verification purposes
650        let segment_proof = segment_proof
651            .split_once(|hash| hash == &[0; _])
652            .map_or(segment_proof.as_slice(), |(before, _after)| before);
653        UnbalancedMerkleTree::verify(
654            super_segment_root,
655            segment_proof,
656            u64::from(segment_position),
657            shard_segment_root.hash(),
658            u64::from(num_segments),
659        )
660    }

713        )
714        .expect("Not zero; qed")
715    }
716
717    /// Segment index that corresponds to this history size
718    #[inline(always)]
719    pub fn segment_index(&self) -> SegmentIndex {
720        self.0
721    }
722
723    /// History size at which expiration check for a sector happens.
724    ///
725    /// Returns `None` on overflow.
726    #[inline(always)]
727    pub fn sector_expiration_check(&self, min_sector_lifetime: Self) -> Option<Self> {
728        self.as_non_zero_u64()
729            .checked_add(min_sector_lifetime.as_non_zero_u64().get())
730            .map(Self::new)
731    }
732}

734/// Progress of an archived block.
735#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd, Hash, TrivialType)]
736#[cfg_attr(feature = "scale-codec", derive(Encode, Decode))]
737#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
738#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
739#[repr(C)]
740pub struct ArchivedBlockProgress {
741    /// Number of partially archived bytes of a block, `0` for a full block
742    bytes: u32,
743}
744
745impl Default for ArchivedBlockProgress {
746    /// We assume a block can always fit into the segment initially, but it is definitely possible
747    /// to be transitioned into the partial state after some overflow checking.
748    #[inline(always)]
749    fn default() -> Self {
750        Self::new_complete()
751    }
752}
753
754impl ArchivedBlockProgress {
755    /// Block is archived fully
756    #[inline(always)]
757    pub const fn new_complete() -> Self {
758        Self { bytes: 0 }
759    }
760
761    /// Block is partially archived with the provided number of bytes
762    #[inline(always)]
763    pub const fn new_partial(new_partial: NonZeroU32) -> Self {
764        Self {
765            bytes: new_partial.get(),
766        }
767    }
768
769    /// Return the number of partially archived bytes if the progress is not complete
770    #[inline(always)]
771    pub const fn partial(&self) -> Option<NonZeroU32> {
772        NonZeroU32::new(self.bytes)
773    }
774}

776/// Last archived block
777#[derive(Debug, Copy, Clone, PartialEq, Eq, Ord, PartialOrd, Hash, TrivialType)]
778#[cfg_attr(feature = "scale-codec", derive(Encode, Decode))]
779#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
780#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
781#[repr(C)]
782pub struct LastArchivedBlock {
783    /// Block number
784    pub number: Unaligned<BlockNumber>,
785    /// Progress of an archived block
786    pub archived_progress: ArchivedBlockProgress,
787}
788
789impl LastArchivedBlock {
790    /// Returns the number of partially archived bytes for a block
791    #[inline(always)]
792    pub fn partial_archived(&self) -> Option<NonZeroU32> {
793        self.archived_progress.partial()
794    }
795
796    /// Sets the number of partially archived bytes if block progress was archived partially
797    #[inline(always)]
798    pub fn set_partial_archived(&mut self, new_partial: NonZeroU32) {
799        self.archived_progress = ArchivedBlockProgress::new_partial(new_partial);
800    }
801
802    /// Indicate the last archived block was archived fully
803    #[inline(always)]
804    pub fn set_complete(&mut self) {
805        self.archived_progress = ArchivedBlockProgress::new_complete();
806    }
807
808    /// Get the block number (unwrap `Unaligned`)
809    pub const fn number(&self) -> BlockNumber {
810        self.number.as_inner()
811    }
812}

814/// Segment header for a specific segment of a shard
815#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, TrivialType)]
816#[cfg_attr(feature = "scale-codec", derive(Encode, Decode))]
817#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
818#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
819#[repr(C)]
820pub struct SegmentHeader {
821    /// Local segment index
822    pub index: Unaligned<LocalSegmentIndex>,
823    /// Root of roots of all records in a segment.
824    pub root: SegmentRoot,
825    /// Hash of the segment header of the previous segment
826    pub prev_segment_header_hash: Blake3Hash,
827    /// Last archived block
828    pub last_archived_block: LastArchivedBlock,
829}
830
831impl SegmentHeader {
832    /// Hash of the whole segment header
833    #[inline(always)]
834    pub fn hash(&self) -> Blake3Hash {
835        const {
836            assert!(size_of::<Self>() <= CHUNK_LEN);
837        }
838        Blake3Hash::new(
839            single_chunk_hash(self.as_bytes())
840                .expect("Less than a single chunk worth of bytes; qed"),
841        )
842    }
843}

845/// Recorded history segment before archiving is applied.
846///
847/// NOTE: This is a stack-allocated data structure and can cause stack overflow!
848#[derive(Copy, Clone, Eq, PartialEq, Deref, DerefMut)]
849#[repr(C)]
850pub struct RecordedHistorySegment([Record; Self::NUM_RAW_RECORDS]);
851
852impl fmt::Debug for RecordedHistorySegment {
853    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
854        f.debug_struct("RecordedHistorySegment")
855            .finish_non_exhaustive()
856    }
857}
858
859impl AsRef<[u8]> for RecordedHistorySegment {
860    #[inline]
861    fn as_ref(&self) -> &[u8] {
862        Record::slice_to_repr(&self.0).as_flattened().as_flattened()
863    }
864}
865
866impl AsMut<[u8]> for RecordedHistorySegment {
867    #[inline]
868    fn as_mut(&mut self) -> &mut [u8] {
869        Record::slice_mut_to_repr(&mut self.0)
870            .as_flattened_mut()
871            .as_flattened_mut()
872    }
873}
874
875impl RecordedHistorySegment {
876    /// Number of raw records in one segment of recorded history
877    pub const NUM_RAW_RECORDS: usize = 128;
878    /// Erasure coding rate for records during the archiving process
879    pub const ERASURE_CODING_RATE: (usize, usize) = (1, 2);
880    /// Number of pieces in one segment of archived history (taking erasure coding rate into
881    /// account)
882    pub const NUM_PIECES: usize =
883        Self::NUM_RAW_RECORDS * Self::ERASURE_CODING_RATE.1 / Self::ERASURE_CODING_RATE.0;
884    /// Size of recorded history segment in bytes.
885    ///
886    /// It includes half of the records (just source records) that will later be erasure coded and
887    /// together with corresponding roots and proofs will result in
888    /// [`Self::NUM_PIECES`] `Piece`s of archival history.
889    pub const SIZE: usize = Record::SIZE * Self::NUM_RAW_RECORDS;
890
891    /// Create boxed value without hitting stack overflow
892    #[inline]
893    #[cfg(feature = "alloc")]
894    pub fn new_boxed() -> Box<Self> {
895        // TODO: Should have been just `::new()`, but https://github.com/rust-lang/rust/issues/53827
896        // SAFETY: Data structure filled with zeroes is a valid invariant
897        unsafe { Box::<Self>::new_zeroed().assume_init() }
898    }
899}
