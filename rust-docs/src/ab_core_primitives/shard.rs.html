13/// Shard index
14#[derive(Debug, Display, Copy, Clone, Hash, Ord, PartialOrd, Eq, PartialEq, TrivialType)]
15#[cfg_attr(
16    feature = "scale-codec",
17    derive(Encode, Decode, TypeInfo, MaxEncodedLen)
18)]
19#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
20#[repr(C)]
21pub struct ShardIndex(u32);
22
23impl ShardIndex {
24    /// Beacon chain
25    pub const BEACON_CHAIN: Self = Self(0);
26    /// Max possible shard index
27    pub const MAX_SHARD_INDEX: u32 = Self::MAX_SHARDS.get() - 1;
28    /// Max possible number of shards
29    pub const MAX_SHARDS: NonZeroU32 = NonZeroU32::new(2u32.pow(20)).expect("Not zero; qed");
30    /// Max possible number of addresses per shard
31    pub const MAX_ADDRESSES_PER_SHARD: NonZeroU128 =
32        NonZeroU128::new(2u128.pow(108)).expect("Not zero; qed");
33
34    // TODO: Remove once traits work in const environment and `From` could be used
35    /// Create shard index from `u32`.
36    ///
37    /// Returns `None` if `shard_index > ShardIndex::MAX_SHARD_INDEX`
38    ///
39    /// This is typically only necessary for low-level code.
40    #[inline(always)]
41    pub const fn new(shard_index: u32) -> Option<Self> {
42        if shard_index > Self::MAX_SHARD_INDEX {
43            return None;
44        }
45
46        Some(Self(shard_index))
47    }
48
49    // TODO: Remove once traits work in const environment and `From` could be used
50    /// Convert shard index to `u32`.
51    ///
52    /// This is typically only necessary for low-level code.
53    #[inline(always)]
54    pub const fn as_u32(self) -> u32 {
55        self.0
56    }
57}
