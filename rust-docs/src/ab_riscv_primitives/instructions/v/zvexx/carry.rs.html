12/// RISC-V ZveXx carry/borrow arithmetic instruction.
13///
14/// All use the OP-V major opcode (0b101_0111) with OPIVV/OPIVX/OPIVI funct3.
15/// The `vm` encoding bit selects carry/borrow-in source or signals no carry:
16///
17/// - vadc:  funct6=0b010000, vm=0 (carry-in from v0, always)
18/// - vmadc: funct6=0b010001, vm=0 (carry-in from v0) or vm=1 (no carry-in)
19/// - vsbc:  funct6=0b010010, vm=0 (borrow-in from v0, always)
20/// - vmsbc: funct6=0b010011, vm=0 (borrow-in from v0) or vm=1 (no borrow-in)
21///
22/// vadc/vsbc produce SEW-wide data results written to vd.
23/// vmadc/vmsbc produce one mask bit per element (carry-out/borrow-out) written to vd.
24#[instruction]
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26#[rustfmt::skip]
27#[doc(hidden)]
28pub enum ZveXxCarryInstruction<Reg> {
29    // vadc: add with carry-in from v0, write SEW-wide result to vd (data register)
30
31    /// `vadc.vvm vd, vs2, vs1, v0`
32    VadcVvm { vd: VReg, vs2: VReg, vs1: VReg },
33    /// `vadc.vxm vd, vs2, rs1, v0`
34    VadcVxm { vd: VReg, vs2: VReg, rs1: Reg },
35    /// `vadc.vim vd, vs2, imm, v0`
36    VadcVim { vd: VReg, vs2: VReg, imm: i8 },
37
38    // vmadc: add and produce carry-out mask in vd
39    // vm=0: carry-in from v0;  vm=1: no carry-in (treat carry-in as zero)
40
41    /// `vmadc.vvm vd, vs2, vs1, v0` - with carry-in
42    VmadcVvm { vd: VReg, vs2: VReg, vs1: VReg },
43    /// `vmadc.vxm vd, vs2, rs1, v0` - with carry-in
44    VmadcVxm { vd: VReg, vs2: VReg, rs1: Reg },
45    /// `vmadc.vim vd, vs2, imm, v0` - with carry-in
46    VmadcVim { vd: VReg, vs2: VReg, imm: i8 },
47    /// `vmadc.vv vd, vs2, vs1` - no carry-in
48    VmadcVv  { vd: VReg, vs2: VReg, vs1: VReg },
49    /// `vmadc.vx vd, vs2, rs1` - no carry-in
50    VmadcVx  { vd: VReg, vs2: VReg, rs1: Reg },
51    /// `vmadc.vi vd, vs2, imm` - no carry-in
52    VmadcVi  { vd: VReg, vs2: VReg, imm: i8 },
53
54    // vsbc: subtract with borrow-in from v0, write SEW-wide result to vd (data register)
55    // No immediate form in the spec.
56
57    /// `vsbc.vvm vd, vs2, vs1, v0`
58    VsbcVvm  { vd: VReg, vs2: VReg, vs1: VReg },
59    /// `vsbc.vxm vd, vs2, rs1, v0`
60    VsbcVxm  { vd: VReg, vs2: VReg, rs1: Reg },
61
62    // vmsbc: subtract and produce borrow-out mask in vd
63    // vm=0: borrow-in from v0;  vm=1: no borrow-in
64
65    /// `vmsbc.vvm vd, vs2, vs1, v0` - with borrow-in
66    VmsbcVvm { vd: VReg, vs2: VReg, vs1: VReg },
67    /// `vmsbc.vxm vd, vs2, rs1, v0` - with borrow-in
68    VmsbcVxm { vd: VReg, vs2: VReg, rs1: Reg },
69    /// `vmsbc.vv vd, vs2, vs1` - no borrow-in
70    VmsbcVv  { vd: VReg, vs2: VReg, vs1: VReg },
71    /// `vmsbc.vx vd, vs2, rs1` - no borrow-in
72    VmsbcVx  { vd: VReg, vs2: VReg, rs1: Reg },
73}

75#[instruction]
76impl<Reg> const Instruction for ZveXxCarryInstruction<Reg>
77where
78    Reg: [const] Register,
79{
80    type Reg = Reg;
81
82    #[inline(always)]
83    fn try_decode(instruction: u32) -> Option<Self> {
84        let opcode = (instruction & 0b111_1111) as u8;
85
86        if opcode != 0b101_0111 {
87            None?;
88        }
89
90        let vd_bits = ((instruction >> 7) & 0x1f) as u8;
91        let funct3 = ((instruction >> 12) & 0b111) as u8;
92        let vs1_bits = ((instruction >> 15) & 0x1f) as u8;
93        let vs2_bits = ((instruction >> 20) & 0x1f) as u8;
94        let vm = ((instruction >> 25) & 1) as u8;
95        let funct6 = ((instruction >> 26) & 0b11_1111) as u8;
96
97        let vd = VReg::from_bits(vd_bits)?;
98        let vs2 = VReg::from_bits(vs2_bits)?;
99
100        match funct3 {
101            // OPIVV
102            0b000 => {
103                let vs1 = VReg::from_bits(vs1_bits)?;
104                match (funct6, vm) {
105                    (0b01_0000, 0) => Some(Self::VadcVvm { vd, vs2, vs1 }),
106                    (0b01_0001, 0) => Some(Self::VmadcVvm { vd, vs2, vs1 }),
107                    (0b01_0001, 1) => Some(Self::VmadcVv { vd, vs2, vs1 }),
108                    (0b01_0010, 0) => Some(Self::VsbcVvm { vd, vs2, vs1 }),
109                    (0b01_0011, 0) => Some(Self::VmsbcVvm { vd, vs2, vs1 }),
110                    (0b01_0011, 1) => Some(Self::VmsbcVv { vd, vs2, vs1 }),
111                    _ => None,
112                }
113            }
114            // OPIVX
115            0b100 => {
116                let rs1 = Reg::from_bits(vs1_bits)?;
117                match (funct6, vm) {
118                    (0b01_0000, 0) => Some(Self::VadcVxm { vd, vs2, rs1 }),
119                    (0b01_0001, 0) => Some(Self::VmadcVxm { vd, vs2, rs1 }),
120                    (0b01_0001, 1) => Some(Self::VmadcVx { vd, vs2, rs1 }),
121                    (0b01_0010, 0) => Some(Self::VsbcVxm { vd, vs2, rs1 }),
122                    (0b01_0011, 0) => Some(Self::VmsbcVxm { vd, vs2, rs1 }),
123                    (0b01_0011, 1) => Some(Self::VmsbcVx { vd, vs2, rs1 }),
124                    _ => None,
125                }
126            }
127            // OPIVI - only vadc and vmadc have immediate forms; vsbc/vmsbc do not
128            0b011 => {
129                let imm = (vs1_bits << 3).cast_signed() >> 3u8;
130                match (funct6, vm) {
131                    (0b01_0000, 0) => Some(Self::VadcVim { vd, vs2, imm }),
132                    (0b01_0001, 0) => Some(Self::VmadcVim { vd, vs2, imm }),
133                    (0b01_0001, 1) => Some(Self::VmadcVi { vd, vs2, imm }),
134                    _ => None,
135                }
136            }
137            _ => None,
138        }
139    }
140
141    #[inline(always)]
142    fn alignment() -> u8 {
143        align_of::<u32>() as u8
144    }
145
146    #[inline(always)]
147    fn size(&self) -> u8 {
148        size_of::<u32>() as u8
149    }
150}

152#[instruction]
153impl<Reg> fmt::Display for ZveXxCarryInstruction<Reg>
154where
155    Reg: fmt::Display,
156{
157    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
158        #[rustfmt::skip]
159        match self {
160            Self::VadcVvm  { vd, vs2, vs1 } => write!(f, "vadc.vvm {vd}, {vs2}, {vs1}, v0"),
161            Self::VadcVxm  { vd, vs2, rs1 } => write!(f, "vadc.vxm {vd}, {vs2}, {rs1}, v0"),
162            Self::VadcVim  { vd, vs2, imm } => write!(f, "vadc.vim {vd}, {vs2}, {imm}, v0"),
163            Self::VmadcVvm { vd, vs2, vs1 } => write!(f, "vmadc.vvm {vd}, {vs2}, {vs1}, v0"),
164            Self::VmadcVxm { vd, vs2, rs1 } => write!(f, "vmadc.vxm {vd}, {vs2}, {rs1}, v0"),
165            Self::VmadcVim { vd, vs2, imm } => write!(f, "vmadc.vim {vd}, {vs2}, {imm}, v0"),
166            Self::VmadcVv  { vd, vs2, vs1 } => write!(f, "vmadc.vv {vd}, {vs2}, {vs1}"),
167            Self::VmadcVx  { vd, vs2, rs1 } => write!(f, "vmadc.vx {vd}, {vs2}, {rs1}"),
168            Self::VmadcVi  { vd, vs2, imm } => write!(f, "vmadc.vi {vd}, {vs2}, {imm}"),
169            Self::VsbcVvm  { vd, vs2, vs1 } => write!(f, "vsbc.vvm {vd}, {vs2}, {vs1}, v0"),
170            Self::VsbcVxm  { vd, vs2, rs1 } => write!(f, "vsbc.vxm {vd}, {vs2}, {rs1}, v0"),
171            Self::VmsbcVvm { vd, vs2, vs1 } => write!(f, "vmsbc.vvm {vd}, {vs2}, {vs1}, v0"),
172            Self::VmsbcVxm { vd, vs2, rs1 } => write!(f, "vmsbc.vxm {vd}, {vs2}, {rs1}, v0"),
173            Self::VmsbcVv  { vd, vs2, vs1 } => write!(f, "vmsbc.vv {vd}, {vs2}, {vs1}"),
174            Self::VmsbcVx  { vd, vs2, rs1 } => write!(f, "vmsbc.vx {vd}, {vs2}, {rs1}"),
175        }
176    }
