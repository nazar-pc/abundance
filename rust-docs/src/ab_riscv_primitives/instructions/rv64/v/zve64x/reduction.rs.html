13/// RISC-V RV64 Zve64x integer reduction instruction.
14///
15/// Reduction operations take a vector source `vs2`, an initial scalar value in element 0 of `vs1`,
16/// and write the scalar result to element 0 of `vd`.
17///
18/// Single-width reductions use OPMVV (funct3=0b010). Widening reductions use OPIVV (funct3=0b000).
19#[instruction]
20#[derive(Debug, Clone, Copy, PartialEq, Eq)]
21#[rustfmt::skip]
22pub(super) enum Rv64Zve64xReductionInstruction<Reg> {
23    /// Sum reduction: `vredsum.vs vd, vs2, vs1, vm`
24    Vredsum { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
25    /// AND reduction: `vredand.vs vd, vs2, vs1, vm`
26    Vredand { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
27    /// OR reduction: `vredor.vs vd, vs2, vs1, vm`
28    Vredor { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
29    /// XOR reduction: `vredxor.vs vd, vs2, vs1, vm`
30    Vredxor { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
31    /// Unsigned minimum reduction: `vredminu.vs vd, vs2, vs1, vm`
32    Vredminu { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
33    /// Signed minimum reduction: `vredmin.vs vd, vs2, vs1, vm`
34    Vredmin { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
35    /// Unsigned maximum reduction: `vredmaxu.vs vd, vs2, vs1, vm`
36    Vredmaxu { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
37    /// Signed maximum reduction: `vredmax.vs vd, vs2, vs1, vm`
38    Vredmax { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
39    /// Widening unsigned sum reduction: `vwredsumu.vs vd, vs2, vs1, vm`
40    Vwredsumu { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
41    /// Widening signed sum reduction: `vwredsum.vs vd, vs2, vs1, vm`
42    Vwredsum { vd: VReg, vs2: VReg, vs1: VReg, vm: bool },
43    #[doc(hidden)]
44    #[expect(dead_code, reason = "Only used for `Reg` generic")]
45    Phantom(PhantomData<Reg>),
46}

48#[instruction]
49impl<Reg> const Instruction for Rv64Zve64xReductionInstruction<Reg>
50where
51    Reg: [const] Register<Type = u64>,
52{
53    type Reg = Reg;
54
55    #[inline(always)]
56    fn try_decode(instruction: u32) -> Option<Self> {
57        let opcode = (instruction & 0b111_1111) as u8;
58
59        // OP-V major opcode
60        if opcode != 0b1010111 {
61            None?;
62        }
63
64        let vd_bits = ((instruction >> 7) & 0x1f) as u8;
65        let funct3 = ((instruction >> 12) & 0b111) as u8;
66        let vs1_bits = ((instruction >> 15) & 0x1f) as u8;
67        let vs2_bits = ((instruction >> 20) & 0x1f) as u8;
68        let vm = ((instruction >> 25) & 1) != 0;
69        let funct6 = ((instruction >> 26) & 0b11_1111) as u8;
70
71        let vd = VReg::from_bits(vd_bits)?;
72        let vs2 = VReg::from_bits(vs2_bits)?;
73        let vs1 = VReg::from_bits(vs1_bits)?;
74
75        match funct3 {
76            // OPMVV: single-width integer reductions
77            0b010 => match funct6 {
78                0b000000 => Some(Self::Vredsum { vd, vs2, vs1, vm }),
79                0b000001 => Some(Self::Vredand { vd, vs2, vs1, vm }),
80                0b000010 => Some(Self::Vredor { vd, vs2, vs1, vm }),
81                0b000011 => Some(Self::Vredxor { vd, vs2, vs1, vm }),
82                0b000100 => Some(Self::Vredminu { vd, vs2, vs1, vm }),
83                0b000101 => Some(Self::Vredmin { vd, vs2, vs1, vm }),
84                0b000110 => Some(Self::Vredmaxu { vd, vs2, vs1, vm }),
85                0b000111 => Some(Self::Vredmax { vd, vs2, vs1, vm }),
86                _ => None,
87            },
88            // OPIVV: widening integer reductions
89            0b000 => match funct6 {
90                0b110000 => Some(Self::Vwredsumu { vd, vs2, vs1, vm }),
91                0b110001 => Some(Self::Vwredsum { vd, vs2, vs1, vm }),
92                _ => None,
93            },
94            _ => None,
95        }
96    }
97
98    #[inline(always)]
99    fn alignment() -> u8 {
100        size_of::<u32>() as u8
101    }
102
103    #[inline(always)]
104    fn size(&self) -> u8 {
105        size_of::<u32>() as u8
106    }
107}

109impl<Reg> fmt::Display for Rv64Zve64xReductionInstruction<Reg> {
110    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
111        #[rustfmt::skip]
112        match self {
113            Self::Vredsum { vd, vs2, vs1, vm } => write!(f, "vredsum.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
114            Self::Vredand { vd, vs2, vs1, vm } => write!(f, "vredand.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
115            Self::Vredor { vd, vs2, vs1, vm } => write!(f, "vredor.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
116            Self::Vredxor { vd, vs2, vs1, vm } => write!(f, "vredxor.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
117            Self::Vredminu { vd, vs2, vs1, vm } => write!(f, "vredminu.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
118            Self::Vredmin { vd, vs2, vs1, vm } => write!(f, "vredmin.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
119            Self::Vredmaxu { vd, vs2, vs1, vm } => write!(f, "vredmaxu.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
120            Self::Vredmax { vd, vs2, vs1, vm } => write!(f, "vredmax.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
121            Self::Vwredsumu { vd, vs2, vs1, vm } => write!(f, "vwredsumu.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
122            Self::Vwredsum { vd, vs2, vs1, vm } => write!(f, "vwredsum.vs {vd}, {vs2}, {vs1}{}", mask_suffix(vm)),
123            Self::Phantom(_) => unreachable!("Never constructed"),
124        }
125    }
126}

128/// Format mask suffix for display
129#[inline(always)]
130fn mask_suffix(vm: &bool) -> &'static str {
131    if *vm { "" } else { ", v0.t" }
