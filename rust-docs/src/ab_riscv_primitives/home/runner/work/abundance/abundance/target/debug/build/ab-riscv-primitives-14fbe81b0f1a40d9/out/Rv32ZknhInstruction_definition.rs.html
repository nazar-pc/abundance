2/// RISC-V RV32 Zknh instruction (SHA-256 and SHA-512 sigma/sum functions).
3///
4/// SHA-256 instructions take a single source register.
5/// SHA-512 instructions take two source registers because 64-bit operands must be split across two
6/// 32-bit registers on RV32. The register conventions differ by instruction and follow the RISC-V
7/// scalar crypto Sail model exactly:
8///
9/// - `sha512sig0l`, `sha512sig1l`: rs1 = LOW word, rs2 = HIGH word
10/// - `sha512sig0h`, `sha512sig1h`: rs1 = HIGH word, rs2 = LOW word
11/// - `sha512sum0r`, `sha512sum1r`: rs1 = LOW word, rs2 = HIGH word
12///
13/// For `sha512sum0r` and `sha512sum1r` the Sail pseudocode builds the 64-bit operand as
14/// `x[63:32] = X(rs2), x[31:0] = X(rs1)` (rs2 is the HIGH half) and writes the low 32 bits of the
15/// result to `rd`.
16#[repr(align(4))]
17#[repr(u16)]
18pub enum Rv32ZknhInstruction<Reg> {
19    Sha256Sig0 {
20        rs1: Reg,
21        ///Second register operand (placeholder, instruction doesn't have it, should be set to Reg::ZERO)
22        #[doc(hidden)]
23        rs2: Reg,
24        rd: Reg,
25    },
26    Sha256Sig1 {
27        rs1: Reg,
28        ///Second register operand (placeholder, instruction doesn't have it, should be set to Reg::ZERO)
29        #[doc(hidden)]
30        rs2: Reg,
31        rd: Reg,
32    },
33    Sha256Sum0 {
34        rs1: Reg,
35        ///Second register operand (placeholder, instruction doesn't have it, should be set to Reg::ZERO)
36        #[doc(hidden)]
37        rs2: Reg,
38        rd: Reg,
39    },
40    Sha256Sum1 {
41        rs1: Reg,
42        ///Second register operand (placeholder, instruction doesn't have it, should be set to Reg::ZERO)
43        #[doc(hidden)]
44        rs2: Reg,
45        rd: Reg,
46    },
47    Sha512Sig0h { rs1: Reg, rs2: Reg, rd: Reg },
48    Sha512Sig0l { rs1: Reg, rs2: Reg, rd: Reg },
49    Sha512Sig1h { rs1: Reg, rs2: Reg, rd: Reg },
50    Sha512Sig1l { rs1: Reg, rs2: Reg, rd: Reg },
51    Sha512Sum0r { rs1: Reg, rs2: Reg, rd: Reg },
52    Sha512Sum1r { rs1: Reg, rs2: Reg, rd: Reg },
