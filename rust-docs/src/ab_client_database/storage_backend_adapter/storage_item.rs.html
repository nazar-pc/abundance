19#[derive(Debug, thiserror::Error)]
20pub(crate) enum StorageItemError {
21    /// Buffer too small
22    #[error("Buffer too small: expected {expected}, actual {actual}")]
23    BufferTooSmall { expected: usize, actual: usize },
24    /// Buffer too large
25    #[error("Buffer too large, {extra_pages} extra pages were provided")]
26    BufferTooLarge { extra_pages: usize },
27    /// Need more bytes
28    #[error("Need {0} more bytes")]
29    NeedMoreBytes(usize),
30    /// Unknown storage item variant
31    #[error("Unknown storage item variant {0}")]
32    UnknownStorageItemVariant(u8),
33    /// Invalid data length
34    #[error("Invalid data length {data_type}: expected {expected}, actual {actual}")]
35    InvalidDataLength {
36        data_type: &'static str,
37        expected: usize,
38        actual: usize,
39    },
40    /// Invalid data alignment
41    #[error("Invalid data alignment {data_type}")]
42    InvalidDataAlignment { data_type: &'static str },
43    /// Checksum mismatch
44    #[error("Checksum mismatch: expected {expected}, actual {actual}")]
45    ChecksumMismatch {
46        expected: Blake3Hash,
47        actual: Blake3Hash,
48    },
49    /// Repeat checksum mismatch
50    #[error("Repeat checksum mismatch: expected {expected}, actual {actual}")]
51    RepeatChecksumMismatch {
52        expected: Blake3Hash,
53        actual: Blake3Hash,
54    },
55    /// Storage item checksum mismatch
56    #[error("Storage item checksum mismatch: expected {expected}, actual {actual}")]
57    StorageItemChecksumMismatch {
58        expected: Blake3Hash,
59        actual: Blake3Hash,
60    },
61    /// Invalid buffer contents
62    #[error("Invalid buffer contents")]
63    InvalidBufferContents,
64}

67pub(crate) struct StorageItemWriteResult<'a> {

76pub(crate) trait StorageItem: fmt::Debug + Send + Sync + Sized + 'static {
77    /// Total number of bytes
78    fn total_bytes(&self) -> usize;
79
80    /// Write a storage item to the provided buffer.
81    ///
82    /// The buffer is required to be aligned to 128-bit.
83    ///
84    /// Returns the storage item variant and the number of bytes written.
85    fn write<'a>(
86        &self,
87        buffer: &'a mut [MaybeUninit<u8>],
88    ) -> Result<StorageItemWriteResult<'a>, StorageItemError>;
89
90    /// The inverse of [`Self::write_to_pages()`]
91    fn read(variant: u8, buffer: &[u8]) -> Result<Self, StorageItemError>;
92}

100#[derive(Debug)]
101pub(super) struct StorageItemContainer<SI> {
102    pub(super) sequence_number: u64,
103    pub(super) storage_item: SI,
104}
105
106impl<SI> StorageItemContainer<SI> {
107    const fn prefix_size() -> usize {
108        // Sequence number + enum variant + storage item size + checksum
109        size_of::<u64>() + size_of::<u8>() + size_of::<u32>() + size_of::<Blake3Hash>()
110    }
111
112    const fn suffix_size() -> usize {
113        // Storage item checksum + repeat of prefix checksum
114        size_of::<Blake3Hash>() * 2
115    }
116}
117
118impl<SI> StorageItemContainer<SI>
119where
120    SI: StorageItem,
121{
122    /// Returns the number of pages necessary to write this storage item
123    pub(super) fn num_pages(&self) -> u32 {
124        let storage_item_size = self.storage_item.total_bytes();
125
126        // Align buffer used by storage item to 128 bytes
127        let prefix_size = Self::prefix_size().next_multiple_of(size_of::<u128>());
128
129        (prefix_size + storage_item_size + Self::suffix_size()).div_ceil(AlignedPage::SIZE) as u32
130    }
131
132    /// Write a storage item to the provided buffer of aligned pages.
133    ///
134    /// Successful write means all provided pages are fully initialized.
135    pub(super) fn write_to_pages(
136        &self,
137        buffer: &mut [MaybeUninit<AlignedPage>],
138    ) -> Result<(), StorageItemError> {
139        let buffer = AlignedPage::uninit_slice_mut_to_repr(buffer);
140        // SAFETY: Same size and alignment, all uninitialized
141        let buffer_bytes = unsafe {
142            mem::transmute::<
143                &mut [MaybeUninit<[u8; AlignedPage::SIZE]>],
144                &mut [[MaybeUninit<u8>; AlignedPage::SIZE]],
145            >(buffer)
146        };
147        let mut buffer = buffer_bytes.as_flattened_mut();
148
149        // Align buffer used by storage item to 128 bytes
150        let prefix_bytes = buffer
151            .split_off_mut(..Self::prefix_size().next_multiple_of(size_of::<u128>()))
152            .expect("Always fits one page; qed");
153        let prefix_bytes = &mut prefix_bytes[..Self::prefix_size()];
154
155        let StorageItemWriteResult {
156            storage_item_variant,
157            storage_item_bytes,
158            mut buffer,
159        } = self.storage_item.write(buffer)?;
160
161        let buffer_len = buffer.len();
162        let suffix_bytes = buffer
163            .split_off_mut(..Self::suffix_size())
164            .ok_or_else(|| StorageItemError::NeedMoreBytes(Self::suffix_size() - buffer_len))?;
165
166        let (before_checksum, checksum_bytes) =
167            prefix_bytes.split_at_mut(size_of::<u64>() + size_of::<u8>() + size_of::<u32>());
168        let (sequence_number_bytes, remainder) = before_checksum.split_at_mut(size_of::<u64>());
169        let (storage_item_variant_bytes, storage_item_size_bytes) =
170            remainder.split_at_mut(size_of::<u8>());
171
172        // Write prefix
173        sequence_number_bytes.write_copy_of_slice(&self.sequence_number.to_le_bytes());
174        storage_item_variant_bytes[0].write(storage_item_variant);
175        let storage_item_size = storage_item_bytes.len() as u32;
176        storage_item_size_bytes.write_copy_of_slice(&storage_item_size.to_le_bytes());
177        // SAFETY: Wrote to all components of `before_checksum` above
178        let checksum = single_block_hash(unsafe { before_checksum.assume_init_ref() })
179            .expect("Less than one block worth of data; qed");
180        checksum_bytes.write_copy_of_slice(&checksum);
181
182        let (storage_item_checksum_bytes, prefix_checksum_repeat_bytes) =
183            suffix_bytes.split_at_mut(size_of::<Blake3Hash>());
184
185        // Write suffix
186        let storage_item_checksum = *hash(storage_item_bytes).as_bytes();
187        storage_item_checksum_bytes.write_copy_of_slice(&storage_item_checksum);
188        prefix_checksum_repeat_bytes.write_copy_of_slice(&checksum);
189
190        if buffer.len() < AlignedPage::SIZE {
191            buffer.write_filled(0);
192        } else {
193            return Err(StorageItemError::BufferTooLarge {
194                extra_pages: buffer.len(),
195            });
196        }
197
198        Ok(())
199    }
200
201    /// The inverse of [`Self::write_to_pages()`]
202    pub(super) fn read_from_pages(pages: &[AlignedPage]) -> Result<Self, StorageItemError> {
203        let mut buffer = AlignedPage::slice_to_repr(pages).as_flattened();
204
205        // Align buffer used by storage item to 128 bytes
206        let prefix_bytes = buffer
207            .split_off(..Self::prefix_size().next_multiple_of(size_of::<u128>()))
208            .expect("Always fits one page; qed");
209        let prefix_bytes = &prefix_bytes[..Self::prefix_size()];
210
211        let (sequence_number_bytes, remainder) = prefix_bytes.split_at(size_of::<u64>());
212        let (storage_item_variant_bytes, remainder) = remainder.split_at(size_of::<u8>());
213        let (storage_item_size_bytes, checksum_bytes) = remainder.split_at(size_of::<u32>());
214        let checksum = Blake3Hash::new(
215            single_block_hash(&prefix_bytes[..prefix_bytes.len() - checksum_bytes.len()])
216                .expect("Less than one block worth of data; qed"),
217        );
218
219        if checksum.as_slice() != checksum_bytes {
220            return Err(StorageItemError::ChecksumMismatch {
221                expected: checksum,
222                actual: Blake3Hash::new(checksum_bytes.try_into().expect("Correct length; qed")),
223            });
224        }
225
226        let sequence_number = u64::from_le_bytes(
227            sequence_number_bytes
228                .try_into()
229                .expect("Correct length; qed"),
230        );
231        let storage_item_variant = storage_item_variant_bytes[0];
232        let storage_item_size = u32::from_le_bytes(
233            storage_item_size_bytes
234                .try_into()
235                .expect("Correct length; qed"),
236        );
237
238        let buffer_len = buffer.len();
239        let storage_item_bytes = buffer.split_off(..storage_item_size as usize).ok_or(
240            StorageItemError::NeedMoreBytes(buffer_len - storage_item_size as usize),
241        )?;
242
243        let buffer_len = buffer.len();
244        let suffix_bytes = buffer
245            .split_off(..Self::suffix_size())
246            .ok_or_else(|| StorageItemError::NeedMoreBytes(Self::suffix_size() - buffer_len))?;
247        let (storage_item_checksum_bytes, prefix_checksum_repeat_bytes) =
248            suffix_bytes.split_at(size_of::<Blake3Hash>());
249
250        if checksum.as_slice() != prefix_checksum_repeat_bytes {
251            return Err(StorageItemError::RepeatChecksumMismatch {
252                expected: checksum,
253                actual: Blake3Hash::new(
254                    prefix_checksum_repeat_bytes
255                        .try_into()
256                        .expect("Correct length; qed"),
257                ),
258            });
259        }
260
261        let storage_item_checksum = Blake3Hash::from(hash(storage_item_bytes));
262        if storage_item_checksum.as_slice() != storage_item_checksum_bytes {
263            return Err(StorageItemError::StorageItemChecksumMismatch {
264                expected: storage_item_checksum,
265                actual: Blake3Hash::new(
266                    storage_item_checksum_bytes
267                        .try_into()
268                        .expect("Correct length; qed"),
269                ),
270            });
271        }
272
273        let storage_item = SI::read(storage_item_variant, storage_item_bytes)?;
274
275        Ok(Self {
276            sequence_number,
277            storage_item,
278        })
279    }
