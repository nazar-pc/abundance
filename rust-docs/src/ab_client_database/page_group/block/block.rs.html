11#[derive(Debug, Copy, Clone, TrivialType)]
12#[repr(C)]
13struct SystemContractStatePrefix {
14    owner: Address,
15    contract: Address,
16    content_len: u32,
17    padding: [u8; 4],
18}

24#[derive(Debug)]
25pub(crate) struct StorageItemBlockBlock {
26    pub(crate) header: SharedAlignedBuffer,
27    pub(crate) body: SharedAlignedBuffer,
28    pub(crate) mmr_with_block: Arc<BlockMerkleMountainRange>,
29    pub(crate) system_contract_states: StdArc<[ContractSlotState]>,
30    // TODO: State, segment headers
31}

33impl StorageItemBlockBlock {
34    pub(super) fn total_bytes(&self) -> usize {
35        Self::total_bytes_inner(
36            self.header.len(),
37            self.body.len(),
38            self.mmr_with_block.as_bytes().len() as u32,
39            self.system_contract_states_len(),
40        )
41    }
42
43    fn system_contract_states_len(&self) -> u32 {
44        let mut len = 0u32;
45        for system_contract_state in self.system_contract_states.as_ref() {
46            len = len.next_multiple_of(size_of::<u64>() as u32);
47            len += SystemContractStatePrefix::SIZE;
48            len = len.next_multiple_of(size_of::<u128>() as u32);
49            len += system_contract_state.contents.len();
50        }
51        len
52    }
53
54    const fn total_bytes_inner(
55        header_len: u32,
56        body_len: u32,
57        mmr_len: u32,
58        system_contract_states_len: u32,
59    ) -> usize {
60        Self::prefix_size()
61            + Self::block_content_size(header_len, body_len, mmr_len, system_contract_states_len)
62    }
63
64    const fn prefix_size() -> usize {
65        // 4 lengths of header/block/mmr/num system contracts states
66        const PREFIX_SIZE: usize = size_of::<u32>() * 4;
67        const {
68            // Ensure always aligned to `u128`
69            assert!(PREFIX_SIZE == size_of::<u128>());
70        }
71        PREFIX_SIZE
72    }
73
74    const fn block_content_size(
75        header_len: u32,
76        body_len: u32,
77        mmr_len: u32,
78        system_contract_states_len: u32,
79    ) -> usize {
80        // Account for alignment
81        let len = (header_len as usize).next_multiple_of(size_of::<u128>())
82            + body_len as usize
83            + mmr_len as usize;
84        len.next_multiple_of(size_of::<u64>()) + system_contract_states_len as usize
85    }
86
87    pub(super) fn write(
88        &self,
89        mut buffer: &mut [MaybeUninit<u8>],
90    ) -> Result<usize, StorageItemError> {
91        // The layout here is as follows:
92        // * header length: u32 as aligned little-endian bytes
93        // * body length: u32 as aligned little-endian bytes
94        // * MMR with block length: u32 as aligned little-endian bytes
95        // * number of system contract states: u32 as aligned little-endian bytes
96        // * block header: naturally aligned to 16-bytes boundary
97        // * padding to 16-bytes boundary (if needed)
98        // * block body
99        // * MMR with block bytes
100        // * for each system contract state:
101        //   * padding to the 8-bytes boundary (if needed)
102        //   * prefix: SystemContractStatePrefix
103        //   * padding to the 16-bytes boundary (if needed)
104        //   * contents: slot contents bytes
105
106        let buffer_len = buffer.len();
107        let total_bytes = self.total_bytes();
108
109        if buffer_len < total_bytes {
110            return Err(StorageItemError::BufferTooSmall {
111                expected: total_bytes,
112                actual: buffer_len,
113            });
114        }
115
116        let header = self.header.as_slice();
117        let body = self.body.as_slice();
118        let mmr_with_block = self.mmr_with_block.as_bytes().as_slice();
119        let system_contract_states = self.system_contract_states.as_ref();
120        let mut written_len = 0usize;
121
122        // Write all lengths
123        {
124            let prefix_bytes = buffer
125                .split_off_mut(..Self::prefix_size())
126                .expect("Total length checked above; qed");
127            let (header_len, remainder) = prefix_bytes.split_at_mut(size_of::<u32>());
128            let (body_len, remainder) = remainder.split_at_mut(size_of::<u32>());
129            let (mmr_len, num_system_contract_states) = remainder.split_at_mut(size_of::<u32>());
130
131            header_len.write_copy_of_slice(&(header.len() as u32).to_le_bytes());
132            body_len.write_copy_of_slice(&(body.len() as u32).to_le_bytes());
133            mmr_len.write_copy_of_slice(&(mmr_with_block.len() as u32).to_le_bytes());
134            num_system_contract_states
135                .write_copy_of_slice(&(system_contract_states.len() as u32).to_le_bytes());
136
137            written_len += prefix_bytes.len();
138        }
139
140        // Write content bytes
141        {
142            let header_bytes = buffer
143                .split_off_mut(..header.len().next_multiple_of(size_of::<u128>()))
144                .expect("Total length checked above; qed");
145
146            // Sub-slice due to possible trailing alignment bytes
147            header_bytes[..header.len()].write_copy_of_slice(header);
148            header_bytes[header.len()..].write_filled(0);
149            written_len += header_bytes.len();
150        }
151        {
152            let body_bytes = buffer
153                .split_off_mut(..body.len())
154                .expect("Total length checked above; qed");
155
156            body_bytes.write_copy_of_slice(body);
157            written_len += body_bytes.len();
158        }
159        {
160            let mmr_raw_bytes = buffer
161                .split_off_mut(..mmr_with_block.len())
162                .expect("Total length checked above; qed");
163
164            mmr_raw_bytes.write_copy_of_slice(mmr_with_block);
165            written_len += mmr_raw_bytes.len();
166        }
167
168        // Alignment padding (if needed)
169        if !written_len.is_multiple_of(size_of::<u64>()) {
170            let new_written_len = written_len.next_multiple_of(size_of::<u64>());
171            buffer
172                .split_off_mut(..(new_written_len - written_len))
173                .expect("Total length checked above; qed")
174                .write_filled(0);
175            written_len = new_written_len;
176        }
177
178        for system_contract_state in system_contract_states {
179            // Alignment padding (if needed)
180            if !written_len.is_multiple_of(size_of::<u64>()) {
181                let new_written_len = written_len.next_multiple_of(size_of::<u64>());
182                buffer
183                    .split_off_mut(..(new_written_len - written_len))
184                    .expect("Total length checked above; qed")
185                    .write_filled(0);
186                written_len = new_written_len;
187            }
188
189            {
190                let prefix_bytes = buffer
191                    .split_off_mut(..size_of::<SystemContractStatePrefix>())
192                    .expect("Total length checked above; qed");
193                prefix_bytes.write_copy_of_slice(
194                    SystemContractStatePrefix {
195                        owner: system_contract_state.owner,
196                        contract: system_contract_state.contract,
197                        content_len: system_contract_state.contents.len(),
198                        padding: [0; _],
199                    }
200                    .as_bytes(),
201                );
202                written_len += prefix_bytes.len();
203            }
204
205            // Alignment padding (if needed)
206            if !written_len.is_multiple_of(size_of::<u128>()) {
207                let new_written_len = written_len.next_multiple_of(size_of::<u128>());
208                buffer
209                    .split_off_mut(..(new_written_len - written_len))
210                    .expect("Total length checked above; qed")
211                    .write_filled(0);
212                written_len = new_written_len;
213            }
214
215            {
216                let contents_bytes = buffer
217                    .split_off_mut(..system_contract_state.contents.len() as usize)
218                    .expect("Total length checked above; qed");
219                contents_bytes.write_copy_of_slice(system_contract_state.contents.as_slice());
220                written_len += contents_bytes.len();
221            }
222        }
223
224        Ok(total_bytes)
225    }

227    pub(super) fn read(mut buffer: &[u8]) -> Result<Self, StorageItemError> {
228        let buffer_len = buffer.len();
229        let prefix_bytes = buffer
230            .split_off(..Self::prefix_size())
231            .ok_or_else(|| StorageItemError::NeedMoreBytes(Self::prefix_size() - buffer_len))?;
232        let mut read_len = prefix_bytes.len();
233
234        let (header_len, remainder) = prefix_bytes.split_at(size_of::<u32>());
235        let (body_len, remainder) = remainder.split_at(size_of::<u32>());
236        let (mmr_len, num_system_contract_states) = remainder.split_at(size_of::<u32>());
237
238        // Read lengths
239        let header_len =
240            u32::from_le_bytes(header_len.try_into().expect("Correct length; qed")) as usize;
241        let body_len =
242            u32::from_le_bytes(body_len.try_into().expect("Correct length; qed")) as usize;
243        let mmr_len = u32::from_le_bytes(mmr_len.try_into().expect("Correct length; qed")) as usize;
244        let num_system_contract_states = u32::from_le_bytes(
245            num_system_contract_states
246                .try_into()
247                .expect("Correct length; qed"),
248        );
249
250        let header = {
251            let buffer_len = buffer.len();
252            let header_bytes = buffer
253                .split_off(..header_len.next_multiple_of(size_of::<u128>()))
254                .ok_or_else(|| {
255                    StorageItemError::NeedMoreBytes(
256                        header_len.next_multiple_of(size_of::<u128>()) - buffer_len,
257                    )
258                })?;
259            let header = SharedAlignedBuffer::from_bytes(&header_bytes[..header_len]);
260            read_len += header_bytes.len();
261            header
262        };
263
264        let body = {
265            let buffer_len = buffer.len();
266            let body_bytes = buffer
267                .split_off(..body_len)
268                .ok_or_else(|| StorageItemError::NeedMoreBytes(body_len - buffer_len))?;
269            let body = SharedAlignedBuffer::from_bytes(body_bytes);
270            read_len += body_bytes.len();
271            body
272        };
273
274        let mmr = {
275            let buffer_len = buffer.len();
276            let mmr_raw_bytes = buffer
277                .split_off(..mmr_len)
278                .ok_or_else(|| StorageItemError::NeedMoreBytes(mmr_len - buffer_len))?;
279
280            let mut mmr_bytes = MerkleMountainRangeBytes::default();
281
282            if mmr_bytes.len() != mmr_raw_bytes.len() {
283                return Err(StorageItemError::InvalidDataLength {
284                    data_type: "MerkleMountainRangeBytes",
285                    expected: mmr_bytes.len(),
286                    actual: mmr_raw_bytes.len(),
287                });
288            }
289
290            mmr_bytes.copy_from_slice(mmr_raw_bytes);
291
292            // SAFETY: Created using `BlockMerkleMountainRange::as_bytes()` and checked data
293            // integrity
294            let mmr = unsafe { BlockMerkleMountainRange::from_bytes(&mmr_bytes) };
295            read_len += mmr_raw_bytes.len();
296            *mmr
297        };
298
299        let mut system_contract_states =
300            StdArc::<[ContractSlotState]>::new_uninit_slice(num_system_contract_states as usize);
301
302        for system_contract_state in
303            // SAFETY: A single pointer and a single use
304            unsafe { StdArc::get_mut_unchecked(&mut system_contract_states) }
305        {
306            // Alignment padding (if needed)
307            if !read_len.is_multiple_of(size_of::<u64>()) {
308                let new_read_len = read_len.next_multiple_of(size_of::<u64>());
309                let buffer_len = buffer.len();
310                buffer.split_off(..(new_read_len - read_len)).ok_or(
311                    StorageItemError::NeedMoreBytes((new_read_len - read_len) - buffer_len),
312                )?;
313                read_len = new_read_len;
314            }
315
316            let prefix = {
317                let buffer_len = buffer.len();
318                let prefix_bytes = buffer
319                    .split_off(..size_of::<SystemContractStatePrefix>())
320                    .ok_or_else(|| {
321                        StorageItemError::NeedMoreBytes(
322                            size_of::<SystemContractStatePrefix>() - buffer_len,
323                        )
324                    })?;
325                // SAFETY: This is a local database, so anything that is read that passes checksum
326                // verification is valid
327                let prefix = unsafe {
328                    SystemContractStatePrefix::from_bytes(prefix_bytes).ok_or(
329                        StorageItemError::InvalidDataAlignment {
330                            data_type: "SystemContractStatePrefix",
331                        },
332                    )?
333                };
334                read_len += prefix_bytes.len();
335                prefix
336            };
337
338            // Alignment padding (if needed)
339            if !read_len.is_multiple_of(size_of::<u128>()) {
340                let new_read_len = read_len.next_multiple_of(size_of::<u128>());
341                let buffer_len = buffer.len();
342                buffer.split_off(..(new_read_len - read_len)).ok_or(
343                    StorageItemError::NeedMoreBytes((new_read_len - read_len) - buffer_len),
344                )?;
345                read_len = new_read_len;
346            }
347
348            let contents = {
349                let buffer_len = buffer.len();
350                let contents_bytes = buffer.split_off(..prefix.content_len as usize).ok_or(
351                    StorageItemError::NeedMoreBytes(prefix.content_len as usize - buffer_len),
352                )?;
353                let contents = SharedAlignedBuffer::from_bytes(contents_bytes);
354                read_len += contents_bytes.len();
355                contents
356            };
357
358            system_contract_state.write(ContractSlotState {
359                owner: prefix.owner,
360                contract: prefix.contract,
361                contents,
362            });
363        }
364
365        // SAFETY: Just initialized all entries
366        let system_contract_states = unsafe { system_contract_states.assume_init() };
367
368        Ok(Self {
369            header,
370            body,
371            mmr_with_block: Arc::new(mmr),
372            system_contract_states,
373        })
374    }
