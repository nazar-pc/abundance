25/// Storage items that are produced as the result of blocks being imported
26#[derive(Debug)]
27pub(crate) enum StorageItemBlock {
28    Block(StorageItemBlockBlock),
29    SegmentHeaders(StorageItemBlockSegmentHeaders),
30}
31
32impl StorageItem for StorageItemBlock {
33    #[inline(always)]
34    fn total_bytes(&self) -> usize {
35        match self {
36            Self::Block(block) => block.total_bytes(),
37            Self::SegmentHeaders(segment_headers) => segment_headers.total_bytes(),
38        }
39    }
40
41    #[inline(always)]
42    fn write<'a>(
43        &self,
44        buffer: &'a mut [MaybeUninit<u8>],
45    ) -> Result<StorageItemWriteResult<'a>, StorageItemError> {
46        let (variant, storage_item_size) = match self {
47            Self::Block(block) => (StorageItemBlockVariant::Block, block.write(buffer)?),
48            StorageItemBlock::SegmentHeaders(segment_headers) => (
49                StorageItemBlockVariant::SegmentHeaders,
50                segment_headers.write(buffer)?,
51            ),
52        };
53
54        let (storage_item_bytes, buffer) = buffer.split_at_mut(storage_item_size);
55        // SAFETY: Storage item bytes were just written to
56        let storage_item_bytes = unsafe { storage_item_bytes.assume_init_mut() };
57
58        Ok(StorageItemWriteResult {
59            storage_item_variant: variant as u8,
60            storage_item_bytes,
61            buffer,
62        })
63    }
64
65    #[inline(always)]
66    fn read(variant: u8, buffer: &[u8]) -> Result<Self, StorageItemError> {
67        let variant = StorageItemBlockVariant::from_repr(variant)
68            .ok_or(StorageItemError::UnknownStorageItemVariant(variant))?;
69
70        Ok(match variant {
71            StorageItemBlockVariant::Block => Self::Block(StorageItemBlockBlock::read(buffer)?),
72            StorageItemBlockVariant::SegmentHeaders => {
73                Self::SegmentHeaders(StorageItemBlockSegmentHeaders::read(buffer)?)
74            }
75        })
76    }
77}

80    #[inline(always)]

82        PageGroupKind::Block
