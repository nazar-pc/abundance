28#[repr(u8)]
29pub(crate) enum PageGroupKind {
30    /// These pages are stored permanently and are never removed
31    Permanent = 0,
32    /// These pages are related to blocks and expire over time as blocks become buried deeper in
33    /// blockchain history
34    Block = 1,
35}

37#[derive(Debug)]
38struct PageGroup {
39    /// Sequence number of the first storage item in this page group
40    first_sequence_number: u64,
41    /// Next page offset within the page group
42    inner_next_page_offset: u32,
43    /// Offset of the first page of this page group in the storage backend
44    first_page_offset: u32,
45}
46
47#[derive(Debug)]
48struct PageGroups {
49    /// Next sequence number to use
50    next_sequence_number: u64,
51    /// A list of page groups.
52    ///
53    /// The front page is the active one, meaning it is being appended to, the back page is the
54    /// oldest page.
55    ///
56    /// If pruning is needed, old pages are freed from back to front without gaps.
57    list: VecDeque<PageGroup>,
58}

107    pub(crate) async fn open<SIHP, SIHB, StorageBackend>(
108        write_buffer_size: usize,
109        mut storage_item_handlers: StorageItemHandlers<SIHP, SIHB>,
110        storage_backend: &StorageBackend,
111    ) -> Result<Self, ClientDatabaseError>
112    where
113        SIHP: FnMut(StorageItemHandlerArg<StorageItemPermanent>) -> Result<(), ClientDatabaseError>,
114        SIHB: FnMut(StorageItemHandlerArg<StorageItemBlock>) -> Result<(), ClientDatabaseError>,
115        StorageBackend: ClientDatabaseStorageBackend,
116    {
117        let database_id;
118        let database_version;
119        let page_group_size;
120        let num_page_groups;
121
122        let mut page_groups = enum_map! {
123            PageGroupKind::Permanent => PageGroups {
124                next_sequence_number: 0,
125                list: VecDeque::new(),
126            },
127            PageGroupKind::Block => PageGroups {
128                next_sequence_number: 0,
129                list: VecDeque::new(),
130            },
131        };
132        let mut free_page_groups = VecDeque::new();
133
134        let mut buffer = Vec::new();
135
136        // Check the first page group. This could have been done in the loop below, but that makes
137        // the code even more ugly than this copy-paste.
138        {
139            buffer = storage_backend
140                .read(buffer, 1, 0)
141                .await
142                .map_err(|_error| ClientDatabaseError::ReadRequestCancelled)?
143                .map_err(|error| ClientDatabaseError::ReadError { error })?;
144
145            let container =
146                match StorageItemContainer::<StorageItemPageGroupHeader>::read_from_pages(&buffer) {
147                    Ok(container) => container,
148                    Err(_error) => {
149                        // Page group header fit the first page, so any deciding error indicates it is
150                        // not a valid page group header
151                        return Err(ClientDatabaseError::Unformatted);
152                    }
153                };
154
155            let page_group_header = &container.storage_item;
156            if page_group_header.database_version != Self::VERSION {
157                return Err(ClientDatabaseError::UnsupportedDatabaseVersion {
158                    database_version: page_group_header.database_version,
159                });
160            }
161            database_id = page_group_header.database_id;
162            database_version = page_group_header.database_version;
163            page_group_size = page_group_header.page_group_size;
164            if page_group_size < 2 {
165                return Err(ClientDatabaseError::PageGroupSizeTooSmall { page_group_size });
166            }
167            num_page_groups = storage_backend.num_pages() / page_group_size;
168
169            match page_group_header.page_group_kind {
170                PageGroupKind::Permanent => {
171                    page_groups[PageGroupKind::Permanent]
172                        .list
173                        .push_front(PageGroup {
174                            first_sequence_number: container.sequence_number,
175                            inner_next_page_offset: container.num_pages(),
176                            first_page_offset: 0,
177                        });
178                }
179                PageGroupKind::Block => {
180                    return Err(ClientDatabaseError::NonPermanentFirstPageGroup);
181                }
182            }
183        }
184
185        // Quick scan through the rest of page groups
186        for page_group_index in 1..num_page_groups {
187            let first_page_offset = page_group_index * page_group_size;
188            buffer.clear();
189            buffer = storage_backend
190                .read(buffer, 1, first_page_offset)
191                .await
192                .map_err(|_error| ClientDatabaseError::ReadRequestCancelled)?
193                .map_err(|error| ClientDatabaseError::ReadError { error })?;
194
195            let container =
196                match StorageItemContainer::<StorageItemPageGroupHeader>::read_from_pages(&buffer) {
197                    Ok(container) => container,
198                    Err(_error) => {
199                        free_page_groups.push_back(first_page_offset);
200                        continue;
201                    }
202                };
203
204            let page_group_header = &container.storage_item;
205            if !(page_group_header.database_id == database_id
206                && page_group_header.database_version == database_version
207                && page_group_header.page_group_size == page_group_size)
208            {
209                free_page_groups.push_back(first_page_offset);
210                continue;
211            }
212
213            let page_group = PageGroup {
214                first_sequence_number: container.sequence_number,
215                inner_next_page_offset: container.num_pages(),
216                first_page_offset,
217            };
218            page_groups[page_group_header.page_group_kind]
219                .list
220                .push_front(page_group);
221        }
222
223        // Sort page groups into the correct order of first sequence numbers
224        for entry in page_groups.values_mut() {
225            entry
226                .list
227                .make_contiguous()
228                .sort_by_key(|page_group| Reverse(page_group.first_sequence_number));
229        }
230
231        // Read all permanent storage groups
232        buffer = StorageBackendAdapter::read_page_groups::<_, StorageItemPermanent, _>(
233            &mut page_groups[PageGroupKind::Permanent],
234            page_group_size,
235            storage_backend,
236            buffer,
237            |container, page_offset| {
238                (storage_item_handlers.permanent)(StorageItemHandlerArg {
239                    storage_item: container.storage_item,
240                    page_offset,
241                })
242            },
243        )
244        .instrument(info_span!("", page_group_kind = ?PageGroupKind::Permanent))
245        .await?;
246
247        // Read all block storage groups
248        let _ = StorageBackendAdapter::read_page_groups(
249            &mut page_groups[PageGroupKind::Block],
250            page_group_size,
251            storage_backend,
252            buffer,
253            |container, page_offset| {
254                (storage_item_handlers.block)(StorageItemHandlerArg {
255                    storage_item: container.storage_item,
256                    page_offset,
257                })
258            },
259        )
260        .instrument(info_span!("", page_group_kind = ?PageGroupKind::Block))
261        .await?;
262
263        Ok(Self {
264            database_id,
265            database_version,
266            page_group_size,
267            write_buffer: (0..write_buffer_size)
268                .map(|_| WriteBufferEntry::Free(Vec::new()))
269                .collect(),
270            page_groups,
271            free_page_groups,
272            had_write_failure: false,
273        })
274    }

276    pub(crate) async fn format<StorageBackend>(
277        storage_backend: &StorageBackend,
278        options: ClientDatabaseFormatOptions,
279    ) -> Result<(), ClientDatabaseFormatError>
280    where
281        StorageBackend: ClientDatabaseStorageBackend,
282    {
283        let mut buffer = Vec::with_capacity(1);
284
285        if !options.force {
286            buffer = storage_backend
287                .read(buffer, 1, 0)
288                .await
289                .map_err(|_error| ClientDatabaseFormatError::ReadRequestCancelled)?
290                .map_err(|error| ClientDatabaseFormatError::ReadError { error })?;
291            buffer.clear();
292
293            if StorageItemContainer::<StorageItemPageGroupHeader>::read_from_pages(&buffer).is_ok()
294            {
295                return Err(ClientDatabaseFormatError::AlreadyFormatted);
296            }
297        }
298
299        let container = StorageItemContainer {
300            sequence_number: 0,
301            storage_item: StorageItemPageGroupHeader {
302                database_id: DatabaseId::new({
303                    let mut id = [0; 32];
304                    OsRng.try_fill_bytes(&mut id)?;
305                    id
306                }),
307                database_version: Self::VERSION,
308                page_group_kind: PageGroupKind::Permanent,
309                padding: [0; _],
310                page_group_size: options.page_group_size.get(),
311            },
312        };
313        Self::write_pages_to_buffer(&container, None, &mut buffer)?;
314
315        let _buffer: Vec<AlignedPage> = storage_backend
316            .write(buffer, 0)
317            .await
318            .map_err(|_cancelled| ClientDatabaseFormatError::WriteRequestCancelled)??;
319
320        Ok(())
321    }

323    /// Read all page groups and call the storage item handler for every storage item except the
324    /// page group header
325    async fn read_page_groups<StorageBackend, SI, SIH>(
326        target_page_groups: &mut PageGroups,
327        page_group_size: u32,
328        storage_backend: &StorageBackend,
329        mut buffer: Vec<AlignedPage>,
330        mut storage_item_handler: SIH,
331    ) -> Result<Vec<AlignedPage>, ClientDatabaseError>
332    where
333        StorageBackend: ClientDatabaseStorageBackend,
334        SI: StorageItem,
335        SIH: FnMut(StorageItemContainer<SI>, u32) -> Result<(), ClientDatabaseError>,
336    {
337        let mut next_sequence_number = 0;
338
339        // Read all page groups from oldest to newest
340        for page_group in target_page_groups.list.iter_mut().rev() {
341            if next_sequence_number == 0 {
342                next_sequence_number = page_group.first_sequence_number;
343            }
344
345            buffer.clear();
346            buffer = storage_backend
347                .read(
348                    buffer,
349                    // Substraction accounts for the page group header, which was already read
350                    page_group_size - page_group.inner_next_page_offset,
351                    page_group.first_page_offset + page_group.inner_next_page_offset,
352                )
353                .await
354                .map_err(|_error| ClientDatabaseError::ReadRequestCancelled)?
355                .map_err(|error| ClientDatabaseError::ReadError { error })?;
356
357            // Account for the page group header that was already read
358            if page_group.first_sequence_number == next_sequence_number {
359                next_sequence_number += 1;
360            } else {
361                error!(
362                    actual = page_group.first_sequence_number,
363                    expected = next_sequence_number,
364                    "Unexpected first sequence number"
365                );
366                return Err(ClientDatabaseError::UnexpectedSequenceNumber {
367                    actual: page_group.first_sequence_number,
368                    expected: next_sequence_number,
369                    page_offset: page_group.first_page_offset,
370                });
371            }
372
373            let mut pages = buffer.as_slice();
374
375            while !pages.is_empty() {
376                let page_offset = page_group.first_page_offset + page_group.inner_next_page_offset;
377                let container = match StorageItemContainer::read_from_pages(pages) {
378                    Ok(container) => container,
379                    Err(error) => {
380                        debug!(
381                            page_offset,
382                            %error,
383                            "Failed to read storage item, considering this to be the end of the \
384                            page group"
385                        );
386                        break;
387                    }
388                };
389
390                let sequence_number = container.sequence_number;
391                let num_pages = container.num_pages();
392
393                if sequence_number == next_sequence_number {
394                    next_sequence_number += 1;
395                } else {
396                    error!(
397                        page_offset,
398                        actual = sequence_number,
399                        expected = next_sequence_number,
400                        "Unexpected sequence number"
401                    );
402                    return Err(ClientDatabaseError::UnexpectedSequenceNumber {
403                        actual: sequence_number,
404                        expected: next_sequence_number,
405                        page_offset,
406                    });
407                }
408
409                storage_item_handler(container, page_offset)?;
410
411                pages = &pages[num_pages as usize..];
412                page_group.inner_next_page_offset += num_pages;
413            }
414        }
415
416        target_page_groups.next_sequence_number = next_sequence_number;
417
418        Ok(buffer)
419    }

421    pub(super) async fn write_storage_item<StorageBackend, SI>(
422        &mut self,
423        storage_backend: &StorageBackend,
424        storage_item: SI,
425    ) -> io::Result<WriteLocation>
426    where
427        StorageBackend: ClientDatabaseStorageBackend,
428        SI: UniqueStorageItem,
429    {
430        if self.had_write_failure {
431            return Err(io::Error::new(
432                io::ErrorKind::Interrupted,
433                "Previous write operation failed, writes are not allowed until restart",
434            ));
435        }
436
437        self.write_storage_item_inner(storage_backend, storage_item)
438            .await
439            .inspect_err(|_error| {
440                self.had_write_failure = true;
441            })
442    }

444    async fn write_storage_item_inner<StorageBackend, SI>(
445        &mut self,
446        storage_backend: &StorageBackend,
447        storage_item: SI,
448    ) -> io::Result<WriteLocation>
449    where
450        StorageBackend: ClientDatabaseStorageBackend,
451        SI: UniqueStorageItem,
452    {
453        let page_group_kind = SI::page_group_kind();
454        let target_page_groups = &mut self.page_groups[page_group_kind];
455
456        let sequence_number = target_page_groups.next_sequence_number;
457        target_page_groups.next_sequence_number += 1;
458
459        let mut container = StorageItemContainer {
460            sequence_number,
461            storage_item,
462        };
463
464        let mut num_pages = container.num_pages();
465        // Ensure a storage item doesn't exceed page group size. `-1` accounts for the page group
466        // header.
467        if num_pages > (self.page_group_size - 1) {
468            return Err(io::Error::new(
469                io::ErrorKind::QuotaExceeded,
470                format!(
471                    "Storage item is too large: {num_pages} pages, max supported is {} pages",
472                    self.page_group_size
473                ),
474            ));
475        }
476
477        // Check if there is an active page group and whether it has enough free pages in it
478        let (active_page_group, maybe_page_group_header) = if let Some(page_group) =
479            target_page_groups.list.front_mut()
480            && let Some(remaining_pages_in_group) = self
481                .page_group_size
482                .checked_sub(page_group.inner_next_page_offset)
483            && remaining_pages_in_group >= num_pages
484        {
485            (page_group, None)
486        } else {
487            // Allocate a new page group
488            let first_page_offset = self.free_page_groups.pop_front().ok_or_else(|| {
489                io::Error::new(
490                    io::ErrorKind::StorageFull,
491                    "No free pages available to write a new storage item",
492                )
493            })?;
494
495            let page_group_header = StorageItemContainer {
496                sequence_number,
497                storage_item: StorageItemPageGroupHeader {
498                    database_id: self.database_id,
499                    database_version: self.database_version,
500                    page_group_kind,
501                    padding: [0; _],
502                    page_group_size: self.page_group_size,
503                },
504            };
505
506            // Adjust sequence numbers since the previous value was reused by a new page group
507            // header
508            target_page_groups.next_sequence_number += 1;
509            container.sequence_number += 1;
510            // Add a page that corresponds to the page group header
511            num_pages += 1;
512
513            let active_page_group = target_page_groups.list.push_front_mut(PageGroup {
514                first_sequence_number: sequence_number,
515                inner_next_page_offset: 0,
516                first_page_offset,
517            });
518
519            (active_page_group, Some(page_group_header))
520        };
521
522        let page_offset =
523            active_page_group.first_page_offset + active_page_group.inner_next_page_offset;
524        active_page_group.inner_next_page_offset += num_pages;
525
526        // In case buffering is disabled, allocate a buffer on demand and wait for write to
527        // finish
528        if self.write_buffer.is_empty() {
529            let mut buffer = Vec::new();
530
531            Self::write_pages_to_buffer(&container, maybe_page_group_header.as_ref(), &mut buffer)?;
532
533            let _buffer: Vec<_> = storage_backend
534                .write(buffer, page_offset)
535                .await
536                .map_err(|_cancelled| {
537                    io::Error::new(
538                        io::ErrorKind::Interrupted,
539                        "Storage backend write was aborted",
540                    )
541                })
542                .flatten()?;
543
544            return Ok(WriteLocation { page_offset });
545        }
546
547        let write_fut = future::poll_fn(|cx| {
548            // Find a free write buffer entry among those that are either completely free or already
549            // finished and can be reused
550            let write_attempt_result = self.write_buffer.iter_mut().find_map(|entry| {
551                replace_with_or_abort_and_return(entry, |entry| {
552                    let mut buffer = match entry {
553                        // Already free buffer
554                        WriteBufferEntry::Free(buffer) => buffer,
555                        WriteBufferEntry::Occupied(mut receiver) => {
556                            // Poll pending write attempt
557                            match receiver.poll_unpin(cx) {
558                                Poll::Ready(Ok(write_result)) => match write_result {
559                                    // Write succeeded, reuse buffer
560                                    Ok(mut buffer) => {
561                                        buffer.clear();
562
563                                        buffer
564                                    }
565                                    // Write failed
566                                    Err(error) => {
567                                        return (
568                                            Some(Err(error)),
569                                            WriteBufferEntry::Occupied(receiver),
570                                        );
571                                    }
572                                },
573                                // Write attempt was aborted
574                                Poll::Ready(Err(_cancelled)) => {
575                                    return (
576                                        Some(Err(io::Error::new(
577                                            io::ErrorKind::Interrupted,
578                                            "Storage backend write was aborted",
579                                        ))),
580                                        WriteBufferEntry::Occupied(receiver),
581                                    );
582                                }
583                                // Still in progress
584                                Poll::Pending => {
585                                    return (None, WriteBufferEntry::Occupied(receiver));
586                                }
587                            }
588                        }
589                    };
590
591                    // Write storage item pages to the buffer
592                    if let Err(error) = Self::write_pages_to_buffer(
593                        &container,
594                        maybe_page_group_header.as_ref(),
595                        &mut buffer,
596                    ) {
597                        buffer.clear();
598                        return (
599                            Some(Err(io::Error::other(error))),
600                            WriteBufferEntry::Free(buffer),
601                        );
602                    }
603
604                    let receiver = storage_backend.write(buffer, page_offset);
605                    (
606                        Some(Ok(WriteLocation { page_offset })),
607                        WriteBufferEntry::Occupied(receiver),
608                    )
609                })
610            });
611
612            match write_attempt_result {
613                Some(result) => Poll::Ready(result),
614                None => Poll::Pending,
615            }
616        });
617
618        write_fut.await
619    }

621    /// Write (append) a storage item with an optional page group header in front of it
622    #[inline(always)]
623    fn write_pages_to_buffer<SI>(
624        container: &StorageItemContainer<SI>,
625        maybe_page_group_header: Option<&StorageItemContainer<StorageItemPageGroupHeader>>,
626        buffer: &mut Vec<AlignedPage>,
627    ) -> io::Result<()>
628    where
629        SI: StorageItem,
630    {
631        if let Some(page_group_header) = maybe_page_group_header {
632            let length = container.num_pages() as usize + 1;
633            buffer.reserve(length);
634
635            let (header, buffer) = buffer.spare_capacity_mut()[..length].split_at_mut(1);
636            page_group_header
637                .write_to_pages(header)
638                .map_err(io::Error::other)?;
639            container.write_to_pages(buffer).map_err(io::Error::other)?;
640        } else {
641            let length = container.num_pages() as usize;
642            buffer.reserve(length);
643
644            container
645                .write_to_pages(&mut buffer.spare_capacity_mut()[..length])
646                .map_err(io::Error::other)?;
647        }
648
649        Ok(())
650    }
