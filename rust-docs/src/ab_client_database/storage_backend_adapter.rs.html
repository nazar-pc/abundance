14pub(crate) struct PageGroup {

16    /// Next page offset within the page group

18    /// Offset of the first page of this page group in the storage backend

22#[derive(Debug)]
23pub(crate) struct PageGroups {
24    /// Next sequence number to use
25    pub(crate) next_sequence_number: u64,
26    /// A list of page groups.
27    ///
28    /// The front page is the active one, meaning it is being appended to, the back page is the
29    /// oldest page.
30    ///
31    /// If pruning is needed, old pages are freed from back to front without gaps.
32    pub(crate) list: VecDeque<PageGroup>,
33}

82    pub(super) async fn write_storage_item<StorageBackend>(
83        &mut self,
84        page_group_kind: PageGroupKind,
85        storage_backend: &StorageBackend,
86        storage_item_kind: StorageItemKind,
87    ) -> io::Result<WriteLocation>
88    where
89        StorageBackend: ClientDatabaseStorageBackend,
90    {
91        if self.had_write_failure {
92            return Err(io::Error::new(
93                io::ErrorKind::Interrupted,
94                "Previous write operation failed, writes are not allowed until restart",
95            ));
96        }
97
98        self.write_storage_item_inner(page_group_kind, storage_backend, storage_item_kind)
99            .await
100            .inspect_err(|_error| {
101                self.had_write_failure = true;
102            })
103    }
104
105    async fn write_storage_item_inner<StorageBackend>(
106        &mut self,
107        page_group_kind: PageGroupKind,
108        storage_backend: &StorageBackend,
109        storage_item_kind: StorageItemKind,
110    ) -> io::Result<WriteLocation>
111    where
112        StorageBackend: ClientDatabaseStorageBackend,
113    {
114        let target_page_groups = &mut self.page_groups[page_group_kind];
115
116        let sequence_number = target_page_groups.next_sequence_number;
117        target_page_groups.next_sequence_number += 1;
118
119        let mut storage_item = StorageItem {
120            sequence_number,
121            storage_item_kind,
122        };
123
124        let mut num_pages = storage_item.num_pages();
125        // Ensure a storage item doesn't exceed page group size. `-1` accounts for the page group
126        // header.
127        if num_pages > (self.page_group_size - 1) {
128            return Err(io::Error::new(
129                io::ErrorKind::QuotaExceeded,
130                format!(
131                    "Storage item is too large: {num_pages} pages, max supported is {} pages",
132                    self.page_group_size
133                ),
134            ));
135        }
136
137        // Check if there is an active page group and whether it has enough free pages in it
138        let (active_page_group, maybe_page_group_header) = if let Some(page_group) =
139            target_page_groups.list.front_mut()
140            && let Some(remaining_pages_in_group) = self
141                .page_group_size
142                .checked_sub(page_group.inner_next_page_offset)
143            && remaining_pages_in_group >= num_pages
144        {
145            (page_group, None)
146        } else {
147            // Allocate a new page group
148            let first_page_offset = self.free_page_groups.pop_front().ok_or_else(|| {
149                io::Error::new(
150                    io::ErrorKind::StorageFull,
151                    "No free pages available to write a new storage item",
152                )
153            })?;
154
155            let page_group_header = StorageItem {
156                sequence_number,
157                storage_item_kind: StorageItemKind::PageGroupHeader(StorageItemPageGroupHeader {
158                    database_id: self.database_id,
159                    database_version: self.database_version,
160                    page_group_kind,
161                    padding: [0; _],
162                    page_group_size: self.page_group_size,
163                }),
164            };
165
166            // Adjust sequence numbers since the previous value was reused by a new page group
167            // header
168            target_page_groups.next_sequence_number += 1;
169            storage_item.sequence_number += 1;
170            // Add a page that corresponds to the page group header
171            num_pages += 1;
172
173            target_page_groups.list.push_front(PageGroup {
174                first_sequence_number: sequence_number,
175                inner_next_page_offset: 0,
176                first_page_offset,
177            });
178            let active_page_group = target_page_groups
179                .list
180                .front_mut()
181                .expect("Just inserted; qed");
182
183            (active_page_group, Some(page_group_header))
184        };
185
186        let page_offset =
187            active_page_group.first_page_offset + active_page_group.inner_next_page_offset;
188        active_page_group.inner_next_page_offset += num_pages;
189
190        // In case buffering is disabled, allocate a buffer on demand and wait for write to
191        // finish
192        if self.write_buffer.is_empty() {
193            let mut buffer = Vec::new();
194
195            Self::write_pages_to_buffer(
196                &storage_item,
197                maybe_page_group_header.as_ref(),
198                &mut buffer,
199            )?;
200
201            let _buffer: Vec<_> = storage_backend
202                .write(buffer, page_offset)
203                .await
204                .map_err(|_cancelled| {
205                    io::Error::new(
206                        io::ErrorKind::Interrupted,
207                        "Storage backend write was aborted",
208                    )
209                })
210                .flatten()?;
211
212            return Ok(WriteLocation { page_offset });
213        }
214
215        let write_fut = future::poll_fn(|cx| {
216            // Find a free write buffer entry among those that are either completely free or already
217            // finished and can be reused
218            let write_attempt_result = self.write_buffer.iter_mut().find_map(|entry| {
219                replace_with_or_abort_and_return(entry, |entry| {
220                    let mut buffer = match entry {
221                        // Already free buffer
222                        WriteBufferEntry::Free(buffer) => buffer,
223                        WriteBufferEntry::Occupied(mut receiver) => {
224                            // Poll pending write attempt
225                            match receiver.poll_unpin(cx) {
226                                Poll::Ready(Ok(write_result)) => match write_result {
227                                    // Write succeeded, reuse buffer
228                                    Ok(buffer) => buffer,
229                                    // Write failed
230                                    Err(error) => {
231                                        return (
232                                            Some(Err(error)),
233                                            WriteBufferEntry::Occupied(receiver),
234                                        );
235                                    }
236                                },
237                                // Write attempt was aborted
238                                Poll::Ready(Err(_cancelled)) => {
239                                    return (
240                                        Some(Err(io::Error::new(
241                                            io::ErrorKind::Interrupted,
242                                            "Storage backend write was aborted",
243                                        ))),
244                                        WriteBufferEntry::Occupied(receiver),
245                                    );
246                                }
247                                // Still in progress
248                                Poll::Pending => {
249                                    return (None, WriteBufferEntry::Occupied(receiver));
250                                }
251                            }
252                        }
253                    };
254
255                    // Resize buffer and write storage item pages
256                    if let Err(error) = Self::write_pages_to_buffer(
257                        &storage_item,
258                        maybe_page_group_header.as_ref(),
259                        &mut buffer,
260                    ) {
261                        return (
262                            Some(Err(io::Error::other(error))),
263                            WriteBufferEntry::Free(buffer),
264                        );
265                    }
266
267                    let receiver = storage_backend.write(buffer, page_offset);
268                    (
269                        Some(Ok(WriteLocation { page_offset })),
270                        WriteBufferEntry::Occupied(receiver),
271                    )
272                })
273            });
274
275            match write_attempt_result {
276                Some(result) => Poll::Ready(result),
277                None => Poll::Pending,
278            }
279        });
280
281        write_fut.await
282    }

284    /// Resize the buffer to the correct size and write storage item with optional prepended page
285    /// group header
286    #[inline(always)]
287    fn write_pages_to_buffer(
288        storage_item: &StorageItem,
289        maybe_page_group_header: Option<&StorageItem>,
290        buffer: &mut Vec<AlignedPage>,
291    ) -> io::Result<()> {
292        if let Some(page_group_header) = maybe_page_group_header {
293            buffer.resize_with(storage_item.num_pages() as usize + 1, AlignedPage::default);
294
295            let (header, buffer) = buffer.split_at_mut(1);
296            page_group_header
297                .write_to_pages(header)
298                .map_err(io::Error::other)?;
299            storage_item
300                .write_to_pages(buffer)
301                .map_err(io::Error::other)?;
302        } else {
303            buffer.resize_with(storage_item.num_pages() as usize, AlignedPage::default);
304
305            storage_item
306                .write_to_pages(buffer)
307                .map_err(io::Error::other)?;
308        }
309
310        Ok(())
311    }
