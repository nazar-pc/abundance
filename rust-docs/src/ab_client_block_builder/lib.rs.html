41/// Error for [`BlockBuilder`]
42#[derive(Debug, thiserror::Error)]
43pub enum BlockBuilderError {
44    /// Custom builder error
45    #[error("Custom builder error: {error}")]
46    Custom {
47        // Custom block builder error
48        #[from]
49        error: anyhow::Error,
50    },
51    /// Failed to seal the block
52    #[error("Failed to seal the block")]
53    FailedToSeal,
54    /// Received invalid seal
55    #[error(
56        "Received invalid seal for pre-seal hash {pre_seal_hash} and public key hash \
57        {public_key_hash}"
58    )]
59    InvalidSeal {
60        /// Public key hash
61        public_key_hash: Blake3Hash,
62        /// Pre-seal hash
63        pre_seal_hash: Blake3Hash,
64    },
65}

70    Block: GenericOwnedBlock,
71{
72    /// Build a new block using provided parameters
73    fn build<SealBlock, SealBlockFut>(
74        &mut self,
75        parent_block_root: &BlockRoot,
76        parent_header: &<Block::Header as GenericOwnedBlockHeader>::Header<'_>,
77        consensus_info: &BlockHeaderConsensusInfo,
78        checkpoints: &[PotCheckpoints],
79        seal_block: SealBlock,
80    ) -> impl Future<Output = Result<Block, BlockBuilderError>> + Send
81    where
82        SealBlock: FnOnce(Blake3Hash) -> SealBlockFut + Send,
83        SealBlockFut: Future<Output = Option<OwnedBlockHeaderSeal>> + Send;
