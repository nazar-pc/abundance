1//! This module contains generic utilities for serializing and deserializing method calls to/from
2//! payload bytes.
3//!
4//! It can be reused to implement a different wallet implementation as well as read and verify the
5//! contents of the transaction (for example, to display it on the screen of the hardware wallet).
6//!
7//! Builder interface requires heap allocations and can be enabled with `payload-builder` feature,
8//! while the rest works in `no_std` environment without a global allocator.

27#[derive(Debug, Copy, Clone, Eq, PartialEq, TrivialType)]
28#[repr(u8)]
29pub enum TransactionMethodContext {
30    /// Call contract under [`Address::NULL`] context (corresponds to [`MethodContext::Reset`])
31    Null,
32    /// Call contract under context of the wallet (corresponds to [`MethodContext::Replace`])
33    Wallet,
34}
35
36impl TransactionMethodContext {
37    // TODO: Implement `TryFrom` once it is available in const environment
38    /// Try to create an instance from its `u8` representation
39    pub const fn try_from_u8(n: u8) -> Option<Self> {
40        Some(match n {
41            0 => Self::Null,
42            1 => Self::Wallet,
43            _ => {
44                return None;
45            }
46        })
47    }
48}

50#[derive(Debug, Copy, Eq, PartialEq, Clone)]
51pub enum TransactionInputType {
52    Value { alignment_power: u8 },
53    OutputIndex { output_index: u8 },
54}
55
56/// The type of transaction input could be either explicit value or output index.
57///
58/// Specifically, if the previous method has `#[output]` or return value, those values are collected
59/// and pushed into a virtual "stack". Then, if [`Self::input_type()`] returns
60/// [`TransactionInputType::OutputIndex`], then the corresponding input will use the value at
61/// `output_index` of this stack instead of what was specified in `external_args`. This allows
62/// composing calls to multiple methods into more sophisticated workflows without writing special
63/// contracts for this.
64#[derive(Debug, Copy, Clone)]
65pub struct TransactionInput(TransactionInputType);
66
67impl TransactionInput {
68    /// Regular input value with specified alignment.
69    ///
70    /// Valid alignment values are: 1, 2, 4, 8, 16.
71    pub const fn new_value(alignment: NonZeroU8) -> Option<Self> {
72        match alignment.get() {
73            1 | 2 | 4 | 8 | 16 => Some(Self(TransactionInputType::Value {
74                alignment_power: alignment.ilog2() as u8,
75            })),
76            _ => None,
77        }
78    }
79
80    /// Output index value.
81    ///
82    /// Valid index values are 0..=127.
83    pub const fn new_output_index(output_index: u8) -> Option<Self> {
84        if output_index > 0b0111_1111 {
85            return None;
86        }
87
88        Some(Self(TransactionInputType::OutputIndex { output_index }))
89    }
90
91    /// Create an instance from `u8`
92    pub const fn from_u8(n: u8) -> Self {
93        // The first bit is set to 1 for value and 0 for output index
94        if n & 0b1000_0000 == 0 {
95            Self(TransactionInputType::OutputIndex { output_index: n })
96        } else {
97            Self(TransactionInputType::Value {
98                alignment_power: n & 0b0111_1111,
99            })
100        }
101    }
102
103    /// Convert instance into `u8`
104    pub const fn into_u8(self) -> u8 {
105        // The first bit is set to 1 for value and 0 for output index
106        match self.0 {
107            TransactionInputType::Value { alignment_power } => 0b1000_0000 | alignment_power,
108            TransactionInputType::OutputIndex { output_index } => output_index,
109        }
110    }
111
112    /// Returns `Some(output_index)` or `None` if regular input value
113    pub const fn input_type(self) -> TransactionInputType {
114        self.0
115    }
116}

118/// Errors for [`TransactionPayloadDecoder`]
119#[derive(Debug, thiserror::Error)]
120pub enum TransactionPayloadDecoderError {
121    /// Payload too small
122    #[error("Payload too small")]
123    PayloadTooSmall,
124    /// `ExternalArgs` buffer too small
125    #[error("`ExternalArgs` buffer too small")]
126    ExternalArgsBufferTooSmall,
127    /// Output index not found
128    #[error("Output index not found: {0}")]
129    OutputIndexNotFound(u8),
130    /// Output buffer too small
131    #[error("Output buffer too small")]
132    OutputBufferTooSmall,
133    /// Output buffer offsets too small
134    #[error("Output buffer offsets too small")]
135    OutputBufferOffsetsTooSmall,
136}

138/// Decoder for transaction payload created using `TransactionPayloadBuilder`.
139pub struct TransactionPayloadDecoder<'a> {
140    payload: &'a [u8],
141    external_args_buffer: &'a mut [*mut c_void],
142    output_buffer: &'a mut [MaybeUninit<u128>],
143    output_buffer_cursor: usize,
144    output_buffer_offsets: SliceVec<'a, (u32, u32)>,
145    map_context: fn(TransactionMethodContext) -> MethodContext,
146}
147
148impl<'a> TransactionPayloadDecoder<'a> {
149    /// Create new instance.
150    ///
151    /// The size of `external_args_buffer` defines max number of bytes allocated for `ExternalArgs`,
152    /// which impacts the number of arguments that can be represented by `ExternalArgs`. The size is
153    /// specified in pointers with `#[slot]` argument using one pointer, `#[input]` two pointers and
154    /// `#[output]` three pointers each.
155    ///
156    /// The size of `output_buffer` defines how big the total size of `#[output]` and return values
157    /// could be in all methods of the payload together.
158    ///
159    /// The size of `output_buffer_offsets` defines how many `#[output]` arguments and return values
160    /// could exist in all methods of the payload together.
161    #[inline]
162    pub fn new(
163        payload: &'a [u128],
164        external_args_buffer: &'a mut [*mut c_void],
165        output_buffer: &'a mut [MaybeUninit<u128>],
166        output_buffer_offsets: &'a mut [(u32, u32)],
167        map_context: fn(TransactionMethodContext) -> MethodContext,
168    ) -> Self {
169        debug_assert_eq!(align_of_val(payload), usize::from(MAX_ALIGNMENT));
170        debug_assert_eq!(align_of_val(output_buffer), usize::from(MAX_ALIGNMENT));
171
172        // SAFETY: Memory is valid and bound by argument's lifetime
173        let payload =
174            unsafe { slice::from_raw_parts(payload.as_ptr().cast::<u8>(), size_of_val(payload)) };
175
176        let mut output_buffer_offsets = SliceVec::from(output_buffer_offsets);
177        output_buffer_offsets.clear();
178
179        Self {
180            payload,
181            external_args_buffer,
182            output_buffer,
183            output_buffer_cursor: 0,
184            output_buffer_offsets,
185            map_context,
186        }
187    }
188}

190impl<'a> TransactionPayloadDecoder<'a> {
191    /// Decode the next method (if any) in the payload
192    pub fn decode_next_method(
193        &mut self,
194    ) -> Result<Option<PreparedMethod<'_>>, TransactionPayloadDecoderError> {
195        TransactionPayloadDecoderInternal::<true>(self).decode_next_method()
196    }
197
198    /// Decode the next method (if any) in the payload without checking size.
199    ///
200    /// # Safety
201    /// Must be used with trusted input created using `TransactionPayloadBuilder` or pre-verified
202    /// using [`Self::decode_next_method()`] earlier.
203    pub unsafe fn decode_next_method_unchecked(&mut self) -> Option<PreparedMethod<'_>> {
204        TransactionPayloadDecoderInternal::<false>(self)
205            .decode_next_method()
206            .expect("No decoding errors are possible with trusted input; qed")
207    }
208}

210/// # Safety
211/// When `VERIFY == false` input must be trusted and created using `TransactionPayloadBuilder` or
212/// pre-verified using `VERIFY == true` earlier.
213struct TransactionPayloadDecoderInternal<'tmp, 'decoder, const VERIFY: bool>(
214    &'tmp mut TransactionPayloadDecoder<'decoder>,
215);
216
217impl<'tmp, 'decoder, const VERIFY: bool> Deref
218    for TransactionPayloadDecoderInternal<'tmp, 'decoder, VERIFY>
219{
220    type Target = TransactionPayloadDecoder<'decoder>;
221
222    #[inline(always)]
223    fn deref(&self) -> &Self::Target {
224        self.0
225    }
226}

237impl<'tmp, 'decoder, const VERIFY: bool> TransactionPayloadDecoderInternal<'tmp, 'decoder, VERIFY> {
238    #[inline(always)]
239    fn decode_next_method(
240        mut self,
241    ) -> Result<Option<PreparedMethod<'decoder>>, TransactionPayloadDecoderError> {
242        if self.payload.len() <= usize::from(MAX_ALIGNMENT) {
243            return Ok(None);
244        }
245
246        let contract = self.get_trivial_type::<Address>()?;
247        let method_fingerprint = self.get_trivial_type::<MethodFingerprint>()?;
248        let method_context =
249            (self.map_context)(*self.get_trivial_type::<TransactionMethodContext>()?);
250        let num_slot_arguments = self.read_u8()?;
251        let num_input_arguments = self.read_u8()?;
252        let num_output_arguments = self.read_u8()?;
253
254        // Slot needs one address pointer, input needs pointers to data and size, output needs
255        // pointers to data, size and capacity
256        let expected_external_args_buffer_size = usize::from(num_slot_arguments)
257            + usize::from(num_input_arguments) * 2
258            + usize::from(num_output_arguments) * 3;
259        if expected_external_args_buffer_size > self.external_args_buffer.len() {
260            return Err(TransactionPayloadDecoderError::ExternalArgsBufferTooSmall);
261        }
262
263        let external_args =
264            NonNull::new(self.external_args_buffer.as_mut_ptr()).expect("Not null; qed");
265        {
266            let mut external_args_cursor = external_args;
267
268            for _ in 0..num_slot_arguments {
269                let slot = self.get_trivial_type::<Address>()?;
270                // SAFETY: Size of `self.external_args_buffer` checked above, buffer is correctly
271                // aligned
272                unsafe {
273                    external_args_cursor.cast::<*const Address>().write(slot);
274                    external_args_cursor = external_args_cursor.offset(1);
275                }
276            }
277
278            for _ in 0..num_input_arguments {
279                let (bytes, size) = match TransactionInput::from_u8(self.read_u8()?).input_type() {
280                    TransactionInputType::Value { alignment_power } => {
281                        let alignment = 2usize.pow(u32::from(alignment_power));
282
283                        let size = self.get_trivial_type::<u32>()?;
284                        let bytes = self.get_bytes(*size, alignment)?;
285
286                        (bytes, size)
287                    }
288                    TransactionInputType::OutputIndex { output_index } => {
289                        let (size_offset, output_offset) = if VERIFY {
290                            *self
291                                .output_buffer_offsets
292                                .get(usize::from(output_index))
293                                .ok_or(TransactionPayloadDecoderError::OutputIndexNotFound(
294                                    output_index,
295                                ))?
296                        } else {
297                            // SAFETY: Unchecked version, see struct description
298                            *unsafe {
299                                self.output_buffer_offsets
300                                    .get_unchecked(usize::from(output_index))
301                            }
302                        };
303
304                        // SAFETY: Offset was created as the result of writing value at the correct
305                        // offset into `output_buffer_offsets` earlier
306                        let size = unsafe {
307                            self.output_buffer
308                                .as_ptr()
309                                .byte_add(size_offset as usize)
310                                .cast::<u32>()
311                                .as_ref_unchecked()
312                        };
313                        // SAFETY: Offset was created as the result of writing value at the correct
314                        // offset into `output_buffer_offsets` earlier
315                        let bytes = unsafe {
316                            let bytes_ptr = self
317                                .output_buffer
318                                .as_ptr()
319                                .cast::<u8>()
320                                .add(output_offset as usize);
321
322                            slice::from_raw_parts(bytes_ptr, *size as usize)
323                        };
324
325                        (bytes, size)
326                    }
327                };
328
329                // SAFETY: Size of `self.external_args_buffer` checked above, buffer is correctly
330                // aligned
331                unsafe {
332                    external_args_cursor
333                        .cast::<*const u8>()
334                        .write(bytes.as_ptr());
335                    external_args_cursor = external_args_cursor.offset(1);
336
337                    external_args_cursor.cast::<*const u32>().write(size);
338                    external_args_cursor = external_args_cursor.offset(1);
339                }
340            }
341
342            for _ in 0..num_output_arguments {
343                let recommended_capacity = self.get_trivial_type::<u32>()?;
344                let alignment_power = *self.get_trivial_type::<u8>()?;
345                let alignment = 2usize.pow(u32::from(alignment_power));
346
347                let (size, data) = self.allocate_output_buffer(*recommended_capacity, alignment)?;
348
349                // SAFETY: Size of `self.external_args_buffer` checked above, buffer is correctly
350                // aligned
351                unsafe {
352                    external_args_cursor.cast::<*mut u8>().write(data.as_ptr());
353                    external_args_cursor = external_args_cursor.offset(1);
354
355                    external_args_cursor.cast::<*mut u32>().write(size.as_ptr());
356                    external_args_cursor = external_args_cursor.offset(1);
357
358                    external_args_cursor
359                        .cast::<*const u32>()
360                        .write(recommended_capacity);
361                    external_args_cursor = external_args_cursor.offset(1);
362                }
363            }
364        }
365
366        Ok(Some(PreparedMethod {
367            contract: *contract,
368            fingerprint: *method_fingerprint,
369            external_args: external_args.cast::<NonNull<c_void>>(),
370            method_context,
371            phantom: PhantomData,
372        }))
373    }

375    #[inline(always)]
376    fn get_trivial_type<T>(&mut self) -> Result<&'decoder T, TransactionPayloadDecoderError>
377    where
378        T: TrivialType,
379    {
380        self.ensure_alignment(align_of::<T>());
381
382        let bytes;
383        if VERIFY {
384            (bytes, self.payload) = self
385                .payload
386                .split_at_checked(size_of::<T>())
387                .ok_or(TransactionPayloadDecoderError::PayloadTooSmall)?;
388        } else {
389            // SAFETY: Unchecked version, see struct description
390            (bytes, self.payload) = unsafe { self.payload.split_at_unchecked(size_of::<T>()) };
391        }
392
393        // SAFETY: Correctly aligned bytes of correct size
394        let value_ref = unsafe { bytes.as_ptr().cast::<T>().as_ref().expect("Not null; qed") };
395
396        Ok(value_ref)
397    }
398
399    #[inline(always)]
400    fn get_bytes(
401        &mut self,
402        size: u32,
403        alignment: usize,
404    ) -> Result<&'decoder [u8], TransactionPayloadDecoderError> {
405        self.ensure_alignment(alignment);
406
407        let bytes;
408        if VERIFY {
409            (bytes, self.payload) = self
410                .payload
411                .split_at_checked(size as usize)
412                .ok_or(TransactionPayloadDecoderError::PayloadTooSmall)?;
413        } else {
414            // SAFETY: Unchecked version, see struct description
415            (bytes, self.payload) = unsafe { self.payload.split_at_unchecked(size as usize) };
416        }
417
418        Ok(bytes)
419    }
420
421    #[inline(always)]
422    fn read_u8(&mut self) -> Result<u8, TransactionPayloadDecoderError> {
423        let value;
424        if VERIFY {
425            (value, self.payload) = self
426                .payload
427                .split_at_checked(1)
428                .ok_or(TransactionPayloadDecoderError::PayloadTooSmall)?;
429        } else {
430            // SAFETY: Unchecked version, see struct description
431            (value, self.payload) = unsafe { self.payload.split_at_unchecked(1) };
432        }
433
434        Ok(value[0])
435    }

437    #[inline(always)]
438    fn ensure_alignment(&mut self, alignment: usize) {
439        // Optimized version of the following that expects `alignment` to be a power of 2:
440        // let unaligned_by = self.payload.len() % alignment;
441        let unaligned_by = self.payload.len() & (alignment - 1);
442        self.payload = &self.payload[unaligned_by..];
443    }
444
445    #[inline(always)]
446    fn allocate_output_buffer(
447        &mut self,
448        capacity: u32,
449        output_alignment: usize,
450    ) -> Result<(NonNull<u32>, NonNull<u8>), TransactionPayloadDecoderError> {
451        if VERIFY && self.output_buffer_offsets.len() == self.output_buffer_offsets.capacity() {
452            return Err(TransactionPayloadDecoderError::OutputBufferOffsetsTooSmall);
453        }
454
455        let (size_offset, size_ptr) = self
456            .allocate_output_buffer_ptr::<u32>(align_of::<u32>(), size_of::<u32>())
457            .ok_or(TransactionPayloadDecoderError::OutputBufferTooSmall)?;
458        let (output_offset, output_ptr) = self
459            .allocate_output_buffer_ptr(output_alignment, capacity as usize)
460            .ok_or(TransactionPayloadDecoderError::OutputBufferTooSmall)?;
461
462        self.output_buffer_offsets
463            .push((size_offset as u32, output_offset as u32));
464
465        Ok((size_ptr, output_ptr))
466    }
467
468    /// Returns `None` if output buffer is not large enough
469    #[inline(always)]
470    fn allocate_output_buffer_ptr<T>(
471        &mut self,
472        alignment: usize,
473        size: usize,
474    ) -> Option<(usize, NonNull<T>)> {
475        debug_assert!(alignment <= usize::from(MAX_ALIGNMENT));
476
477        // Optimized version of the following that expects `alignment` to be a power of 2:
478        // let unaligned_by = self.output_buffer_cursor % alignment;
479        let unaligned_by = self.output_buffer_cursor & (alignment - 1);
480        if VERIFY
481            && self.output_buffer_cursor + unaligned_by + size > size_of_val(self.output_buffer)
482        {
483            return None;
484        }
485
486        // SAFETY: Bounds and alignment checks are done above
487        let buffer_ptr = unsafe {
488            NonNull::new_unchecked(
489                self.output_buffer
490                    .as_mut_ptr()
491                    .byte_add(self.output_buffer_cursor + unaligned_by)
492                    .cast::<T>(),
493            )
494        };
495        let offset = self.output_buffer_cursor + unaligned_by;
496        self.output_buffer_cursor += unaligned_by + size;
497
498        Some((offset, buffer_ptr))
499    }
